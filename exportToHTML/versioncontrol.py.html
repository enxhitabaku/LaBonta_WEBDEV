56   def is_url(name):
57       # type: (Union[str, Text]) -> bool
58       """ 
59       Return true if the name looks like a URL. 
60       """
61       scheme = get_url_scheme(name)
62       if scheme is None:
63           return False
64       return scheme in ['http', 'https', 'file', 'ftp'] + vcs.all_schemes

67   def make_vcs_requirement_url(repo_url, rev, project_name, subdir=None):
68       # type: (str, str, str, Optional[str]) -> str
69       """ 
70       Return the URL for a VCS requirement. 
71    
72       Args: 
73         repo_url: the remote VCS url, with any needed VCS prefix (e.g. "git+"). 
74         project_name: the (unescaped) project name. 
75       """
76       egg_project_name = pkg_resources.to_filename(project_name)
77       req = '{}@{}#egg={}'.format(repo_url, rev, egg_project_name)
78       if subdir:
79           req += '&subdirectory={}'.format(subdir)
80   
81       return req

84   def call_subprocess(
85       cmd,  # type: Union[List[str], CommandArgs]
86       cwd=None,  # type: Optional[str]
87       extra_environ=None,  # type: Optional[Mapping[str, Any]]
88       extra_ok_returncodes=None,  # type: Optional[Iterable[int]]
89       log_failed_cmd=True  # type: Optional[bool]
90   ):
91       # type: (...) -> Text
92       """ 
93       Args: 
94         extra_ok_returncodes: an iterable of integer return codes that are 
95           acceptable, in addition to 0. Defaults to None, which means []. 
96         log_failed_cmd: if false, failed commands are not logged, 
97           only raised. 
98       """
99       if extra_ok_returncodes is None:
100          extra_ok_returncodes = []
101  
102      # log the subprocess output at DEBUG level.
103      log_subprocess = subprocess_logger.debug
104  
105      env = os.environ.copy()
106      if extra_environ:
107          env.update(extra_environ)
108  
109      # Whether the subprocess will be visible in the console.
110      showing_subprocess = True
111  
112      command_desc = format_command_args(cmd)
113      try:
114          proc = subprocess.Popen(
115              # Convert HiddenText objects to the underlying str.
116              reveal_command_args(cmd),
117              stdout=subprocess.PIPE,
118              stderr=subprocess.PIPE,
119              cwd=cwd
120          )
121          if proc.stdin:
122              proc.stdin.close()
123      except Exception as exc:
124          if log_failed_cmd:
125              subprocess_logger.critical(
126                  "Error %s while executing command %s", exc, command_desc,
127              )
128          raise
129      all_output = []
130      while True:
131          # The "line" value is a unicode string in Python 2.
132          line = None
133          if proc.stdout:
134              line = console_to_str(proc.stdout.readline())
135          if not line:
136              break
137          line = line.rstrip()
138          all_output.append(line + '\n')
139  
140          # Show the line immediately.
141          log_subprocess(line)
142      try:
143          proc.wait()
144      finally:
145          if proc.stdout:
146              proc.stdout.close()
147  
148      proc_had_error = (
149          proc.returncode and proc.returncode not in extra_ok_returncodes
150      )
151      if proc_had_error:
152          if not showing_subprocess and log_failed_cmd:
153              # Then the subprocess streams haven't been logged to the
154              # console yet.
155              msg = make_subprocess_output_error(
156                  cmd_args=cmd,
157                  cwd=cwd,
158                  lines=all_output,
159                  exit_status=proc.returncode,
160              )
161              subprocess_logger.error(msg)
162          exc_msg = (
163              'Command errored out with exit status {}: {} '
164              'Check the logs for full command output.'
165          ).format(proc.returncode, command_desc)
166          raise SubProcessError(exc_msg)
167      return ''.join(all_output)

170  def find_path_to_setup_from_repo_root(location, repo_root):
171      # type: (str, str) -> Optional[str]
172      """ 
173      Find the path to `setup.py` by searching up the filesystem from `location`. 
174      Return the path to `setup.py` relative to `repo_root`. 
175      Return None if `setup.py` is in `repo_root` or cannot be found. 
176      """
177      # find setup.py
178      orig_location = location
179      while not os.path.exists(os.path.join(location, 'setup.py')):
180          last_location = location
181          location = os.path.dirname(location)
182          if location == last_location:
183              # We've traversed up to the root of the filesystem without
184              # finding setup.py
185              logger.warning(
186                  "Could not find setup.py for directory %s (tried all "
187                  "parent directories)",
188                  orig_location,
189              )
190              return None
191  
192      if samefile(repo_root, location):
193          return None
194  
195      return os.path.relpath(location, repo_root)

202  class RevOptions(object):
203  
204      """ 
205      Encapsulates a VCS-specific revision to install, along with any VCS 
206      install options. 
207   
208      Instances of this class should be treated as if immutable. 
209      """
210  
211      def __init__(
212          self,
213          vc_class,  # type: Type[VersionControl]
214          rev=None,  # type: Optional[str]
215          extra_args=None,  # type: Optional[CommandArgs]
216      ):
217          # type: (...) -> None
218          """ 
219          Args: 
220            vc_class: a VersionControl subclass. 
221            rev: the name of the revision to install. 
222            extra_args: a list of extra options. 
223          """
224          if extra_args is None:
225              extra_args = []
226  
227          self.extra_args = extra_args
228          self.rev = rev
229          self.vc_class = vc_class
230          self.branch_name = None  # type: Optional[str]
231  
232      def __repr__(self):
233          # type: () -> str
234          return '<RevOptions {}: rev={!r}>'.format(self.vc_class.name, self.rev)
235  
236      @property
237      def arg_rev(self):
238          # type: () -> Optional[str]
239          if self.rev is None:
240              return self.vc_class.default_arg_rev
241  
242          return self.rev
243  
244      def to_args(self):
245          # type: () -> CommandArgs
246          """ 
247          Return the VCS-specific command arguments. 
248          """
249          args = []  # type: CommandArgs
250          rev = self.arg_rev
251          if rev is not None:
252              args += self.vc_class.get_base_rev_args(rev)
253          args += self.extra_args
254  
255          return args
256  
257      def to_display(self):
258          # type: () -> str
259          if not self.rev:
260              return ''
261  
262          return ' (to revision {})'.format(self.rev)
263  
264      def make_new(self, rev):
265          # type: (str) -> RevOptions
266          """ 
267          Make a copy of the current instance, but with a new rev. 
268   
269          Args: 
270            rev: the name of the revision for the new object. 
271          """
272          return self.vc_class.make_rev_options(rev, extra_args=self.extra_args)

275  class VcsSupport(object):
276      _registry = {}  # type: Dict[str, VersionControl]
277      schemes = ['ssh', 'git', 'hg', 'bzr', 'sftp', 'svn']
278  
279      def __init__(self):
280          # type: () -> None
281          # Register more schemes with urlparse for various version control
282          # systems
283          urllib_parse.uses_netloc.extend(self.schemes)
284          # Python >= 2.7.4, 3.3 doesn't have uses_fragment
285          if getattr(urllib_parse, 'uses_fragment', None):
286              urllib_parse.uses_fragment.extend(self.schemes)
287          super(VcsSupport, self).__init__()
288  
289      def __iter__(self):
290          # type: () -> Iterator[str]
291          return self._registry.__iter__()
292  
293      @property
294      def backends(self):
295          # type: () -> List[VersionControl]
296          return list(self._registry.values())
297  
298      @property
299      def dirnames(self):
300          # type: () -> List[str]
301          return [backend.dirname for backend in self.backends]
302  
303      @property
304      def all_schemes(self):
305          # type: () -> List[str]
306          schemes = []  # type: List[str]
307          for backend in self.backends:
308              schemes.extend(backend.schemes)
309          return schemes
310  
311      def register(self, cls):
312          # type: (Type[VersionControl]) -> None
313          if not hasattr(cls, 'name'):
314              logger.warning('Cannot register VCS %s', cls.__name__)
315              return
316          if cls.name not in self._registry:
317              self._registry[cls.name] = cls()
318              logger.debug('Registered VCS backend: %s', cls.name)
319  
320      def unregister(self, name):
321          # type: (str) -> None
322          if name in self._registry:
323              del self._registry[name]
324  
325      def get_backend_for_dir(self, location):
326          # type: (str) -> Optional[VersionControl]
327          """ 
328          Return a VersionControl object if a repository of that type is found 
329          at the given directory. 
330          """
331          vcs_backends = {}
332          for vcs_backend in self._registry.values():
333              repo_path = vcs_backend.get_repository_root(location)
334              if not repo_path:
335                  continue
336              logger.debug('Determine that %s uses VCS: %s',
337                           location, vcs_backend.name)
338              vcs_backends[repo_path] = vcs_backend
339  
340          if not vcs_backends:
341              return None
342  
343          # Choose the VCS in the inner-most directory. Since all repository
344          # roots found here would be either `location` or one of its
345          # parents, the longest path should have the most path components,
346          # i.e. the backend representing the inner-most repository.
347          inner_most_repo_path = max(vcs_backends, key=len)
348          return vcs_backends[inner_most_repo_path]
349  
350      def get_backend_for_scheme(self, scheme):
351          # type: (str) -> Optional[VersionControl]
352          """ 
353          Return a VersionControl object or None. 
354          """
355          for vcs_backend in self._registry.values():
356              if scheme in vcs_backend.schemes:
357                  return vcs_backend
358          return None
359  
360      def get_backend(self, name):
361          # type: (str) -> Optional[VersionControl]
362          """ 
363          Return a VersionControl object or None. 
364          """
365          name = name.lower()
366          return self._registry.get(name)

374      dirname = ''

382      @classmethod
383      def should_add_vcs_url_prefix(cls, remote_url):
384          # type: (str) -> bool
385          """ 
386          Return whether the vcs prefix (e.g. "git+") should be added to a 
387          repository's remote url when used in a requirement. 
388          """
389          return not remote_url.lower().startswith('{}:'.format(cls.name))

396          Return None if setup.py is in the repo root. 
397          """
398          return None

400      @classmethod
401      def get_requirement_revision(cls, repo_dir):
402          # type: (str) -> str
403          """ 
404          Return the revision string that should be used in a requirement. 
405          """
406          return cls.get_revision(repo_dir)

408      @classmethod
409      def get_src_requirement(cls, repo_dir, project_name):
410          # type: (str, str) -> Optional[str]
411          """ 
412          Return the requirement string to use to redownload the files 
413          currently at the given repository directory. 
414   
415          Args: 
416            project_name: the (unescaped) project name. 
417   
418          The return value has a form similar to the following: 
419   
420              {repository_url}@{revision}#egg={project_name} 
421          """
422          repo_url = cls.get_remote_url(repo_dir)
423          if repo_url is None:
424              return None
425  
426          if cls.should_add_vcs_url_prefix(repo_url):
427              repo_url = '{}+{}'.format(cls.name, repo_url)
428  
429          revision = cls.get_requirement_revision(repo_dir)
430          subdir = cls.get_subdirectory(repo_dir)
431          req = make_vcs_requirement_url(repo_url, revision, project_name,
432                                         subdir=subdir)
433  
434          return req

493      @classmethod
494      def get_netloc_and_auth(cls, netloc, scheme):
495          # type: (str, str) -> Tuple[str, Tuple[Optional[str], Optional[str]]]
496          """ 
497          Parse the repository URL's netloc, and return the new netloc to use 
498          along with auth information. 
499   
500          Args: 
501            netloc: the original repository URL netloc. 
502            scheme: the repository URL's scheme without the vcs prefix. 
503   
504          This is mainly for the Subversion class to override, so that auth 
505          information can be provided via the --username and --password options 
506          instead of through the URL.  For other subclasses like Git without 
507          such an option, auth information must stay in the URL. 
508   
509          Returns: (netloc, (username, password)). 
510          """
511          return netloc, (None, None)
512  
513      @classmethod
514      def get_url_rev_and_auth(cls, url):
515          # type: (str) -> Tuple[str, Optional[str], AuthInfo]
516          """ 
517          Parse the repository URL to use, and return the URL, revision, 
518          and auth info to use. 
519   
520          Returns: (url, rev, (username, password)). 
521          """
522          scheme, netloc, path, query, frag = urllib_parse.urlsplit(url)
523          if '+' not in scheme:
524              raise ValueError(
525                  "Sorry, {!r} is a malformed VCS url. "
526                  "The format is <vcs>+<protocol>://<url>, "
527                  "e.g. svn+http://myrepo/svn/MyApp#egg=MyApp".format(url)
528              )
529          # Remove the vcs prefix.
530          scheme = scheme.split('+', 1)[1]
531          netloc, user_pass = cls.get_netloc_and_auth(netloc, scheme)
532          rev = None
533          if '@' in path:
534              path, rev = path.rsplit('@', 1)
535              if not rev:
536                  raise InstallationError(
537                      "The URL {!r} has an empty revision (after @) "
538                      "which is not supported. Include a revision after @ "
539                      "or remove @ from the URL.".format(url)
540                  )
541          url = urllib_parse.urlunsplit((scheme, netloc, path, query, ''))
542          return url, rev, user_pass

544      @staticmethod
545      def make_rev_args(username, password):
546          # type: (Optional[str], Optional[HiddenText]) -> CommandArgs

552      def get_url_rev_options(self, url):
553          # type: (HiddenText) -> Tuple[HiddenText, RevOptions]
554          """ 
555          Return the URL and RevOptions object to use in obtain() and in 
556          some cases export(), as a tuple (url, rev_options). 
557          """
558          secret_url, rev, user_pass = self.get_url_rev_and_auth(url.secret)
559          username, secret_password = user_pass
560          password = None  # type: Optional[HiddenText]
561          if secret_password is not None:
562              password = hide_value(secret_password)
563          extra_args = self.make_rev_args(username, password)
564          rev_options = self.make_rev_options(rev, extra_args=extra_args)
565  
566          return hide_url(secret_url), rev_options

629      def obtain(self, dest, url):
630          # type: (str, HiddenText) -> None
631          """ 
632          Install or update in editable mode the package represented by this 
633          VersionControl object. 
634   
635          :param dest: the repository directory in which to install or update. 
636          :param url: the repository URL starting with a vcs prefix. 
637          """
638          url, rev_options = self.get_url_rev_options(url)
639  
640          if not os.path.exists(dest):
641              self.fetch_new(dest, url, rev_options)
642              return
643  
644          rev_display = rev_options.to_display()
645          if self.is_repository_directory(dest):
646              existing_url = self.get_remote_url(dest)
647              if self.compare_urls(existing_url, url.secret):
648                  logger.debug(
649                      '%s in %s exists, and has correct URL (%s)',
650                      self.repo_name.title(),
651                      display_path(dest),
652                      url,
653                  )
654                  if not self.is_commit_id_equal(dest, rev_options.rev):
655                      logger.info(
656                          'Updating %s %s%s',
657                          display_path(dest),
658                          self.repo_name,
659                          rev_display,
660                      )
661                      self.update(dest, url, rev_options)
662                  else:
663                      logger.info('Skipping because already up-to-date.')
664                  return
665  
666              logger.warning(
667                  '%s %s in %s exists with URL %s',
668                  self.name,
669                  self.repo_name,
670                  display_path(dest),
671                  existing_url,
672              )
673              prompt = ('(s)witch, (i)gnore, (w)ipe, (b)ackup ',
674                        ('s', 'i', 'w', 'b'))
675          else:
676              logger.warning(
677                  'Directory %s already exists, and is not a %s %s.',
678                  dest,
679                  self.name,
680                  self.repo_name,
681              )
682              # https://github.com/python/mypy/issues/1174
683              prompt = ('(i)gnore, (w)ipe, (b)ackup ',  # type: ignore
684                        ('i', 'w', 'b'))
685  
686          logger.warning(
687              'The plan is to install the %s repository %s',
688              self.name,
689              url,
690          )
691          response = ask_path_exists('What to do?  {}'.format(
692              prompt[0]), prompt[1])
693  
694          if response == 'a':
695              sys.exit(-1)
696  
697          if response == 'w':
698              logger.warning('Deleting %s', display_path(dest))
699              rmtree(dest)
700              self.fetch_new(dest, url, rev_options)
701              return
702  
703          if response == 'b':
704              dest_dir = backup_dir(dest)
705              logger.warning(
706                  'Backing up %s to %s', display_path(dest), dest_dir,
707              )
708              shutil.move(dest, dest_dir)
709              self.fetch_new(dest, url, rev_options)
710              return
711  
712          # Do nothing if the response is "i".
713          if response == 's':
714              logger.info(
715                  'Switching %s %s to %s%s',
716                  self.repo_name,
717                  display_path(dest),
718                  url,
719                  rev_display,
720              )
721              self.switch(dest, url, rev_options)

723      def unpack(self, location, url):
724          # type: (str, HiddenText) -> None
725          """ 
726          Clean up current location and download the url repository 
727          (and vcs infos) into location 
728   
729          :param url: the repository URL starting with a vcs prefix. 
730          """
731          if os.path.exists(location):
732              rmtree(location)
733          self.obtain(location, url=url)

754      @classmethod
755      def run_command(
756          cls,
757          cmd,  # type: Union[List[str], CommandArgs]
758          cwd=None,  # type: Optional[str]
759          extra_environ=None,  # type: Optional[Mapping[str, Any]]
760          extra_ok_returncodes=None,  # type: Optional[Iterable[int]]
761          log_failed_cmd=True  # type: bool
762      ):
763          # type: (...) -> Text
764          """ 
765          Run a VCS subcommand 
766          This is simply a wrapper around call_subprocess that adds the VCS 
767          command name, and checks that the VCS is available 
768          """
769          cmd = make_command(cls.name, *cmd)
770          try:
771              return call_subprocess(cmd, cwd,
772                                     extra_environ=extra_environ,
773                                     extra_ok_returncodes=extra_ok_returncodes,
774                                     log_failed_cmd=log_failed_cmd)
775          except OSError as e:
776              # errno.ENOENT = no such file or directory
777              # In other words, the VCS executable isn't available
778              if e.errno == errno.ENOENT:
779                  raise BadCommand(
780                      'Cannot find command {cls.name!r} - do you have '
781                      '{cls.name!r} installed and in your '
782                      'PATH?'.format(**locals()))
783              else:
784                  raise  # re-raise exception if a different error occurred

786      @classmethod
787      def is_repository_directory(cls, path):
788          # type: (str) -> bool
789          """ 
790          Return whether a directory path is a repository directory. 
791          """
792          logger.debug('Checking in %s for %s (%s)...',
793                       path, cls.dirname, cls.name)
794          return os.path.exists(os.path.join(path, cls.dirname))

796      @classmethod
797      def get_repository_root(cls, location):
798          # type: (str) -> Optional[str]
799          """ 
800          Return the "root" (top-level) directory controlled by the vcs, 
801          or `None` if the directory is not in any. 
802   
803          It is meant to be overridden to implement smarter detection 
804          mechanisms for specific vcs. 
805   
806          This can do more than is_repository_directory() alone. For 
807          example, the Git override checks that Git is actually available. 
808          """
809          if cls.is_repository_directory(location):
810              return location
811          return None
