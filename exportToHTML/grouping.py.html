17   def _group_matching(tlist, cls):
18       """Groups Tokens that have beginning and end."""
19       opens = []
20       tidx_offset = 0
21       for idx, token in enumerate(list(tlist)):
22           tidx = idx - tidx_offset
23   
24           if token.is_whitespace:
25               # ~50% of tokens will be whitespace. Will checking early
26               # for them avoid 3 comparisons, but then add 1 more comparison
27               # for the other ~50% of tokens...
28               continue
29   
30           if token.is_group and not isinstance(token, cls):
31               # Check inside previously grouped (i.e. parenthesis) if group
32               # of different type is inside (i.e., case). though ideally  should
33               # should check for all open/close tokens at once to avoid recursion
34               _group_matching(token, cls)
35               continue
36   
37           if token.match(*cls.M_OPEN):
38               opens.append(tidx)
39   
40           elif token.match(*cls.M_CLOSE):
41               try:
42                   open_idx = opens.pop()
43               except IndexError:
44                   # this indicates invalid sql and unbalanced tokens.
45                   # instead of break, continue in case other "valid" groups exist
46                   continue
47               close_idx = tidx
48               tlist.group_tokens(cls, open_idx, close_idx)
49               tidx_offset += close_idx - open_idx

103  def group_typed_literal(tlist):
104      # definitely not complete, see e.g.:
105      # https://docs.microsoft.com/en-us/sql/odbc/reference/appendixes/interval-literal-syntax
106      # https://docs.microsoft.com/en-us/sql/odbc/reference/appendixes/interval-literals
107      # https://www.postgresql.org/docs/9.1/datatype-datetime.html
108      # https://www.postgresql.org/docs/9.1/functions-datetime.html
109      def match(token):
110          return imt(token, m=sql.TypedLiteral.M_OPEN)
111  
112      def match_to_extend(token):
113          return isinstance(token, sql.TypedLiteral)
114  
115      def valid_prev(token):
116          return token is not None
117  
118      def valid_next(token):
119          return token is not None and token.match(*sql.TypedLiteral.M_CLOSE)
120  
121      def valid_final(token):
122          return token is not None and token.match(*sql.TypedLiteral.M_EXTEND)
123  
124      def post(tlist, pidx, tidx, nidx):
125          return tidx, nidx
126  
127      _group(tlist, sql.TypedLiteral, match, valid_prev, valid_next,
128             post, extend=False)
129      _group(tlist, sql.TypedLiteral, match_to_extend, valid_prev, valid_final,
130             post, extend=True)

133  def group_period(tlist):
134      def match(token):
135          return token.match(T.Punctuation, '.')
136  
137      def valid_prev(token):
138          sqlcls = sql.SquareBrackets, sql.Identifier
139          ttypes = T.Name, T.String.Symbol
140          return imt(token, i=sqlcls, t=ttypes)
141  
142      def valid_next(token):
143          # issue261, allow invalid next token
144          return True
145  
146      def post(tlist, pidx, tidx, nidx):
147          # next_ validation is being performed here. issue261
148          sqlcls = sql.SquareBrackets, sql.Function
149          ttypes = T.Name, T.String.Symbol, T.Wildcard
150          next_ = tlist[nidx] if nidx is not None else None
151          valid_next = imt(next_, i=sqlcls, t=ttypes)
152  
153          return (pidx, nidx) if valid_next else (pidx, tidx)
154  
155      _group(tlist, sql.Identifier, match, valid_prev, valid_next, post)

158  def group_as(tlist):
159      def match(token):
160          return token.is_keyword and token.normalized == 'AS'
161  
162      def valid_prev(token):
163          return token.normalized == 'NULL' or not token.is_keyword
164  
165      def valid_next(token):
166          ttypes = T.DML, T.DDL, T.CTE
167          return not imt(token, t=ttypes) and token is not None
168  
169      def post(tlist, pidx, tidx, nidx):
170          return pidx, nidx
171  
172      _group(tlist, sql.Identifier, match, valid_prev, valid_next, post)

175  def group_assignment(tlist):
176      def match(token):
177          return token.match(T.Assignment, ':=')
178  
179      def valid(token):
180          return token is not None and token.ttype not in (T.Keyword)
181  
182      def post(tlist, pidx, tidx, nidx):
183          m_semicolon = T.Punctuation, ';'
184          snidx, _ = tlist.token_next_by(m=m_semicolon, idx=nidx)
185          nidx = snidx or nidx
186          return pidx, nidx
187  
188      valid_prev = valid_next = valid
189      _group(tlist, sql.Assignment, match, valid_prev, valid_next, post)

192  def group_comparison(tlist):
193      sqlcls = (sql.Parenthesis, sql.Function, sql.Identifier,
194                sql.Operation, sql.TypedLiteral)
195      ttypes = T_NUMERICAL + T_STRING + T_NAME
196  
197      def match(token):
198          return token.ttype == T.Operator.Comparison
199  
200      def valid(token):
201          if imt(token, t=ttypes, i=sqlcls):
202              return True
203          elif token and token.is_keyword and token.normalized == 'NULL':
204              return True
205          else:
206              return False
207  
208      def post(tlist, pidx, tidx, nidx):
209          return pidx, nidx
210  
211      valid_prev = valid_next = valid
212      _group(tlist, sql.Comparison, match,
213             valid_prev, valid_next, post, extend=False)

216  @recurse(sql.Identifier)
217  def group_identifier(tlist):
218      ttypes = (T.String.Symbol, T.Name)
219  
220      tidx, token = tlist.token_next_by(t=ttypes)
221      while token:
222          tlist.group_tokens(sql.Identifier, tidx, tidx)
223          tidx, token = tlist.token_next_by(t=ttypes, idx=tidx)

246  def group_operator(tlist):
247      ttypes = T_NUMERICAL + T_STRING + T_NAME
248      sqlcls = (sql.SquareBrackets, sql.Parenthesis, sql.Function,
249                sql.Identifier, sql.Operation, sql.TypedLiteral)
250  
251      def match(token):
252          return imt(token, t=(T.Operator, T.Wildcard))
253  
254      def valid(token):
255          return imt(token, i=sqlcls, t=ttypes) \
256              or (token and token.match(
257                  T.Keyword,
258                  ('CURRENT_DATE', 'CURRENT_TIME', 'CURRENT_TIMESTAMP')))
259  
260      def post(tlist, pidx, tidx, nidx):
261          tlist[tidx].ttype = T.Operator
262          return pidx, nidx
263  
264      valid_prev = valid_next = valid
265      _group(tlist, sql.Operation, match,
266             valid_prev, valid_next, post, extend=False)

269  def group_identifier_list(tlist):
270      m_role = T.Keyword, ('null', 'role')
271      sqlcls = (sql.Function, sql.Case, sql.Identifier, sql.Comparison,
272                sql.IdentifierList, sql.Operation)
273      ttypes = (T_NUMERICAL + T_STRING + T_NAME
274                + (T.Keyword, T.Comment, T.Wildcard))
275  
276      def match(token):
277          return token.match(T.Punctuation, ',')
278  
279      def valid(token):
280          return imt(token, i=sqlcls, m=m_role, t=ttypes)
281  
282      def post(tlist, pidx, tidx, nidx):
283          return pidx, nidx
284  
285      valid_prev = valid_next = valid
286      _group(tlist, sql.IdentifierList, match,
287             valid_prev, valid_next, post, extend=True)

290  @recurse(sql.Comment)
291  def group_comments(tlist):
292      tidx, token = tlist.token_next_by(t=T.Comment)
293      while token:
294          eidx, end = tlist.token_not_matching(
295              lambda tk: imt(tk, t=T.Comment) or tk.is_whitespace, idx=tidx)
296          if end is not None:
297              eidx, end = tlist.token_prev(eidx, skip_ws=False)
298              tlist.group_tokens(sql.Comment, tidx, eidx)
299  
300          tidx, token = tlist.token_next_by(t=T.Comment, idx=tidx)

303  @recurse(sql.Where)
304  def group_where(tlist):
305      tidx, token = tlist.token_next_by(m=sql.Where.M_OPEN)
306      while token:
307          eidx, end = tlist.token_next_by(m=sql.Where.M_CLOSE, idx=tidx)
308  
309          if end is None:
310              end = tlist._groupable_tokens[-1]
311          else:
312              end = tlist.tokens[eidx - 1]
313          # TODO: convert this to eidx instead of end token.
314          # i think above values are len(tlist) and eidx-1
315          eidx = tlist.token_index(end)
316          tlist.group_tokens(sql.Where, tidx, eidx)
317          tidx, token = tlist.token_next_by(m=sql.Where.M_OPEN, idx=tidx)

320  @recurse()
321  def group_aliased(tlist):
322      I_ALIAS = (sql.Parenthesis, sql.Function, sql.Case, sql.Identifier,
323                 sql.Operation, sql.Comparison)
324  
325      tidx, token = tlist.token_next_by(i=I_ALIAS, t=T.Number)
326      while token:
327          nidx, next_ = tlist.token_next(tidx)
328          if isinstance(next_, sql.Identifier):
329              tlist.group_tokens(sql.Identifier, tidx, nidx, extend=True)
330          tidx, token = tlist.token_next_by(i=I_ALIAS, t=T.Number, idx=tidx)

333  @recurse(sql.Function)
334  def group_functions(tlist):
335      has_create = False
336      has_table = False
337      for tmp_token in tlist.tokens:
338          if tmp_token.value == 'CREATE':
339              has_create = True
340          if tmp_token.value == 'TABLE':
341              has_table = True
342      if has_create and has_table:
343          return
344  
345      tidx, token = tlist.token_next_by(t=T.Name)
346      while token:
347          nidx, next_ = tlist.token_next(tidx)
348          if isinstance(next_, sql.Parenthesis):
349              tlist.group_tokens(sql.Function, tidx, nidx)
350          tidx, token = tlist.token_next_by(t=T.Name, idx=tidx)

353  def group_order(tlist):
354      """Group together Identifier and Asc/Desc token"""
355      tidx, token = tlist.token_next_by(t=T.Keyword.Order)
356      while token:
357          pidx, prev_ = tlist.token_prev(tidx)
358          if imt(prev_, i=sql.Identifier, t=T.Number):
359              tlist.group_tokens(sql.Identifier, pidx, tidx)
360              tidx = pidx
361          tidx, token = tlist.token_next_by(t=T.Keyword.Order, idx=tidx)

364  @recurse()
365  def align_comments(tlist):
366      tidx, token = tlist.token_next_by(i=sql.Comment)
367      while token:
368          pidx, prev_ = tlist.token_prev(tidx)
369          if isinstance(prev_, sql.TokenList):
370              tlist.group_tokens(sql.TokenList, pidx, tidx, extend=True)
371              tidx = pidx
372          tidx, token = tlist.token_next_by(i=sql.Comment, idx=tidx)

375  def group_values(tlist):
376      tidx, token = tlist.token_next_by(m=(T.Keyword, 'VALUES'))
377      start_idx = tidx
378      end_idx = -1
379      while token:
380          if isinstance(token, sql.Parenthesis):
381              end_idx = tidx
382          tidx, token = tlist.token_next(tidx)
383      if end_idx != -1:
384          tlist.group_tokens(sql.Values, start_idx, end_idx, extend=True)

422  def _group(tlist, cls, match,
423             valid_prev=lambda t: True,
424             valid_next=lambda t: True,
425             post=None,
426             extend=True,
427             recurse=True
428             ):
429      """Groups together tokens that are joined by a middle token. i.e. x < y"""
430  
431      tidx_offset = 0
432      pidx, prev_ = None, None
433      for idx, token in enumerate(list(tlist)):
434          tidx = idx - tidx_offset
435          if tidx < 0:  # tidx shouldn't get negative
436              continue
437  
438          if token.is_whitespace:
439              continue
440  
441          if recurse and token.is_group and not isinstance(token, cls):
442              _group(token, cls, match, valid_prev, valid_next, post, extend)
443  
444          if match(token):
445              nidx, next_ = tlist.token_next(tidx)
446              if prev_ and valid_prev(prev_) and valid_next(next_):
447                  from_idx, to_idx = post(tlist, pidx, tidx, nidx)
448                  grp = tlist.group_tokens(cls, from_idx, to_idx, extend=extend)
449  
450                  tidx_offset += to_idx - from_idx
451                  pidx, prev_ = from_idx, grp
452                  continue
453  
454          pidx, prev_ = tidx, token
455
