19   class Serializer(base.Serializer):
20       """Serialize a QuerySet to XML."""
21   
22       def indent(self, level):
23           if self.options.get('indent') is not None:
24               self.xml.ignorableWhitespace('\n' + ' ' * self.options.get('indent') * level)
25   
26       def start_serialization(self):
27           """ 
28           Start serialization -- open the XML document and the root element. 
29           """
30           self.xml = SimplerXMLGenerator(self.stream, self.options.get("encoding", settings.DEFAULT_CHARSET))
31           self.xml.startDocument()
32           self.xml.startElement("django-objects", {"version": "1.0"})
33   
34       def end_serialization(self):
35           """ 
36           End serialization -- end the document. 
37           """
38           self.indent(0)
39           self.xml.endElement("django-objects")
40           self.xml.endDocument()
41   
42       def start_object(self, obj):
43           """ 
44           Called as each object is handled. 
45           """
46           if not hasattr(obj, "_meta"):
47               raise base.SerializationError("Non-model object (%s) encountered during serialization" % type(obj))
48   
49           self.indent(1)
50           attrs = {'model': str(obj._meta)}
51           if not self.use_natural_primary_keys or not hasattr(obj, 'natural_key'):
52               obj_pk = obj.pk
53               if obj_pk is not None:
54                   attrs['pk'] = str(obj_pk)
55   
56           self.xml.startElement("object", attrs)
57   
58       def end_object(self, obj):
59           """ 
60           Called after handling all fields for an object. 
61           """
62           self.indent(1)
63           self.xml.endElement("object")
64   
65       def handle_field(self, obj, field):
66           """ 
67           Handle each field on an object (except for ForeignKeys and 
68           ManyToManyFields). 
69           """
70           self.indent(2)
71           self.xml.startElement('field', {
72               'name': field.name,
73               'type': field.get_internal_type(),
74           })
75   
76           # Get a "string version" of the object's data.
77           if getattr(obj, field.name) is not None:
78               try:
79                   self.xml.characters(field.value_to_string(obj))
80               except UnserializableContentError:
81                   raise ValueError("%s.%s (pk:%s) contains unserializable characters" % (
82                       obj.__class__.__name__, field.name, obj.pk))
83           else:
84               self.xml.addQuickElement("None")
85   
86           self.xml.endElement("field")
87   
88       def handle_fk_field(self, obj, field):
89           """ 
90           Handle a ForeignKey (they need to be treated slightly 
91           differently from regular fields). 
92           """
93           self._start_relational_field(field)
94           related_att = getattr(obj, field.get_attname())
95           if related_att is not None:
96               if self.use_natural_foreign_keys and hasattr(field.remote_field.model, 'natural_key'):
97                   related = getattr(obj, field.name)
98                   # If related object has a natural key, use it
99                   related = related.natural_key()
100                  # Iterable natural keys are rolled out as subelements
101                  for key_value in related:
102                      self.xml.startElement("natural", {})
103                      self.xml.characters(str(key_value))
104                      self.xml.endElement("natural")
105              else:
106                  self.xml.characters(str(related_att))
107          else:
108              self.xml.addQuickElement("None")
109          self.xml.endElement("field")
110  
111      def handle_m2m_field(self, obj, field):
112          """ 
113          Handle a ManyToManyField. Related objects are only serialized as 
114          references to the object's PK (i.e. the related *data* is not dumped, 
115          just the relation). 
116          """
117          if field.remote_field.through._meta.auto_created:
118              self._start_relational_field(field)
119              if self.use_natural_foreign_keys and hasattr(field.remote_field.model, 'natural_key'):
120                  # If the objects in the m2m have a natural key, use it
121                  def handle_m2m(value):
122                      natural = value.natural_key()
123                      # Iterable natural keys are rolled out as subelements
124                      self.xml.startElement("object", {})
125                      for key_value in natural:
126                          self.xml.startElement("natural", {})
127                          self.xml.characters(str(key_value))
128                          self.xml.endElement("natural")
129                      self.xml.endElement("object")
130              else:
131                  def handle_m2m(value):
132                      self.xml.addQuickElement("object", attrs={
133                          'pk': str(value.pk)
134                      })
135              for relobj in getattr(obj, field.name).iterator():
136                  handle_m2m(relobj)
137  
138              self.xml.endElement("field")
139  
140      def _start_relational_field(self, field):
141          """Output the <field> element for relational fields."""
142          self.indent(2)
143          self.xml.startElement('field', {
144              'name': field.name,
145              'rel': field.remote_field.__class__.__name__,
146              'to': str(field.remote_field.model._meta),
147          })

150  class Deserializer(base.Deserializer):
151      """Deserialize XML."""
152  
153      def __init__(self, stream_or_string, *, using=DEFAULT_DB_ALIAS, ignorenonexistent=False, **options):
154          super().__init__(stream_or_string, **options)
155          self.handle_forward_references = options.pop('handle_forward_references', False)
156          self.event_stream = pulldom.parse(self.stream, self._make_parser())
157          self.db = using
158          self.ignore = ignorenonexistent
159  
160      def _make_parser(self):
161          """Create a hardened XML parser (no custom/external entities)."""
162          return DefusedExpatParser()
163  
164      def __next__(self):
165          for event, node in self.event_stream:
166              if event == "START_ELEMENT" and node.nodeName == "object":
167                  self.event_stream.expandNode(node)
168                  return self._handle_object(node)
169          raise StopIteration
170  
171      def _handle_object(self, node):
172          """Convert an <object> node to a DeserializedObject."""
173          # Look up the model using the model loading mechanism. If this fails,
174          # bail.
175          Model = self._get_model_from_node(node, "model")
176  
177          # Start building a data dictionary from the object.
178          data = {}
179          if node.hasAttribute('pk'):
180              data[Model._meta.pk.attname] = Model._meta.pk.to_python(
181                  node.getAttribute('pk'))
182  
183          # Also start building a dict of m2m data (this is saved as
184          # {m2m_accessor_attribute : [list_of_related_objects]})
185          m2m_data = {}
186          deferred_fields = {}
187  
188          field_names = {f.name for f in Model._meta.get_fields()}
189          # Deserialize each field.
190          for field_node in node.getElementsByTagName("field"):
191              # If the field is missing the name attribute, bail (are you
192              # sensing a pattern here?)
193              field_name = field_node.getAttribute("name")
194              if not field_name:
195                  raise base.DeserializationError("<field> node is missing the 'name' attribute")
196  
197              # Get the field from the Model. This will raise a
198              # FieldDoesNotExist if, well, the field doesn't exist, which will
199              # be propagated correctly unless ignorenonexistent=True is used.
200              if self.ignore and field_name not in field_names:
201                  continue
202              field = Model._meta.get_field(field_name)
203  
204              # As is usually the case, relation fields get the special treatment.
205              if field.remote_field and isinstance(field.remote_field, models.ManyToManyRel):
206                  value = self._handle_m2m_field_node(field_node, field)
207                  if value == base.DEFER_FIELD:
208                      deferred_fields[field] = [
209                          [
210                              getInnerText(nat_node).strip()
211                              for nat_node in obj_node.getElementsByTagName('natural')
212                          ]
213                          for obj_node in field_node.getElementsByTagName('object')
214                      ]
215                  else:
216                      m2m_data[field.name] = value
217              elif field.remote_field and isinstance(field.remote_field, models.ManyToOneRel):
218                  value = self._handle_fk_field_node(field_node, field)
219                  if value == base.DEFER_FIELD:
220                      deferred_fields[field] = [
221                          getInnerText(k).strip()
222                          for k in field_node.getElementsByTagName('natural')
223                      ]
224                  else:
225                      data[field.attname] = value
226              else:
227                  if field_node.getElementsByTagName('None'):
228                      value = None
229                  else:
230                      value = field.to_python(getInnerText(field_node).strip())
231                  data[field.name] = value
232  
233          obj = base.build_instance(Model, data, self.db)
234  
235          # Return a DeserializedObject so that the m2m data has a place to live.
236          return base.DeserializedObject(obj, m2m_data, deferred_fields)
237  
238      def _handle_fk_field_node(self, node, field):
239          """ 
240          Handle a <field> node for a ForeignKey 
241          """
242          # Check if there is a child node named 'None', returning None if so.
243          if node.getElementsByTagName('None'):
244              return None
245          else:
246              model = field.remote_field.model
247              if hasattr(model._default_manager, 'get_by_natural_key'):
248                  keys = node.getElementsByTagName('natural')
249                  if keys:
250                      # If there are 'natural' subelements, it must be a natural key
251                      field_value = [getInnerText(k).strip() for k in keys]
252                      try:
253                          obj = model._default_manager.db_manager(self.db).get_by_natural_key(*field_value)
254                      except ObjectDoesNotExist:
255                          if self.handle_forward_references:
256                              return base.DEFER_FIELD
257                          else:
258                              raise
259                      obj_pk = getattr(obj, field.remote_field.field_name)
260                      # If this is a natural foreign key to an object that
261                      # has a FK/O2O as the foreign key, use the FK value
262                      if field.remote_field.model._meta.pk.remote_field:
263                          obj_pk = obj_pk.pk
264                  else:
265                      # Otherwise, treat like a normal PK
266                      field_value = getInnerText(node).strip()
267                      obj_pk = model._meta.get_field(field.remote_field.field_name).to_python(field_value)
268                  return obj_pk
269              else:
270                  field_value = getInnerText(node).strip()
271                  return model._meta.get_field(field.remote_field.field_name).to_python(field_value)
272  
273      def _handle_m2m_field_node(self, node, field):
274          """ 
275          Handle a <field> node for a ManyToManyField. 
276          """
277          model = field.remote_field.model
278          default_manager = model._default_manager
279          if hasattr(default_manager, 'get_by_natural_key'):
280              def m2m_convert(n):
281                  keys = n.getElementsByTagName('natural')
282                  if keys:
283                      # If there are 'natural' subelements, it must be a natural key
284                      field_value = [getInnerText(k).strip() for k in keys]
285                      obj_pk = default_manager.db_manager(self.db).get_by_natural_key(*field_value).pk
286                  else:
287                      # Otherwise, treat like a normal PK value.
288                      obj_pk = model._meta.pk.to_python(n.getAttribute('pk'))
289                  return obj_pk
290          else:
291              def m2m_convert(n):
292                  return model._meta.pk.to_python(n.getAttribute('pk'))
293          values = []
294          try:
295              for c in node.getElementsByTagName('object'):
296                  values.append(m2m_convert(c))
297          except Exception as e:
298              if isinstance(e, ObjectDoesNotExist) and self.handle_forward_references:
299                  return base.DEFER_FIELD
300              else:
301                  raise base.M2MDeserializationError(e, c)
302          else:
303              return values
304  
305      def _get_model_from_node(self, node, attr):
306          """ 
307          Look up a model from a <object model=...> or a <field rel=... to=...> 
308          node. 
309          """
310          model_identifier = node.getAttribute(attr)
311          if not model_identifier:
312              raise base.DeserializationError(
313                  "<%s> node is missing the required '%s' attribute"
314                  % (node.nodeName, attr))
315          try:
316              return apps.get_model(model_identifier)
317          except (LookupError, TypeError):
318              raise base.DeserializationError(
319                  "<%s> node has invalid model identifier: '%s'"
320                  % (node.nodeName, model_identifier))

323  def getInnerText(node):
324      """Get all the inner text of a DOM node (recursively)."""
325      # inspired by https://mail.python.org/pipermail/xml-sig/2005-March/011022.html
326      inner_text = []
327      for child in node.childNodes:
328          if child.nodeType == child.TEXT_NODE or child.nodeType == child.CDATA_SECTION_NODE:
329              inner_text.append(child.data)
330          elif child.nodeType == child.ELEMENT_NODE:
331              inner_text.extend(getInnerText(child))
332          else:
333              pass
334      return "".join(inner_text)

340  class DefusedExpatParser(_ExpatParser):
341      """ 
342      An expat parser hardened against XML bomb attacks. 
343   
344      Forbid DTDs, external entity references 
345      """
346      def __init__(self, *args, **kwargs):
347          super().__init__(*args, **kwargs)
348          self.setFeature(handler.feature_external_ges, False)
349          self.setFeature(handler.feature_external_pes, False)
350  
351      def start_doctype_decl(self, name, sysid, pubid, has_internal_subset):
352          raise DTDForbidden(name, sysid, pubid)
353  
354      def entity_decl(self, name, is_parameter_entity, value, base,
355                      sysid, pubid, notation_name):
356          raise EntitiesForbidden(name, value, base, sysid, pubid, notation_name)
357  
358      def unparsed_entity_decl(self, name, base, sysid, pubid, notation_name):
359          # expat 1.2
360          raise EntitiesForbidden(name, None, base, sysid, pubid, notation_name)
361  
362      def external_entity_ref_handler(self, context, base, sysid, pubid):
363          raise ExternalReferenceForbidden(context, base, sysid, pubid)
364  
365      def reset(self):
366          _ExpatParser.reset(self)
367          parser = self._parser
368          parser.StartDoctypeDeclHandler = self.start_doctype_decl
369          parser.EntityDeclHandler = self.entity_decl
370          parser.UnparsedEntityDeclHandler = self.unparsed_entity_decl
371          parser.ExternalEntityRefHandler = self.external_entity_ref_handler

374  class DefusedXmlException(ValueError):
375      """Base exception."""
376      def __repr__(self):
377          return str(self)
378  
379  
380  class DTDForbidden(DefusedXmlException):
381      """Document type definition is forbidden."""
382      def __init__(self, name, sysid, pubid):
383          super().__init__()
384          self.name = name
385          self.sysid = sysid
386          self.pubid = pubid
387  
388      def __str__(self):
389          tpl = "DTDForbidden(name='{}', system_id={!r}, public_id={!r})"
390          return tpl.format(self.name, self.sysid, self.pubid)
391
