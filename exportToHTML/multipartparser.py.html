44   class MultiPartParser:
45       """ 
46       A rfc2388 multipart/form-data parser. 
47    
48       ``MultiValueDict.parse()`` reads the input stream in ``chunk_size`` chunks 
49       and returns a tuple of ``(MultiValueDict(POST), MultiValueDict(FILES))``. 
50       """
51       def __init__(self, META, input_data, upload_handlers, encoding=None):
52           """ 
53           Initialize the MultiPartParser object. 
54    
55           :META: 
56               The standard ``META`` dictionary in Django request objects. 
57           :input_data: 
58               The raw post data, as a file-like object. 
59           :upload_handlers: 
60               A list of UploadHandler instances that perform operations on the 
61               uploaded data. 
62           :encoding: 
63               The encoding with which to treat the incoming data. 
64           """
65           # Content-Type should contain multipart and the boundary information.
66           content_type = META.get('CONTENT_TYPE', '')
67           if not content_type.startswith('multipart/'):
68               raise MultiPartParserError('Invalid Content-Type: %s' % content_type)
69   
70           # Parse the header to get the boundary to split the parts.
71           try:
72               ctypes, opts = parse_header(content_type.encode('ascii'))
73           except UnicodeEncodeError:
74               raise MultiPartParserError('Invalid non-ASCII Content-Type in multipart: %s' % force_str(content_type))
75           boundary = opts.get('boundary')
76           if not boundary or not cgi.valid_boundary(boundary):
77               raise MultiPartParserError('Invalid boundary in multipart: %s' % force_str(boundary))
78   
79           # Content-Length should contain the length of the body we are about
80           # to receive.
81           try:
82               content_length = int(META.get('CONTENT_LENGTH', 0))
83           except (ValueError, TypeError):
84               content_length = 0
85   
86           if content_length < 0:
87               # This means we shouldn't continue...raise an error.
88               raise MultiPartParserError("Invalid content length: %r" % content_length)
89   
90           if isinstance(boundary, str):
91               boundary = boundary.encode('ascii')
92           self._boundary = boundary
93           self._input_data = input_data
94   
95           # For compatibility with low-level network APIs (with 32-bit integers),
96           # the chunk size should be < 2^31, but still divisible by 4.
97           possible_sizes = [x.chunk_size for x in upload_handlers if x.chunk_size]
98           self._chunk_size = min([2 ** 31 - 4] + possible_sizes)
99   
100          self._meta = META
101          self._encoding = encoding or settings.DEFAULT_CHARSET
102          self._content_length = content_length
103          self._upload_handlers = upload_handlers
104  
105      def parse(self):
106          """ 
107          Parse the POST data and break it into a FILES MultiValueDict and a POST 
108          MultiValueDict. 
109   
110          Return a tuple containing the POST and FILES dictionary, respectively. 
111          """
112          from django.http import QueryDict
113  
114          encoding = self._encoding
115          handlers = self._upload_handlers
116  
117          # HTTP spec says that Content-Length >= 0 is valid
118          # handling content-length == 0 before continuing
119          if self._content_length == 0:
120              return QueryDict(encoding=self._encoding), MultiValueDict()
121  
122          # See if any of the handlers take care of the parsing.
123          # This allows overriding everything if need be.
124          for handler in handlers:
125              result = handler.handle_raw_input(
126                  self._input_data,
127                  self._meta,
128                  self._content_length,
129                  self._boundary,
130                  encoding,
131              )
132              # Check to see if it was handled
133              if result is not None:
134                  return result[0], result[1]
135  
136          # Create the data structures to be used later.
137          self._post = QueryDict(mutable=True)
138          self._files = MultiValueDict()
139  
140          # Instantiate the parser and stream:
141          stream = LazyStream(ChunkIter(self._input_data, self._chunk_size))
142  
143          # Whether or not to signal a file-completion at the beginning of the loop.
144          old_field_name = None
145          counters = [0] * len(handlers)
146  
147          # Number of bytes that have been read.
148          num_bytes_read = 0
149          # To count the number of keys in the request.
150          num_post_keys = 0
151          # To limit the amount of data read from the request.
152          read_size = None
153  
154          try:
155              for item_type, meta_data, field_stream in Parser(stream, self._boundary):
156                  if old_field_name:
157                      # We run this at the beginning of the next loop
158                      # since we cannot be sure a file is complete until
159                      # we hit the next boundary/part of the multipart content.
160                      self.handle_file_complete(old_field_name, counters)
161                      old_field_name = None
162  
163                  try:
164                      disposition = meta_data['content-disposition'][1]
165                      field_name = disposition['name'].strip()
166                  except (KeyError, IndexError, AttributeError):
167                      continue
168  
169                  transfer_encoding = meta_data.get('content-transfer-encoding')
170                  if transfer_encoding is not None:
171                      transfer_encoding = transfer_encoding[0].strip()
172                  field_name = force_str(field_name, encoding, errors='replace')
173  
174                  if item_type == FIELD:
175                      # Avoid storing more than DATA_UPLOAD_MAX_NUMBER_FIELDS.
176                      num_post_keys += 1
177                      if (settings.DATA_UPLOAD_MAX_NUMBER_FIELDS is not None and
178                              settings.DATA_UPLOAD_MAX_NUMBER_FIELDS < num_post_keys):
179                          raise TooManyFieldsSent(
180                              'The number of GET/POST parameters exceeded '
181                              'settings.DATA_UPLOAD_MAX_NUMBER_FIELDS.'
182                          )
183  
184                      # Avoid reading more than DATA_UPLOAD_MAX_MEMORY_SIZE.
185                      if settings.DATA_UPLOAD_MAX_MEMORY_SIZE is not None:
186                          read_size = settings.DATA_UPLOAD_MAX_MEMORY_SIZE - num_bytes_read
187  
188                      # This is a post field, we can just set it in the post
189                      if transfer_encoding == 'base64':
190                          raw_data = field_stream.read(size=read_size)
191                          num_bytes_read += len(raw_data)
192                          try:
193                              data = base64.b64decode(raw_data)
194                          except binascii.Error:
195                              data = raw_data
196                      else:
197                          data = field_stream.read(size=read_size)
198                          num_bytes_read += len(data)
199  
200                      # Add two here to make the check consistent with the
201                      # x-www-form-urlencoded check that includes '&='.
202                      num_bytes_read += len(field_name) + 2
203                      if (settings.DATA_UPLOAD_MAX_MEMORY_SIZE is not None and
204                              num_bytes_read > settings.DATA_UPLOAD_MAX_MEMORY_SIZE):
205                          raise RequestDataTooBig('Request body exceeded settings.DATA_UPLOAD_MAX_MEMORY_SIZE.')
206  
207                      self._post.appendlist(field_name, force_str(data, encoding, errors='replace'))
208                  elif item_type == FILE:
209                      # This is a file, use the handler...
210                      file_name = disposition.get('filename')
211                      if file_name:
212                          file_name = os.path.basename(file_name)
213                          file_name = force_str(file_name, encoding, errors='replace')
214                          file_name = self.IE_sanitize(html.unescape(file_name))
215                      if not file_name:
216                          continue
217  
218                      content_type, content_type_extra = meta_data.get('content-type', ('', {}))
219                      content_type = content_type.strip()
220                      charset = content_type_extra.get('charset')
221  
222                      try:
223                          content_length = int(meta_data.get('content-length')[0])
224                      except (IndexError, TypeError, ValueError):
225                          content_length = None
226  
227                      counters = [0] * len(handlers)
228                      try:
229                          for handler in handlers:
230                              try:
231                                  handler.new_file(
232                                      field_name, file_name, content_type,
233                                      content_length, charset, content_type_extra,
234                                  )
235                              except StopFutureHandlers:
236                                  break
237  
238                          for chunk in field_stream:
239                              if transfer_encoding == 'base64':
240                                  # We only special-case base64 transfer encoding
241                                  # We should always decode base64 chunks by multiple of 4,
242                                  # ignoring whitespace.
243  
244                                  stripped_chunk = b"".join(chunk.split())
245  
246                                  remaining = len(stripped_chunk) % 4
247                                  while remaining != 0:
248                                      over_chunk = field_stream.read(4 - remaining)
249                                      stripped_chunk += b"".join(over_chunk.split())
250                                      remaining = len(stripped_chunk) % 4
251  
252                                  try:
253                                      chunk = base64.b64decode(stripped_chunk)
254                                  except Exception as exc:
255                                      # Since this is only a chunk, any error is an unfixable error.
256                                      raise MultiPartParserError("Could not decode base64 data.") from exc
257  
258                              for i, handler in enumerate(handlers):
259                                  chunk_length = len(chunk)
260                                  chunk = handler.receive_data_chunk(chunk, counters[i])
261                                  counters[i] += chunk_length
262                                  if chunk is None:
263                                      # Don't continue if the chunk received by
264                                      # the handler is None.
265                                      break
266  
267                      except SkipFile:
268                          self._close_files()
269                          # Just use up the rest of this file...
270                          exhaust(field_stream)
271                      else:
272                          # Handle file upload completions on next iteration.
273                          old_field_name = field_name
274                  else:
275                      # If this is neither a FIELD or a FILE, just exhaust the stream.
276                      exhaust(stream)
277          except StopUpload as e:
278              self._close_files()
279              if not e.connection_reset:
280                  exhaust(self._input_data)
281          else:
282              # Make sure that the request data is all fed
283              exhaust(self._input_data)
284  
285          # Signal that the upload has completed.
286          # any() shortcircuits if a handler's upload_complete() returns a value.
287          any(handler.upload_complete() for handler in handlers)
288          self._post._mutable = False
289          return self._post, self._files
290  
291      def handle_file_complete(self, old_field_name, counters):
292          """ 
293          Handle all the signaling that takes place when a file is complete. 
294          """
295          for i, handler in enumerate(self._upload_handlers):
296              file_obj = handler.file_complete(counters[i])
297              if file_obj:
298                  # If it returns a file object, then set the files dict.
299                  self._files.appendlist(force_str(old_field_name, self._encoding, errors='replace'), file_obj)
300                  break
301  
302      def IE_sanitize(self, filename):
303          """Cleanup filename from Internet Explorer full paths."""
304          return filename and filename[filename.rfind("\\") + 1:].strip()
305  
306      def _close_files(self):
307          # Free up all file handles.
308          # FIXME: this currently assumes that upload handlers store the file as 'file'
309          # We should document that... (Maybe add handler.free_file to complement new_file)
310          for handler in self._upload_handlers:
311              if hasattr(handler, 'file'):
312                  handler.file.close()

315  class LazyStream:
316      """ 
317      The LazyStream wrapper allows one to get and "unget" bytes from a stream. 
318   
319      Given a producer object (an iterator that yields bytestrings), the 
320      LazyStream object will support iteration, reading, and keeping a "look-back" 
321      variable in case you need to "unget" some bytes. 
322      """
323      def __init__(self, producer, length=None):
324          """ 
325          Every LazyStream must have a producer when instantiated. 
326   
327          A producer is an iterable that returns a string each time it 
328          is called. 
329          """
330          self._producer = producer
331          self._empty = False
332          self._leftover = b''
333          self.length = length
334          self.position = 0
335          self._remaining = length
336          self._unget_history = []
337  
338      def tell(self):
339          return self.position
340  
341      def read(self, size=None):
342          def parts():
343              remaining = self._remaining if size is None else size
344              # do the whole thing in one shot if no limit was provided.
345              if remaining is None:
346                  yield b''.join(self)
347                  return
348  
349              # otherwise do some bookkeeping to return exactly enough
350              # of the stream and stashing any extra content we get from
351              # the producer
352              while remaining != 0:
353                  assert remaining > 0, 'remaining bytes to read should never go negative'
354  
355                  try:
356                      chunk = next(self)
357                  except StopIteration:
358                      return
359                  else:
360                      emitting = chunk[:remaining]
361                      self.unget(chunk[remaining:])
362                      remaining -= len(emitting)
363                      yield emitting
364  
365          return b''.join(parts())
366  
367      def __next__(self):
368          """ 
369          Used when the exact number of bytes to read is unimportant. 
370   
371          Return whatever chunk is conveniently returned from the iterator. 
372          Useful to avoid unnecessary bookkeeping if performance is an issue. 
373          """
374          if self._leftover:
375              output = self._leftover
376              self._leftover = b''
377          else:
378              output = next(self._producer)
379              self._unget_history = []
380          self.position += len(output)
381          return output
382  
383      def close(self):
384          """ 
385          Used to invalidate/disable this lazy stream. 
386   
387          Replace the producer with an empty list. Any leftover bytes that have 
388          already been read will still be reported upon read() and/or next(). 
389          """
390          self._producer = []
391  
392      def __iter__(self):
393          return self
394  
395      def unget(self, bytes):
396          """ 
397          Place bytes back onto the front of the lazy stream. 
398   
399          Future calls to read() will return those bytes first. The 
400          stream position and thus tell() will be rewound. 
401          """
402          if not bytes:
403              return
404          self._update_unget_history(len(bytes))
405          self.position -= len(bytes)
406          self._leftover = bytes + self._leftover
407  
408      def _update_unget_history(self, num_bytes):
409          """ 
410          Update the unget history as a sanity check to see if we've pushed 
411          back the same number of bytes in one chunk. If we keep ungetting the 
412          same number of bytes many times (here, 50), we're mostly likely in an 
413          infinite loop of some sort. This is usually caused by a 
414          maliciously-malformed MIME request. 
415          """
416          self._unget_history = [num_bytes] + self._unget_history[:49]
417          number_equal = len([
418              current_number for current_number in self._unget_history
419              if current_number == num_bytes
420          ])
421  
422          if number_equal > 40:
423              raise SuspiciousMultipartForm(
424                  "The multipart parser got stuck, which shouldn't happen with"
425                  " normal uploaded files. Check for malicious upload activity;"
426                  " if there is none, report this to the Django developers."
427              )

430  class ChunkIter:
431      """ 
432      An iterable that will yield chunks of data. Given a file-like object as the 
433      constructor, yield chunks of read operations from that object. 
434      """
435      def __init__(self, flo, chunk_size=64 * 1024):
436          self.flo = flo
437          self.chunk_size = chunk_size
438  
439      def __next__(self):
440          try:
441              data = self.flo.read(self.chunk_size)
442          except InputStreamExhausted:
443              raise StopIteration()
444          if data:
445              return data
446          else:
447              raise StopIteration()
448  
449      def __iter__(self):
450          return self

453  class InterBoundaryIter:
454      """ 
455      A Producer that will iterate over boundaries. 
456      """
457      def __init__(self, stream, boundary):
458          self._stream = stream
459          self._boundary = boundary
460  
461      def __iter__(self):
462          return self
463  
464      def __next__(self):
465          try:
466              return LazyStream(BoundaryIter(self._stream, self._boundary))
467          except InputStreamExhausted:
468              raise StopIteration()

471  class BoundaryIter:
472      """ 
473      A Producer that is sensitive to boundaries. 
474   
475      Will happily yield bytes until a boundary is found. Will yield the bytes 
476      before the boundary, throw away the boundary bytes themselves, and push the 
477      post-boundary bytes back on the stream. 
478   
479      The future calls to next() after locating the boundary will raise a 
480      StopIteration exception. 
481      """
482  
483      def __init__(self, stream, boundary):
484          self._stream = stream
485          self._boundary = boundary
486          self._done = False
487          # rollback an additional six bytes because the format is like
488          # this: CRLF<boundary>[--CRLF]
489          self._rollback = len(boundary) + 6
490  
491          # Try to use mx fast string search if available. Otherwise
492          # use Python find. Wrap the latter for consistency.
493          unused_char = self._stream.read(1)
494          if not unused_char:
495              raise InputStreamExhausted()
496          self._stream.unget(unused_char)
497  
498      def __iter__(self):
499          return self
500  
501      def __next__(self):
502          if self._done:
503              raise StopIteration()
504  
505          stream = self._stream
506          rollback = self._rollback
507  
508          bytes_read = 0
509          chunks = []
510          for bytes in stream:
511              bytes_read += len(bytes)
512              chunks.append(bytes)
513              if bytes_read > rollback:
514                  break
515              if not bytes:
516                  break
517          else:
518              self._done = True
519  
520          if not chunks:
521              raise StopIteration()
522  
523          chunk = b''.join(chunks)
524          boundary = self._find_boundary(chunk)
525  
526          if boundary:
527              end, next = boundary
528              stream.unget(chunk[next:])
529              self._done = True
530              return chunk[:end]
531          else:
532              # make sure we don't treat a partial boundary (and
533              # its separators) as data
534              if not chunk[:-rollback]:  # and len(chunk) >= (len(self._boundary) + 6):
535                  # There's nothing left, we should just return and mark as done.
536                  self._done = True
537                  return chunk
538              else:
539                  stream.unget(chunk[-rollback:])
540                  return chunk[:-rollback]
541  
542      def _find_boundary(self, data):
543          """ 
544          Find a multipart boundary in data. 
545   
546          Should no boundary exist in the data, return None. Otherwise, return 
547          a tuple containing the indices of the following: 
548           * the end of current encapsulation 
549           * the start of the next encapsulation 
550          """
551          index = data.find(self._boundary)
552          if index < 0:
553              return None
554          else:
555              end = index
556              next = index + len(self._boundary)
557              # backup over CRLF
558              last = max(0, end - 1)
559              if data[last:last + 1] == b'\n':
560                  end -= 1
561              last = max(0, end - 1)
562              if data[last:last + 1] == b'\r':
563                  end -= 1
564              return end, next

567  def exhaust(stream_or_iterable):
568      """Exhaust an iterator or stream."""
569      try:
570          iterator = iter(stream_or_iterable)
571      except TypeError:
572          iterator = ChunkIter(stream_or_iterable, 16384)
573      collections.deque(iterator, maxlen=0)  # consume iterator quickly.

576  def parse_boundary_stream(stream, max_header_size):
577      """ 
578      Parse one and exactly one stream that encapsulates a boundary. 
579      """
580      # Stream at beginning of header, look for end of header
581      # and parse it if found. The header must fit within one
582      # chunk.
583      chunk = stream.read(max_header_size)
584  
585      # 'find' returns the top of these four bytes, so we'll
586      # need to munch them later to prevent them from polluting
587      # the payload.
588      header_end = chunk.find(b'\r\n\r\n')
589  
590      def _parse_header(line):
591          main_value_pair, params = parse_header(line)
592          try:
593              name, value = main_value_pair.split(':', 1)
594          except ValueError:
595              raise ValueError("Invalid header: %r" % line)
596          return name, (value, params)
597  
598      if header_end == -1:
599          # we find no header, so we just mark this fact and pass on
600          # the stream verbatim
601          stream.unget(chunk)
602          return (RAW, {}, stream)
603  
604      header = chunk[:header_end]
605  
606      # here we place any excess chunk back onto the stream, as
607      # well as throwing away the CRLFCRLF bytes from above.
608      stream.unget(chunk[header_end + 4:])
609  
610      TYPE = RAW
611      outdict = {}
612  
613      # Eliminate blank lines
614      for line in header.split(b'\r\n'):
615          # This terminology ("main value" and "dictionary of
616          # parameters") is from the Python docs.
617          try:
618              name, (value, params) = _parse_header(line)
619          except ValueError:
620              continue
621  
622          if name == 'content-disposition':
623              TYPE = FIELD
624              if params.get('filename'):
625                  TYPE = FILE
626  
627          outdict[name] = value, params
628  
629      if TYPE == RAW:
630          stream.unget(chunk)
631  
632      return (TYPE, outdict, stream)

635  class Parser:
636      def __init__(self, stream, boundary):
637          self._stream = stream
638          self._separator = b'--' + boundary
639  
640      def __iter__(self):
641          boundarystream = InterBoundaryIter(self._stream, self._separator)
642          for sub_stream in boundarystream:
643              # Iterate over each part
644              yield parse_boundary_stream(sub_stream, 1024)

647  def parse_header(line):
648      """ 
649      Parse the header into a key-value. 
650   
651      Input (line): bytes, output: str for key/name, bytes for values which 
652      will be decoded later. 
653      """
654      plist = _parse_header_params(b';' + line)
655      key = plist.pop(0).lower().decode('ascii')
656      pdict = {}
657      for p in plist:
658          i = p.find(b'=')
659          if i >= 0:
660              has_encoding = False
661              name = p[:i].strip().lower().decode('ascii')
662              if name.endswith('*'):
663                  # Lang/encoding embedded in the value (like "filename*=UTF-8''file.ext")
664                  # http://tools.ietf.org/html/rfc2231#section-4
665                  name = name[:-1]
666                  if p.count(b"'") == 2:
667                      has_encoding = True
668              value = p[i + 1:].strip()
669              if len(value) >= 2 and value[:1] == value[-1:] == b'"':
670                  value = value[1:-1]
671                  value = value.replace(b'\\\\', b'\\').replace(b'\\"', b'"')
672              if has_encoding:
673                  encoding, lang, value = value.split(b"'")
674                  value = unquote(value.decode(), encoding=encoding.decode())
675              pdict[name] = value
676      return key, pdict

679  def _parse_header_params(s):
680      plist = []
681      while s[:1] == b';':
682          s = s[1:]
683          end = s.find(b';')
684          while end > 0 and s.count(b'"', 0, end) % 2:
685              end = s.find(b';', end + 1)
686          if end < 0:
687              end = len(s)
688          f = s[:end]
689          plist.append(f.strip())
690          s = s[end:]
691      return plist
