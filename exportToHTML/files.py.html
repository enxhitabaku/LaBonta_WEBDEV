14   class FieldFile(File):
15       def __init__(self, instance, field, name):
16           super().__init__(None, name)
17           self.instance = instance
18           self.field = field
19           self.storage = field.storage
20           self._committed = True
21   
22       def __eq__(self, other):
23           # Older code may be expecting FileField values to be simple strings.
24           # By overriding the == operator, it can remain backwards compatibility.
25           if hasattr(other, 'name'):
26               return self.name == other.name
27           return self.name == other
28   
29       def __hash__(self):
30           return hash(self.name)
31   
32       # The standard File contains most of the necessary properties, but
33       # FieldFiles can be instantiated without a name, so that needs to
34       # be checked for here.
35   
36       def _require_file(self):
37           if not self:
38               raise ValueError("The '%s' attribute has no file associated with it." % self.field.name)
39   
40       def _get_file(self):
41           self._require_file()
42           if getattr(self, '_file', None) is None:
43               self._file = self.storage.open(self.name, 'rb')
44           return self._file
45   
46       def _set_file(self, file):
47           self._file = file
48   
49       def _del_file(self):
50           del self._file
51   
52       file = property(_get_file, _set_file, _del_file)
53   
54       @property
55       def path(self):
56           self._require_file()
57           return self.storage.path(self.name)
58   
59       @property
60       def url(self):
61           self._require_file()
62           return self.storage.url(self.name)
63   
64       @property
65       def size(self):
66           self._require_file()
67           if not self._committed:
68               return self.file.size
69           return self.storage.size(self.name)
70   
71       def open(self, mode='rb'):
72           self._require_file()
73           if getattr(self, '_file', None) is None:
74               self.file = self.storage.open(self.name, mode)
75           else:
76               self.file.open(mode)
77           return self
78       # open() doesn't alter the file's contents, but it does reset the pointer
79       open.alters_data = True
80   
81       # In addition to the standard File API, FieldFiles have extra methods
82       # to further manipulate the underlying file, as well as update the
83       # associated model instance.
84   
85       def save(self, name, content, save=True):
86           name = self.field.generate_filename(self.instance, name)
87           self.name = self.storage.save(name, content, max_length=self.field.max_length)
88           setattr(self.instance, self.field.name, self.name)
89           self._committed = True
90   
91           # Save the object because it has changed, unless save is False
92           if save:
93               self.instance.save()
94       save.alters_data = True
95   
96       def delete(self, save=True):
97           if not self:
98               return
99           # Only close the file if it's already open, which we know by the
100          # presence of self._file
101          if hasattr(self, '_file'):
102              self.close()
103              del self.file
104  
105          self.storage.delete(self.name)
106  
107          self.name = None
108          setattr(self.instance, self.field.name, self.name)
109          self._committed = False
110  
111          if save:
112              self.instance.save()
113      delete.alters_data = True
114  
115      @property
116      def closed(self):
117          file = getattr(self, '_file', None)
118          return file is None or file.closed
119  
120      def close(self):
121          file = getattr(self, '_file', None)
122          if file is not None:
123              file.close()
124  
125      def __getstate__(self):
126          # FieldFile needs access to its associated model field, an instance and
127          # the file's name. Everything else will be restored later, by
128          # FileDescriptor below.
129          return {
130              'name': self.name,
131              'closed': False,
132              '_committed': True,
133              '_file': None,
134              'instance': self.instance,
135              'field': self.field,
136          }
137  
138      def __setstate__(self, state):
139          self.__dict__.update(state)
140          self.storage = self.field.storage

143  class FileDescriptor:
144      """ 
145      The descriptor for the file attribute on the model instance. Return a 
146      FieldFile when accessed so you can write code like:: 
147   
148          >>> from myapp.models import MyModel 
149          >>> instance = MyModel.objects.get(pk=1) 
150          >>> instance.file.size 
151   
152      Assign a file object on assignment so you can do:: 
153   
154          >>> with open('/path/to/hello.world') as f: 
155          ...     instance.file = File(f) 
156      """
157      def __init__(self, field):
158          self.field = field
159  
160      def __get__(self, instance, cls=None):
161          if instance is None:
162              return self
163  
164          # This is slightly complicated, so worth an explanation.
165          # instance.file`needs to ultimately return some instance of `File`,
166          # probably a subclass. Additionally, this returned object needs to have
167          # the FieldFile API so that users can easily do things like
168          # instance.file.path and have that delegated to the file storage engine.
169          # Easy enough if we're strict about assignment in __set__, but if you
170          # peek below you can see that we're not. So depending on the current
171          # value of the field we have to dynamically construct some sort of
172          # "thing" to return.
173  
174          # The instance dict contains whatever was originally assigned
175          # in __set__.
176          if self.field.name in instance.__dict__:
177              file = instance.__dict__[self.field.name]
178          else:
179              instance.refresh_from_db(fields=[self.field.name])
180              file = getattr(instance, self.field.name)
181  
182          # If this value is a string (instance.file = "path/to/file") or None
183          # then we simply wrap it with the appropriate attribute class according
184          # to the file field. [This is FieldFile for FileFields and
185          # ImageFieldFile for ImageFields; it's also conceivable that user
186          # subclasses might also want to subclass the attribute class]. This
187          # object understands how to convert a path to a file, and also how to
188          # handle None.
189          if isinstance(file, str) or file is None:
190              attr = self.field.attr_class(instance, self.field, file)
191              instance.__dict__[self.field.name] = attr
192  
193          # Other types of files may be assigned as well, but they need to have
194          # the FieldFile interface added to them. Thus, we wrap any other type of
195          # File inside a FieldFile (well, the field's attr_class, which is
196          # usually FieldFile).
197          elif isinstance(file, File) and not isinstance(file, FieldFile):
198              file_copy = self.field.attr_class(instance, self.field, file.name)
199              file_copy.file = file
200              file_copy._committed = False
201              instance.__dict__[self.field.name] = file_copy
202  
203          # Finally, because of the (some would say boneheaded) way pickle works,
204          # the underlying FieldFile might not actually itself have an associated
205          # file. So we need to reset the details of the FieldFile in those cases.
206          elif isinstance(file, FieldFile) and not hasattr(file, 'field'):
207              file.instance = instance
208              file.field = self.field
209              file.storage = self.field.storage
210  
211          # Make sure that the instance is correct.
212          elif isinstance(file, FieldFile) and instance is not file.instance:
213              file.instance = instance
214  
215          # That was fun, wasn't it?
216          return instance.__dict__[self.field.name]
217  
218      def __set__(self, instance, value):
219          instance.__dict__[self.field.name] = value
220

222  class FileField(Field):
223  
224      # The class to wrap instance attributes in. Accessing the file object off
225      # the instance will always return an instance of attr_class.
226      attr_class = FieldFile
227  
228      # The descriptor to use for accessing the attribute off of the class.
229      descriptor_class = FileDescriptor
230  
231      description = _("File")
232  
233      def __init__(self, verbose_name=None, name=None, upload_to='', storage=None, **kwargs):
234          self._primary_key_set_explicitly = 'primary_key' in kwargs
235  
236          self.storage = storage or default_storage
237          if callable(self.storage):
238              # Hold a reference to the callable for deconstruct().
239              self._storage_callable = self.storage
240              self.storage = self.storage()
241              if not isinstance(self.storage, Storage):
242                  raise TypeError(
243                      "%s.storage must be a subclass/instance of %s.%s"
244                      % (self.__class__.__qualname__, Storage.__module__, Storage.__qualname__)
245                  )
246          self.upload_to = upload_to
247  
248          kwargs.setdefault('max_length', 100)
249          super().__init__(verbose_name, name, **kwargs)
250  
251      def check(self, **kwargs):
252          return [
253              *super().check(**kwargs),
254              *self._check_primary_key(),
255              *self._check_upload_to(),
256          ]
257  
258      def _check_primary_key(self):
259          if self._primary_key_set_explicitly:
260              return [
261                  checks.Error(
262                      "'primary_key' is not a valid argument for a %s." % self.__class__.__name__,
263                      obj=self,
264                      id='fields.E201',
265                  )
266              ]
267          else:
268              return []
269  
270      def _check_upload_to(self):
271          if isinstance(self.upload_to, str) and self.upload_to.startswith('/'):
272              return [
273                  checks.Error(
274                      "%s's 'upload_to' argument must be a relative path, not an "
275                      "absolute path." % self.__class__.__name__,
276                      obj=self,
277                      id='fields.E202',
278                      hint='Remove the leading slash.',
279                  )
280              ]
281          else:
282              return []
283  
284      def deconstruct(self):
285          name, path, args, kwargs = super().deconstruct()
286          if kwargs.get("max_length") == 100:
287              del kwargs["max_length"]
288          kwargs['upload_to'] = self.upload_to
289          if self.storage is not default_storage:
290              kwargs['storage'] = getattr(self, '_storage_callable', self.storage)
291          return name, path, args, kwargs
292  
293      def get_internal_type(self):
294          return "FileField"
295  
296      def get_prep_value(self, value):
297          value = super().get_prep_value(value)
298          # Need to convert File objects provided via a form to string for database insertion
299          if value is None:
300              return None
301          return str(value)
302  
303      def pre_save(self, model_instance, add):
304          file = super().pre_save(model_instance, add)
305          if file and not file._committed:
306              # Commit the file to storage prior to saving the model
307              file.save(file.name, file.file, save=False)
308          return file
309  
310      def contribute_to_class(self, cls, name, **kwargs):
311          super().contribute_to_class(cls, name, **kwargs)
312          setattr(cls, self.name, self.descriptor_class(self))
313  
314      def generate_filename(self, instance, filename):
315          """ 
316          Apply (if callable) or prepend (if a string) upload_to to the filename, 
317          then delegate further processing of the name to the storage backend. 
318          Until the storage layer, all file paths are expected to be Unix style 
319          (with forward slashes). 
320          """
321          if callable(self.upload_to):
322              filename = self.upload_to(instance, filename)
323          else:
324              dirname = datetime.datetime.now().strftime(str(self.upload_to))
325              filename = posixpath.join(dirname, filename)
326          return self.storage.generate_filename(filename)
327  
328      def save_form_data(self, instance, data):
329          # Important: None means "no change", other false value means "clear"
330          # This subtle distinction (rather than a more explicit marker) is
331          # needed because we need to consume values that are also sane for a
332          # regular (non Model-) Form to find in its cleaned_data dictionary.
333          if data is not None:
334              # This value will be converted to str and stored in the
335              # database, so leaving False as-is is not acceptable.
336              setattr(instance, self.name, data or '')
337  
338      def formfield(self, **kwargs):
339          return super().formfield(**{
340              'form_class': forms.FileField,
341              'max_length': self.max_length,
342              **kwargs,
343          })

346  class ImageFileDescriptor(FileDescriptor):
347      """ 
348      Just like the FileDescriptor, but for ImageFields. The only difference is 
349      assigning the width/height to the width_field/height_field, if appropriate. 
350      """
351      def __set__(self, instance, value):
352          previous_file = instance.__dict__.get(self.field.name)
353          super().__set__(instance, value)
354  
355          # To prevent recalculating image dimensions when we are instantiating
356          # an object from the database (bug #11084), only update dimensions if
357          # the field had a value before this assignment.  Since the default
358          # value for FileField subclasses is an instance of field.attr_class,
359          # previous_file will only be None when we are called from
360          # Model.__init__().  The ImageField.update_dimension_fields method
361          # hooked up to the post_init signal handles the Model.__init__() cases.
362          # Assignment happening outside of Model.__init__() will trigger the
363          # update right here.
364          if previous_file is not None:
365              self.field.update_dimension_fields(instance, force=True)

368  class ImageFieldFile(ImageFile, FieldFile):
369      def delete(self, save=True):
370          # Clear the image dimensions cache
371          if hasattr(self, '_dimensions_cache'):
372              del self._dimensions_cache
373          super().delete(save)

376  class ImageField(FileField):
377      attr_class = ImageFieldFile
378      descriptor_class = ImageFileDescriptor
379      description = _("Image")
380  
381      def __init__(self, verbose_name=None, name=None, width_field=None, height_field=None, **kwargs):
382          self.width_field, self.height_field = width_field, height_field
383          super().__init__(verbose_name, name, **kwargs)
384  
385      def check(self, **kwargs):
386          return [
387              *super().check(**kwargs),
388              *self._check_image_library_installed(),
389          ]
390  
391      def _check_image_library_installed(self):
392          try:
393              from PIL import Image  # NOQA
394          except ImportError:
395              return [
396                  checks.Error(
397                      'Cannot use ImageField because Pillow is not installed.',
398                      hint=('Get Pillow at https://pypi.org/project/Pillow/ '
399                            'or run command "python -m pip install Pillow".'),
400                      obj=self,
401                      id='fields.E210',
402                  )
403              ]
404          else:
405              return []
406  
407      def deconstruct(self):
408          name, path, args, kwargs = super().deconstruct()
409          if self.width_field:
410              kwargs['width_field'] = self.width_field
411          if self.height_field:
412              kwargs['height_field'] = self.height_field
413          return name, path, args, kwargs
414  
415      def contribute_to_class(self, cls, name, **kwargs):
416          super().contribute_to_class(cls, name, **kwargs)
417          # Attach update_dimension_fields so that dimension fields declared
418          # after their corresponding image field don't stay cleared by
419          # Model.__init__, see bug #11196.
420          # Only run post-initialization dimension update on non-abstract models
421          if not cls._meta.abstract:
422              signals.post_init.connect(self.update_dimension_fields, sender=cls)
423  
424      def update_dimension_fields(self, instance, force=False, *args, **kwargs):
425          """ 
426          Update field's width and height fields, if defined. 
427   
428          This method is hooked up to model's post_init signal to update 
429          dimensions after instantiating a model instance.  However, dimensions 
430          won't be updated if the dimensions fields are already populated.  This 
431          avoids unnecessary recalculation when loading an object from the 
432          database. 
433   
434          Dimensions can be forced to update with force=True, which is how 
435          ImageFileDescriptor.__set__ calls this method. 
436          """
437          # Nothing to update if the field doesn't have dimension fields or if
438          # the field is deferred.
439          has_dimension_fields = self.width_field or self.height_field
440          if not has_dimension_fields or self.attname not in instance.__dict__:
441              return
442  
443          # getattr will call the ImageFileDescriptor's __get__ method, which
444          # coerces the assigned value into an instance of self.attr_class
445          # (ImageFieldFile in this case).
446          file = getattr(instance, self.attname)
447  
448          # Nothing to update if we have no file and not being forced to update.
449          if not file and not force:
450              return
451  
452          dimension_fields_filled = not(
453              (self.width_field and not getattr(instance, self.width_field)) or
454              (self.height_field and not getattr(instance, self.height_field))
455          )
456          # When both dimension fields have values, we are most likely loading
457          # data from the database or updating an image field that already had
458          # an image stored.  In the first case, we don't want to update the
459          # dimension fields because we are already getting their values from the
460          # database.  In the second case, we do want to update the dimensions
461          # fields and will skip this return because force will be True since we
462          # were called from ImageFileDescriptor.__set__.
463          if dimension_fields_filled and not force:
464              return
465  
466          # file should be an instance of ImageFieldFile or should be None.
467          if file:
468              width = file.width
469              height = file.height
470          else:
471              # No file, so clear dimensions fields.
472              width = None
473              height = None
474  
475          # Update the width and height fields.
476          if self.width_field:
477              setattr(instance, self.width_field, width)
478          if self.height_field:
479              setattr(instance, self.height_field, height)
480  
481      def formfield(self, **kwargs):
482          return super().formfield(**{
483              'form_class': forms.ImageField,
484              **kwargs,
485          })
