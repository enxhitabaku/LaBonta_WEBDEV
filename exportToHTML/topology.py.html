topology.py 
1    """ 
2     This module houses the GEOS ctypes prototype functions for the 
3     topological operations on geometries. 
4    """
5    from ctypes import c_double, c_int
6    
7    from django.contrib.gis.geos.libgeos import GEOM_PTR, GEOSFuncFactory
8    from django.contrib.gis.geos.prototypes.errcheck import (
9        check_geom, check_minus_one, check_string,
10   )
11   from django.contrib.gis.geos.prototypes.geom import geos_char_p
12   
13   
14   class Topology(GEOSFuncFactory):
15       "For GEOS unary topology functions."
16       argtypes = [GEOM_PTR]
17       restype = GEOM_PTR
18       errcheck = staticmethod(check_geom)
19   
20   
21   # Topology Routines
22   geos_boundary = Topology('GEOSBoundary')
23   geos_buffer = Topology('GEOSBuffer', argtypes=[GEOM_PTR, c_double, c_int])
24   geos_bufferwithstyle = Topology('GEOSBufferWithStyle', argtypes=[GEOM_PTR, c_double, c_int, c_int, c_int, c_double])
25   geos_centroid = Topology('GEOSGetCentroid')
26   geos_convexhull = Topology('GEOSConvexHull')
27   geos_difference = Topology('GEOSDifference', argtypes=[GEOM_PTR, GEOM_PTR])
28   geos_envelope = Topology('GEOSEnvelope')
29   geos_intersection = Topology('GEOSIntersection', argtypes=[GEOM_PTR, GEOM_PTR])
30   geos_linemerge = Topology('GEOSLineMerge')
31   geos_pointonsurface = Topology('GEOSPointOnSurface')
32   geos_preservesimplify = Topology('GEOSTopologyPreserveSimplify', argtypes=[GEOM_PTR, c_double])
33   geos_simplify = Topology('GEOSSimplify', argtypes=[GEOM_PTR, c_double])
34   geos_symdifference = Topology('GEOSSymDifference', argtypes=[GEOM_PTR, GEOM_PTR])
35   geos_union = Topology('GEOSUnion', argtypes=[GEOM_PTR, GEOM_PTR])
36   
37   geos_unary_union = GEOSFuncFactory('GEOSUnaryUnion', argtypes=[GEOM_PTR], restype=GEOM_PTR)
38   
39   # GEOSRelate returns a string, not a geometry.
40   geos_relate = GEOSFuncFactory(
41       'GEOSRelate', argtypes=[GEOM_PTR, GEOM_PTR], restype=geos_char_p, errcheck=check_string
42   )
43   
44   # Linear referencing routines
45   geos_project = GEOSFuncFactory(
46       'GEOSProject', argtypes=[GEOM_PTR, GEOM_PTR], restype=c_double, errcheck=check_minus_one
47   )
48   geos_interpolate = Topology('GEOSInterpolate', argtypes=[GEOM_PTR, c_double])
49   
50   geos_project_normalized = GEOSFuncFactory(
51       'GEOSProjectNormalized', argtypes=[GEOM_PTR, GEOM_PTR], restype=c_double, errcheck=check_minus_one
52   )
53   geos_interpolate_normalized = Topology('GEOSInterpolateNormalized', argtypes=[GEOM_PTR, c_double])
