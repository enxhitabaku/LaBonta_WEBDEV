115  # ### Base I/O Class ###
116  class IOBase(GEOSBase):
117      "Base class for GEOS I/O objects."
118      def __init__(self):
119          # Getting the pointer with the constructor.
120          self.ptr = self._constructor()
121          # Loading the real destructor function at this point as doing it in
122          # __del__ is too late (import error).
123          self.destructor.func

128  # Non-public WKB/WKT reader classes for internal use because
129  # their `read` methods return _pointers_ instead of GEOSGeometry
130  # objects.
131  class _WKTReader(IOBase):
132      _constructor = wkt_reader_create
133      ptr_type = WKT_READ_PTR
134      destructor = wkt_reader_destroy
135  
136      def read(self, wkt):
137          if not isinstance(wkt, (bytes, str)):
138              raise TypeError
139          return wkt_reader_read(self.ptr, force_bytes(wkt))

142  class _WKBReader(IOBase):
143      _constructor = wkb_reader_create
144      ptr_type = WKB_READ_PTR
145      destructor = wkb_reader_destroy
146  
147      def read(self, wkb):
148          "Return a _pointer_ to C GEOS Geometry object from the given WKB."
149          if isinstance(wkb, memoryview):
150              wkb_s = bytes(wkb)
151              return wkb_reader_read(self.ptr, wkb_s, len(wkb_s))
152          elif isinstance(wkb, (bytes, str)):
153              return wkb_reader_read_hex(self.ptr, wkb, len(wkb))
154          else:
155              raise TypeError

158  # ### WKB/WKT Writer Classes ###
159  class WKTWriter(IOBase):
160      _constructor = wkt_writer_create
161      ptr_type = WKT_WRITE_PTR
162      destructor = wkt_writer_destroy
163  
164      _trim = False
165      _precision = None
166  
167      def __init__(self, dim=2, trim=False, precision=None):
168          super().__init__()
169          if bool(trim) != self._trim:
170              self.trim = trim
171          if precision is not None:
172              self.precision = precision
173          self.outdim = dim
174  
175      def write(self, geom):
176          "Return the WKT representation of the given geometry."
177          return wkt_writer_write(self.ptr, geom.ptr)
178  
179      @property
180      def outdim(self):
181          return wkt_writer_get_outdim(self.ptr)
182  
183      @outdim.setter
184      def outdim(self, new_dim):
185          if new_dim not in (2, 3):
186              raise ValueError('WKT output dimension must be 2 or 3')
187          wkt_writer_set_outdim(self.ptr, new_dim)
188  
189      @property
190      def trim(self):
191          return self._trim
192  
193      @trim.setter
194      def trim(self, flag):
195          if bool(flag) != self._trim:
196              self._trim = bool(flag)
197              wkt_writer_set_trim(self.ptr, self._trim)
198  
199      @property
200      def precision(self):
201          return self._precision
202  
203      @precision.setter
204      def precision(self, precision):
205          if (not isinstance(precision, int) or precision < 0) and precision is not None:
206              raise AttributeError('WKT output rounding precision must be non-negative integer or None.')
207          if precision != self._precision:
208              self._precision = precision
209              wkt_writer_set_precision(self.ptr, -1 if precision is None else precision)

212  class WKBWriter(IOBase):
213      _constructor = wkb_writer_create
214      ptr_type = WKB_WRITE_PTR
215      destructor = wkb_writer_destroy
216      geos_version = geos_version_tuple()
217  
218      def __init__(self, dim=2):
219          super().__init__()
220          self.outdim = dim
221  
222      def _handle_empty_point(self, geom):
223          from django.contrib.gis.geos import Point
224          if isinstance(geom, Point) and geom.empty:
225              if self.srid:
226                  # PostGIS uses POINT(NaN NaN) for WKB representation of empty
227                  # points. Use it for EWKB as it's a PostGIS specific format.
228                  # https://trac.osgeo.org/postgis/ticket/3181
229                  geom = Point(float('NaN'), float('NaN'), srid=geom.srid)
230              else:
231                  raise ValueError('Empty point is not representable in WKB.')
232          return geom
233  
234      def write(self, geom):
235          "Return the WKB representation of the given geometry."
236          from django.contrib.gis.geos import Polygon
237          geom = self._handle_empty_point(geom)
238          wkb = wkb_writer_write(self.ptr, geom.ptr, byref(c_size_t()))
239          if self.geos_version < (3, 6, 1) and isinstance(geom, Polygon) and geom.empty:
240              # Fix GEOS output for empty polygon.
241              # See https://trac.osgeo.org/geos/ticket/680.
242              wkb = wkb[:-8] + b'\0' * 4
243          return memoryview(wkb)
244  
245      def write_hex(self, geom):
246          "Return the HEXEWKB representation of the given geometry."
247          from django.contrib.gis.geos.polygon import Polygon
248          geom = self._handle_empty_point(geom)
249          wkb = wkb_writer_write_hex(self.ptr, geom.ptr, byref(c_size_t()))
250          if self.geos_version < (3, 6, 1) and isinstance(geom, Polygon) and geom.empty:
251              wkb = wkb[:-16] + b'0' * 8
252          return wkb
253  
254      # ### WKBWriter Properties ###
255  
256      # Property for getting/setting the byteorder.
257      def _get_byteorder(self):
258          return wkb_writer_get_byteorder(self.ptr)
259  
260      def _set_byteorder(self, order):
261          if order not in (0, 1):
262              raise ValueError('Byte order parameter must be 0 (Big Endian) or 1 (Little Endian).')
263          wkb_writer_set_byteorder(self.ptr, order)
264  
265      byteorder = property(_get_byteorder, _set_byteorder)
266  
267      # Property for getting/setting the output dimension.
268      @property
269      def outdim(self):
270          return wkb_writer_get_outdim(self.ptr)
271  
272      @outdim.setter
273      def outdim(self, new_dim):
274          if new_dim not in (2, 3):
275              raise ValueError('WKB output dimension must be 2 or 3')
276          wkb_writer_set_outdim(self.ptr, new_dim)
277  
278      # Property for getting/setting the include srid flag.
279      @property
280      def srid(self):
281          return bool(wkb_writer_get_include_srid(self.ptr))
282  
283      @srid.setter
284      def srid(self, include):
285          wkb_writer_set_include_srid(self.ptr, bool(include))

288  # `ThreadLocalIO` object holds instances of the WKT and WKB reader/writer
289  # objects that are local to the thread.  The `GEOSGeometry` internals
290  # access these instances by calling the module-level functions, defined
291  # below.
292  class ThreadLocalIO(threading.local):
293      wkt_r = None
294      wkt_w = None
295      wkb_r = None
296      wkb_w = None
297      ewkb_w = None
298  
299  
300  thread_context = ThreadLocalIO()
301  
302  
303  # These module-level routines return the I/O object that is local to the
304  # thread. If the I/O object does not exist yet it will be initialized.
305  def wkt_r():
306      thread_context.wkt_r = thread_context.wkt_r or _WKTReader()
307      return thread_context.wkt_r
308  
309  
310  def wkt_w(dim=2, trim=False, precision=None):
311      if not thread_context.wkt_w:
312          thread_context.wkt_w = WKTWriter(dim=dim, trim=trim, precision=precision)
313      else:
314          thread_context.wkt_w.outdim = dim
315          thread_context.wkt_w.trim = trim
316          thread_context.wkt_w.precision = precision
317      return thread_context.wkt_w
318  
319  
320  def wkb_r():
321      thread_context.wkb_r = thread_context.wkb_r or _WKBReader()
322      return thread_context.wkb_r
323  
324  
325  def wkb_w(dim=2):
326      if not thread_context.wkb_w:
327          thread_context.wkb_w = WKBWriter(dim=dim)
328      else:
329          thread_context.wkb_w.outdim = dim
330      return thread_context.wkb_w
331  
332  
333  def ewkb_w(dim=2):
334      if not thread_context.ewkb_w:
335          thread_context.ewkb_w = WKBWriter(dim=dim)
336          thread_context.ewkb_w.srid = True
337      else:
338          thread_context.ewkb_w.outdim = dim
339      return thread_context.ewkb_w
