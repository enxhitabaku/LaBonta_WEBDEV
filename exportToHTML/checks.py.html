24   def check_model_name_lengths(app_configs=None, **kwargs):
25       if app_configs is None:
26           models = apps.get_models()
27       else:
28           models = chain.from_iterable(app_config.get_models() for app_config in app_configs)
29       errors = []
30       for model in models:
31           if len(model._meta.model_name) > 100:
32               errors.append(
33                   Error(
34                       'Model names must be at most 100 characters (got %d).' % (
35                           len(model._meta.model_name),
36                       ),
37                       obj=model,
38                       id='contenttypes.E005',
39                   )
40               )
41       return errors
