56   def _match_vcs_scheme(url):
57       # type: (str) -> Optional[str]
58       """Look for VCS schemes in the URL. 
59    
60       Returns the matched VCS scheme, or None if there's no match. 
61       """
62       for scheme in vcs.schemes:
63           if url.lower().startswith(scheme) and url[len(scheme)] in '+:':
64               return scheme
65       return None

68   def _is_url_like_archive(url):
69       # type: (str) -> bool
70       """Return whether the URL looks like an archive. 
71       """
72       filename = Link(url).filename
73       for bad_ext in ARCHIVE_EXTENSIONS:
74           if filename.endswith(bad_ext):
75               return True
76       return False

87   def _ensure_html_header(response):
88       # type: (Response) -> None
89       """Check the Content-Type header to ensure the response contains HTML. 
90    
91       Raises `_NotHTML` if the content type is not text/html. 
92       """
93       content_type = response.headers.get("Content-Type", "")
94       if not content_type.lower().startswith("text/html"):
95           raise _NotHTML(content_type, response.request.method)

102  def _ensure_html_response(url, session):
103      # type: (str, PipSession) -> None
104      """Send a HEAD request to the URL, and ensure the response contains HTML. 
105   
106      Raises `_NotHTTP` if the URL is not available for a HEAD request, or 
107      `_NotHTML` if the content type is not text/html. 
108      """
109      scheme, netloc, path, query, fragment = urllib_parse.urlsplit(url)
110      if scheme not in {'http', 'https'}:
111          raise _NotHTTP()
112  
113      resp = session.head(url, allow_redirects=True)
114      raise_for_status(resp)
115  
116      _ensure_html_header(resp)

119  def _get_html_response(url, session):
120      # type: (str, PipSession) -> Response
121      """Access an HTML page with GET, and return the response. 
122   
123      This consists of three parts: 
124   
125      1. If the URL looks suspiciously like an archive, send a HEAD first to 
126         check the Content-Type is HTML, to avoid downloading a large file. 
127         Raise `_NotHTTP` if the content type cannot be determined, or 
128         `_NotHTML` if it is not HTML. 
129      2. Actually perform the request. Raise HTTP exceptions on network failures. 
130      3. Check the Content-Type header to make sure we got HTML, and raise 
131         `_NotHTML` otherwise. 
132      """
133      if _is_url_like_archive(url):
134          _ensure_html_response(url, session=session)
135  
136      logger.debug('Getting page %s', redact_auth_from_url(url))
137  
138      resp = session.get(
139          url,
140          headers={
141              "Accept": "text/html",
142              # We don't want to blindly returned cached data for
143              # /simple/, because authors generally expecting that
144              # twine upload && pip install will function, but if
145              # they've done a pip install in the last ~10 minutes
146              # it won't. Thus by setting this to zero we will not
147              # blindly use any cached data, however the benefit of
148              # using max-age=0 instead of no-cache, is that we will
149              # still support conditional requests, so we will still
150              # minimize traffic sent in cases where the page hasn't
151              # changed at all, we will just always incur the round
152              # trip for the conditional GET now instead of only
153              # once per 10 minutes.
154              # For more information, please see pypa/pip#5670.
155              "Cache-Control": "max-age=0",
156          },
157      )
158      raise_for_status(resp)
159  
160      # The check for archives above only works if the url ends with
161      # something that looks like an archive. However that is not a
162      # requirement of an url. Unless we issue a HEAD request on every
163      # url we cannot know ahead of time for sure if something is HTML
164      # or not. However we can check after we've downloaded it.
165      _ensure_html_header(resp)
166  
167      return resp

170  def _get_encoding_from_headers(headers):
171      # type: (ResponseHeaders) -> Optional[str]
172      """Determine if we have any encoding information in our headers. 
173      """
174      if headers and "Content-Type" in headers:
175          content_type, params = cgi.parse_header(headers["Content-Type"])
176          if "charset" in params:
177              return params['charset']
178      return None

181  def _determine_base_url(document, page_url):
182      # type: (HTMLElement, str) -> str
183      """Determine the HTML document's base URL. 
184   
185      This looks for a ``<base>`` tag in the HTML document. If present, its href 
186      attribute denotes the base URL of anchor tags in the document. If there is 
187      no such tag (or if it does not have a valid href attribute), the HTML 
188      file's URL is used as the base URL. 
189   
190      :param document: An HTML document representation. The current 
191          implementation expects the result of ``html5lib.parse()``. 
192      :param page_url: The URL of the HTML document. 
193      """
194      for base in document.findall(".//base"):
195          href = base.get("href")
196          if href is not None:
197              return href
198      return page_url

201  def _clean_url_path_part(part):
202      # type: (str) -> str
203      """ 
204      Clean a "part" of a URL path (i.e. after splitting on "@" characters). 
205      """
206      # We unquote prior to quoting to make sure nothing is double quoted.
207      return urllib_parse.quote(urllib_parse.unquote(part))

210  def _clean_file_url_path(part):
211      # type: (str) -> str
212      """ 
213      Clean the first part of a URL path that corresponds to a local 
214      filesystem path (i.e. the first part after splitting on "@" characters). 
215      """
216      # We unquote prior to quoting to make sure nothing is double quoted.
217      # Also, on Windows the path part might contain a drive letter which
218      # should not be quoted. On Linux where drive letters do not
219      # exist, the colon should be quoted. We rely on urllib.request
220      # to do the right thing here.
221      return urllib_request.pathname2url(urllib_request.url2pathname(part))

224  # percent-encoded:                   /
225  _reserved_chars_re = re.compile('(@|%2F)', re.IGNORECASE)
226  
227  
228  def _clean_url_path(path, is_local_path):
229      # type: (str, bool) -> str
230      """ 
231      Clean the path portion of a URL. 
232      """
233      if is_local_path:
234          clean_func = _clean_file_url_path
235      else:
236          clean_func = _clean_url_path_part
237  
238      # Split on the reserved characters prior to cleaning so that
239      # revision strings in VCS URLs are properly preserved.
240      parts = _reserved_chars_re.split(path)
241  
242      cleaned_parts = []
243      for to_clean, reserved in pairwise(itertools.chain(parts, [''])):
244          cleaned_parts.append(clean_func(to_clean))
245          # Normalize %xx escapes (e.g. %2f -> %2F)
246          cleaned_parts.append(reserved.upper())
247  
248      return ''.join(cleaned_parts)

251  def _clean_link(url):
252      # type: (str) -> str
253      """ 
254      Make sure a link is fully quoted. 
255      For example, if ' ' occurs in the URL, it will be replaced with "%20", 
256      and without double-quoting other characters. 
257      """
258      # Split the URL into parts according to the general structure
259      # `scheme://netloc/path;parameters?query#fragment`.
260      result = urllib_parse.urlparse(url)
261      # If the netloc is empty, then the URL refers to a local filesystem path.
262      is_local_path = not result.netloc
263      path = _clean_url_path(result.path, is_local_path=is_local_path)
264      return urllib_parse.urlunparse(result._replace(path=path))

267  def _create_link_from_element(
268      anchor,    # type: HTMLElement
269      page_url,  # type: str
270      base_url,  # type: str
271  ):
272      # type: (...) -> Optional[Link]
273      """ 
274      Convert an anchor element in a simple repository page to a Link. 
275      """
276      href = anchor.get("href")
277      if not href:
278          return None
279  
280      url = _clean_link(urllib_parse.urljoin(base_url, href))
281      pyrequire = anchor.get('data-requires-python')
282      pyrequire = unescape(pyrequire) if pyrequire else None
283  
284      yanked_reason = anchor.get('data-yanked')
285      if yanked_reason:
286          # This is a unicode string in Python 2 (and 3).
287          yanked_reason = unescape(yanked_reason)
288  
289      link = Link(
290          url,
291          comes_from=page_url,
292          requires_python=pyrequire,
293          yanked_reason=yanked_reason,
294      )
295  
296      return link

315  def with_cached_html_pages(
316      fn,    # type: Callable[[HTMLPage], Iterable[Link]]
317  ):
318      # type: (...) -> Callable[[HTMLPage], List[Link]]
319      """ 
320      Given a function that parses an Iterable[Link] from an HTMLPage, cache the 
321      function's result (keyed by CacheablePageContent), unless the HTMLPage 
322      `page` has `page.cache_link_parsing == False`. 
323      """
324  
325      @lru_cache(maxsize=None)
326      def wrapper(cacheable_page):
327          # type: (CacheablePageContent) -> List[Link]
328          return list(fn(cacheable_page.page))
329  
330      @functools.wraps(fn)
331      def wrapper_wrapper(page):
332          # type: (HTMLPage) -> List[Link]
333          if page.cache_link_parsing:
334              return wrapper(CacheablePageContent(page))
335          return list(fn(page))
336  
337      return wrapper_wrapper

340  @with_cached_html_pages
341  def parse_links(page):
342      # type: (HTMLPage) -> Iterable[Link]
343      """ 
344      Parse an HTML document, and yield its anchor elements as Link objects. 
345      """
346      document = html5lib.parse(
347          page.content,
348          transport_encoding=page.encoding,
349          namespaceHTMLElements=False,
350      )
351  
352      url = page.url
353      base_url = _determine_base_url(document, url)
354      for anchor in document.findall(".//a"):
355          link = _create_link_from_element(
356              anchor,
357              page_url=url,
358              base_url=base_url,
359          )
360          if link is None:
361              continue
362          yield link

414  def _get_html_page(link, session=None):
415      # type: (Link, Optional[PipSession]) -> Optional[HTMLPage]
416      if session is None:
417          raise TypeError(
418              "_get_html_page() missing 1 required keyword argument: 'session'"
419          )
420  
421      url = link.url.split('#', 1)[0]
422  
423      # Check for VCS schemes that do not support lookup as web pages.
424      vcs_scheme = _match_vcs_scheme(url)
425      if vcs_scheme:
426          logger.warning('Cannot look at %s URL %s because it does not support '
427                         'lookup as web pages.', vcs_scheme, link)
428          return None
429  
430      # Tack index.html onto file:// URLs that point to directories
431      scheme, _, path, _, _, _ = urllib_parse.urlparse(url)
432      if (scheme == 'file' and os.path.isdir(urllib_request.url2pathname(path))):
433          # add trailing slash if not present so urljoin doesn't trim
434          # final segment
435          if not url.endswith('/'):
436              url += '/'
437          url = urllib_parse.urljoin(url, 'index.html')
438          logger.debug(' file: URL is directory, getting %s', url)
439  
440      try:
441          resp = _get_html_response(url, session=session)
442      except _NotHTTP:
443          logger.warning(
444              'Skipping page %s because it looks like an archive, and cannot '
445              'be checked by a HTTP HEAD request.', link,
446          )
447      except _NotHTML as exc:
448          logger.warning(
449              'Skipping page %s because the %s request got Content-Type: %s.'
450              'The only supported Content-Type is text/html',
451              link, exc.request_desc, exc.content_type,
452          )
453      except NetworkConnectionError as exc:
454          _handle_get_page_fail(link, exc)
455      except RetryError as exc:
456          _handle_get_page_fail(link, exc)
457      except SSLError as exc:
458          reason = "There was a problem confirming the ssl certificate: "
459          reason += str(exc)
460          _handle_get_page_fail(link, reason, meth=logger.info)
461      except requests.ConnectionError as exc:
462          _handle_get_page_fail(link, "connection error: {}".format(exc))
463      except requests.Timeout:
464          _handle_get_page_fail(link, "timed out")
465      else:
466          return _make_html_page(resp,
467                                 cache_link_parsing=link.cache_link_parsing)
468      return None

480  def group_locations(locations, expand_dir=False):
481      # type: (Sequence[str], bool) -> Tuple[List[str], List[str]]
482      """ 
483      Divide a list of locations into two groups: "files" (archives) and "urls." 
484   
485      :return: A pair of lists (files, urls). 
486      """
487      files = []
488      urls = []
489  
490      # puts the url for the given file path into the appropriate list
491      def sort_path(path):
492          # type: (str) -> None
493          url = path_to_url(path)
494          if mimetypes.guess_type(url, strict=False)[0] == 'text/html':
495              urls.append(url)
496          else:
497              files.append(url)
498  
499      for url in locations:
500  
501          is_local_path = os.path.exists(url)
502          is_file_url = url.startswith('file:')
503  
504          if is_local_path or is_file_url:
505              if is_local_path:
506                  path = url
507              else:
508                  path = url_to_path(url)
509              if os.path.isdir(path):
510                  if expand_dir:
511                      path = os.path.realpath(path)
512                      for item in os.listdir(path):
513                          sort_path(os.path.join(path, item))
514                  elif is_file_url:
515                      urls.append(url)
516                  else:
517                      logger.warning(
518                          "Path '%s' is ignored: it is a directory.", path,
519                      )
520              elif os.path.isfile(path):
521                  sort_path(path)
522              else:
523                  logger.warning(
524                      "Url '%s' is ignored: it is neither a file "
525                      "nor a directory.", url,
526                  )
527          elif is_url(url):
528              # Only add url with clear scheme
529              urls.append(url)
530          else:
531              logger.warning(
532                  "Url '%s' is ignored. It is either a non-existing "
533                  "path or lacks a specific scheme.", url,
534              )
535  
536      return files, urls

539  class CollectedLinks(object):
540  
541      """ 
542      Encapsulates the return value of a call to LinkCollector.collect_links(). 
543   
544      The return value includes both URLs to project pages containing package 
545      links, as well as individual package Link objects collected from other 
546      sources. 
547   
548      This info is stored separately as: 
549   
550      (1) links from the configured file locations, 
551      (2) links from the configured find_links, and 
552      (3) urls to HTML project pages, as described by the PEP 503 simple 
553          repository API. 
554      """
555  
556      def __init__(
557          self,
558          files,         # type: List[Link]
559          find_links,    # type: List[Link]
560          project_urls,  # type: List[Link]
561      ):
562          # type: (...) -> None
563          """ 
564          :param files: Links from file locations. 
565          :param find_links: Links from find_links. 
566          :param project_urls: URLs to HTML project pages, as described by 
567              the PEP 503 simple repository API. 
568          """
569          self.files = files
570          self.find_links = find_links
571          self.project_urls = project_urls

574  class LinkCollector(object):
575  
576      """ 
577      Responsible for collecting Link objects from all configured locations, 
578      making network requests as needed. 
579   
580      The class's main method is its collect_links() method. 
581      """
582  
583      def __init__(
584          self,
585          session,       # type: PipSession
586          search_scope,  # type: SearchScope
587      ):
588          # type: (...) -> None
589          self.search_scope = search_scope
590          self.session = session
591  
592      @classmethod
593      def create(cls, session, options, suppress_no_index=False):
594          # type: (PipSession, Values, bool) -> LinkCollector
595          """ 
596          :param session: The Session to use to make requests. 
597          :param suppress_no_index: Whether to ignore the --no-index option 
598              when constructing the SearchScope object. 
599          """
600          index_urls = [options.index_url] + options.extra_index_urls
601          if options.no_index and not suppress_no_index:
602              logger.debug(
603                  'Ignoring indexes: %s',
604                  ','.join(redact_auth_from_url(url) for url in index_urls),
605              )
606              index_urls = []
607  
608          # Make sure find_links is a list before passing to create().
609          find_links = options.find_links or []
610  
611          search_scope = SearchScope.create(
612              find_links=find_links, index_urls=index_urls,
613          )
614          link_collector = LinkCollector(
615              session=session, search_scope=search_scope,
616          )
617          return link_collector
618  
619      @property
620      def find_links(self):
621          # type: () -> List[str]
622          return self.search_scope.find_links
623  
624      def fetch_page(self, location):
625          # type: (Link) -> Optional[HTMLPage]
626          """ 
627          Fetch an HTML page containing package links. 
628          """
629          return _get_html_page(location, session=self.session)
630  
631      def collect_links(self, project_name):
632          # type: (str) -> CollectedLinks
633          """Find all available links for the given project name. 
634   
635          :return: All the Link objects (unfiltered), as a CollectedLinks object. 
636          """
637          search_scope = self.search_scope
638          index_locations = search_scope.get_index_urls_locations(project_name)
639          index_file_loc, index_url_loc = group_locations(index_locations)
640          fl_file_loc, fl_url_loc = group_locations(
641              self.find_links, expand_dir=True,
642          )
643  
644          file_links = [
645              Link(url) for url in itertools.chain(index_file_loc, fl_file_loc)
646          ]
647  
648          # We trust every directly linked archive in find_links
649          find_link_links = [Link(url, '-f') for url in self.find_links]
650  
651          # We trust every url that the user has given us whether it was given
652          # via --index-url or --find-links.
653          # We want to filter out anything that does not have a secure origin.
654          url_locations = [
655              link for link in itertools.chain(
656                  # Mark PyPI indices as "cache_link_parsing == False" -- this
657                  # will avoid caching the result of parsing the page for links.
658                  (Link(url, cache_link_parsing=False) for url in index_url_loc),
659                  (Link(url) for url in fl_url_loc),
660              )
661              if self.session.is_secure_origin(link)
662          ]
663  
664          url_locations = _remove_duplicate_links(url_locations)
665          lines = [
666              '{} location(s) to search for versions of {}:'.format(
667                  len(url_locations), project_name,
668              ),
669          ]
670          for link in url_locations:
671              lines.append('* {}'.format(link))
672          logger.debug('\n'.join(lines))
673  
674          return CollectedLinks(
675              files=file_links,
676              find_links=find_link_links,
677              project_urls=url_locations,
678          )
