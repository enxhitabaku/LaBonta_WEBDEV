1    """ 
2    This module holds simple classes to convert geospatial values from the 
3    database. 
4    """
5    from decimal import Decimal
6    
7    from django.contrib.gis.measure import Area, Distance
8    from django.db import models
9    
10   
11   class AreaField(models.FloatField):
12       "Wrapper for Area values."
13       def __init__(self, geo_field):
14           super().__init__()
15           self.geo_field = geo_field
16   
17       def get_prep_value(self, value):
18           if not isinstance(value, Area):
19               raise ValueError('AreaField only accepts Area measurement objects.')
20           return value
21   
22       def get_db_prep_value(self, value, connection, prepared=False):
23           if value is None:
24               return
25           area_att = connection.ops.get_area_att_for_field(self.geo_field)
26           return getattr(value, area_att) if area_att else value
27   
28       def from_db_value(self, value, expression, connection):
29           if value is None:
30               return
31           # If the database returns a Decimal, convert it to a float as expected
32           # by the Python geometric objects.
33           if isinstance(value, Decimal):
34               value = float(value)
35           # If the units are known, convert value into area measure.
36           area_att = connection.ops.get_area_att_for_field(self.geo_field)
37           return Area(**{area_att: value}) if area_att else value
38   
39       def get_internal_type(self):
40           return 'AreaField'
41   
42   
43   class DistanceField(models.FloatField):
44       "Wrapper for Distance values."
45       def __init__(self, geo_field):
46           super().__init__()
47           self.geo_field = geo_field
48   
49       def get_prep_value(self, value):
50           if isinstance(value, Distance):
51               return value
52           return super().get_prep_value(value)
53   
54       def get_db_prep_value(self, value, connection, prepared=False):
55           if not isinstance(value, Distance):
56               return value
57           distance_att = connection.ops.get_distance_att_for_field(self.geo_field)
58           if not distance_att:
59               raise ValueError('Distance measure is supplied, but units are unknown for result.')
60           return getattr(value, distance_att)
61   
62       def from_db_value(self, value, expression, connection):
63           if value is None:
64               return
65           distance_att = connection.ops.get_distance_att_for_field(self.geo_field)
66           return Distance(**{distance_att: value}) if distance_att else value
67   
68       def get_internal_type(self):
69           return 'DistanceField'
