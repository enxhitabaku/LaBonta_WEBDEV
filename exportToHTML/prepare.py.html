80   def _get_prepared_distribution(
81           req,  # type: InstallRequirement
82           req_tracker,  # type: RequirementTracker
83           finder,  # type: PackageFinder
84           build_isolation  # type: bool
85   ):
86       # type: (...) -> AbstractDistribution
87       """Prepare a distribution for installation. 
88       """
89       abstract_dist = make_distribution_for_install_requirement(req)
90       with req_tracker.track(req):
91           abstract_dist.prepare_distribution_metadata(finder, build_isolation)
92       return abstract_dist

95   def unpack_vcs_link(link, location):
96       # type: (Link, str) -> None
97       vcs_backend = vcs.get_backend_for_scheme(link.scheme)
98       assert vcs_backend is not None
99       vcs_backend.unpack(location, url=hide_url(link.url))

109  def get_http_url(
110      link,  # type: Link
111      downloader,  # type: Downloader
112      download_dir=None,  # type: Optional[str]
113      hashes=None,  # type: Optional[Hashes]
114  ):
115      # type: (...) -> File
116      temp_dir = TempDirectory(kind="unpack", globally_managed=True)
117      # If a download dir is specified, is the file already downloaded there?
118      already_downloaded_path = None
119      if download_dir:
120          already_downloaded_path = _check_download_dir(
121              link, download_dir, hashes
122          )
123  
124      if already_downloaded_path:
125          from_path = already_downloaded_path
126          content_type = mimetypes.guess_type(from_path)[0]
127      else:
128          # let's download to a tmp dir
129          from_path, content_type = _download_http_url(
130              link, downloader, temp_dir.path, hashes
131          )
132  
133      return File(from_path, content_type)

136  def _copy2_ignoring_special_files(src, dest):
137      # type: (str, str) -> None
138      """Copying special files is not supported, but as a convenience to users 
139      we skip errors copying them. This supports tools that may create e.g. 
140      socket files in the project source directory. 
141      """
142      try:
143          copy2_fixed(src, dest)
144      except shutil.SpecialFileError as e:
145          # SpecialFileError may be raised due to either the source or
146          # destination. If the destination was the cause then we would actually
147          # care, but since the destination directory is deleted prior to
148          # copy we ignore all of them assuming it is caused by the source.
149          logger.warning(
150              "Ignoring special file error '%s' encountered copying %s to %s.",
151              str(e),
152              path_to_display(src),
153              path_to_display(dest),
154          )

157  def _copy_source_tree(source, target):
158      # type: (str, str) -> None
159      target_abspath = os.path.abspath(target)
160      target_basename = os.path.basename(target_abspath)
161      target_dirname = os.path.dirname(target_abspath)
162  
163      def ignore(d, names):
164          # type: (str, List[str]) -> List[str]
165          skipped = []  # type: List[str]
166          if d == source:
167              # Pulling in those directories can potentially be very slow,
168              # exclude the following directories if they appear in the top
169              # level dir (and only it).
170              # See discussion at https://github.com/pypa/pip/pull/6770
171              skipped += ['.tox', '.nox']
172          if os.path.abspath(d) == target_dirname:
173              # Prevent an infinite recursion if the target is in source.
174              # This can happen when TMPDIR is set to ${PWD}/...
175              # and we copy PWD to TMPDIR.
176              skipped += [target_basename]
177          return skipped
178  
179      kwargs = dict(ignore=ignore, symlinks=True)  # type: CopytreeKwargs
180  
181      if not PY2:
182          # Python 2 does not support copy_function, so we only ignore
183          # errors on special file copy in Python 3.
184          kwargs['copy_function'] = _copy2_ignoring_special_files
185  
186      shutil.copytree(source, target, **kwargs)

189  def get_file_url(
190      link,  # type: Link
191      download_dir=None,  # type: Optional[str]
192      hashes=None  # type: Optional[Hashes]
193  ):
194      # type: (...) -> File
195      """Get file and optionally check its hash. 
196      """
197      # If a download dir is specified, is the file already there and valid?
198      already_downloaded_path = None
199      if download_dir:
200          already_downloaded_path = _check_download_dir(
201              link, download_dir, hashes
202          )
203  
204      if already_downloaded_path:
205          from_path = already_downloaded_path
206      else:
207          from_path = link.file_path
208  
209      # If --require-hashes is off, `hashes` is either empty, the
210      # link's embedded hash, or MissingHashes; it is required to
211      # match. If --require-hashes is on, we are satisfied by any
212      # hash in `hashes` matching: a URL-based or an option-based
213      # one; no internet-sourced hash will be in `hashes`.
214      if hashes:
215          hashes.check_against_path(from_path)
216  
217      content_type = mimetypes.guess_type(from_path)[0]
218  
219      return File(from_path, content_type)

222  def unpack_url(
223      link,  # type: Link
224      location,  # type: str
225      downloader,  # type: Downloader
226      download_dir=None,  # type: Optional[str]
227      hashes=None,  # type: Optional[Hashes]
228  ):
229      # type: (...) -> Optional[File]
230      """Unpack link into location, downloading if required. 
231   
232      :param hashes: A Hashes object, one of whose embedded hashes must match, 
233          or HashMismatch will be raised. If the Hashes is empty, no matches are 
234          required, and unhashable types of requirements (like VCS ones, which 
235          would ordinarily raise HashUnsupported) are allowed. 
236      """
237      # non-editable vcs urls
238      if link.is_vcs:
239          unpack_vcs_link(link, location)
240          return None
241  
242      # If it's a url to a local directory
243      if link.is_existing_dir():
244          if os.path.isdir(location):
245              rmtree(location)
246          _copy_source_tree(link.file_path, location)
247          return None
248  
249      # file urls
250      if link.is_file:
251          file = get_file_url(link, download_dir, hashes=hashes)
252  
253      # http urls
254      else:
255          file = get_http_url(
256              link,
257              downloader,
258              download_dir,
259              hashes=hashes,
260          )
261  
262      # unpack the archive to the build dir location. even when only downloading
263      # archives, they have to be unpacked to parse dependencies, except wheels
264      if not link.is_wheel:
265          unpack_file(file.path, location, file.content_type)
266  
267      return file

270  def _download_http_url(
271      link,  # type: Link
272      downloader,  # type: Downloader
273      temp_dir,  # type: str
274      hashes,  # type: Optional[Hashes]
275  ):
276      # type: (...) -> Tuple[str, str]
277      """Download link url into temp_dir using provided session"""
278      download = downloader(link)
279  
280      file_path = os.path.join(temp_dir, download.filename)
281      with open(file_path, 'wb') as content_file:
282          for chunk in download.chunks:
283              content_file.write(chunk)
284  
285      if hashes:
286          hashes.check_against_path(file_path)
287  
288      return file_path, download.response.headers.get('content-type', '')

291  def _check_download_dir(link, download_dir, hashes):
292      # type: (Link, str, Optional[Hashes]) -> Optional[str]
293      """ Check download_dir for previously downloaded file with correct hash 
294          If a correct file is found return its path else None 
295      """
296      download_path = os.path.join(download_dir, link.filename)
297  
298      if not os.path.exists(download_path):
299          return None
300  
301      # If already downloaded, does its hash match?
302      logger.info('File was already downloaded %s', download_path)
303      if hashes:
304          try:
305              hashes.check_against_path(download_path)
306          except HashMismatch:
307              logger.warning(
308                  'Previously-downloaded file %s has bad hash. '
309                  'Re-downloading.',
310                  download_path
311              )
312              os.unlink(download_path)
313              return None
314      return download_path

317  class RequirementPreparer(object):
318      """Prepares a Requirement 
319      """
320  
321      def __init__(
322          self,
323          build_dir,  # type: str
324          download_dir,  # type: Optional[str]
325          src_dir,  # type: str
326          wheel_download_dir,  # type: Optional[str]
327          build_isolation,  # type: bool
328          req_tracker,  # type: RequirementTracker
329          downloader,  # type: Downloader
330          finder,  # type: PackageFinder
331          require_hashes,  # type: bool
332          use_user_site,  # type: bool
333      ):
334          # type: (...) -> None
335          super(RequirementPreparer, self).__init__()
336  
337          self.src_dir = src_dir
338          self.build_dir = build_dir
339          self.req_tracker = req_tracker
340          self.downloader = downloader
341          self.finder = finder
342  
343          # Where still-packed archives should be written to. If None, they are
344          # not saved, and are deleted immediately after unpacking.
345          self.download_dir = download_dir
346  
347          # Where still-packed .whl files should be written to. If None, they are
348          # written to the download_dir parameter. Separate to download_dir to
349          # permit only keeping wheel archives for pip wheel.
350          self.wheel_download_dir = wheel_download_dir
351  
352          # NOTE
353          # download_dir and wheel_download_dir overlap semantically and may
354          # be combined if we're willing to have non-wheel archives present in
355          # the wheelhouse output by 'pip wheel'.
356  
357          # Is build isolation allowed?
358          self.build_isolation = build_isolation
359  
360          # Should hash-checking be required?
361          self.require_hashes = require_hashes
362  
363          # Should install in user site-packages?
364          self.use_user_site = use_user_site
365  
366      @property
367      def _download_should_save(self):
368          # type: () -> bool
369          if not self.download_dir:
370              return False
371  
372          if os.path.exists(self.download_dir):
373              return True
374  
375          logger.critical('Could not find download directory')
376          raise InstallationError(
377              "Could not find or access download directory '{}'"
378              .format(self.download_dir))
379  
380      def _log_preparing_link(self, req):
381          # type: (InstallRequirement) -> None
382          """Log the way the link prepared."""
383          if req.link.is_file:
384              path = req.link.file_path
385              logger.info('Processing %s', display_path(path))
386          else:
387              logger.info('Collecting %s', req.req or req)
388  
389      def _ensure_link_req_src_dir(self, req, download_dir, parallel_builds):
390          # type: (InstallRequirement, Optional[str], bool) -> None
391          """Ensure source_dir of a linked InstallRequirement."""
392          # Since source_dir is only set for editable requirements.
393          if req.link.is_wheel:
394              # We don't need to unpack wheels, so no need for a source
395              # directory.
396              return
397          assert req.source_dir is None
398          # We always delete unpacked sdists after pip runs.
399          req.ensure_has_source_dir(
400              self.build_dir,
401              autodelete=True,
402              parallel_builds=parallel_builds,
403          )
404  
405          # If a checkout exists, it's unwise to keep going.  version
406          # inconsistencies are logged later, but do not fail the
407          # installation.
408          # FIXME: this won't upgrade when there's an existing
409          # package unpacked in `req.source_dir`
410          if os.path.exists(os.path.join(req.source_dir, 'setup.py')):
411              raise PreviousBuildDirError(
412                  "pip can't proceed with requirements '{}' due to a"
413                  "pre-existing build directory ({}). This is likely "
414                  "due to a previous installation that failed . pip is "
415                  "being responsible and not assuming it can delete this. "
416                  "Please delete it and try again.".format(req, req.source_dir)
417              )
418  
419      def _get_linked_req_hashes(self, req):
420          # type: (InstallRequirement) -> Hashes
421          # By the time this is called, the requirement's link should have
422          # been checked so we can tell what kind of requirements req is
423          # and raise some more informative errors than otherwise.
424          # (For example, we can raise VcsHashUnsupported for a VCS URL
425          # rather than HashMissing.)
426          if not self.require_hashes:
427              return req.hashes(trust_internet=True)
428  
429          # We could check these first 2 conditions inside unpack_url
430          # and save repetition of conditions, but then we would
431          # report less-useful error messages for unhashable
432          # requirements, complaining that there's no hash provided.
433          if req.link.is_vcs:
434              raise VcsHashUnsupported()
435          if req.link.is_existing_dir():
436              raise DirectoryUrlHashUnsupported()
437  
438          # Unpinned packages are asking for trouble when a new version
439          # is uploaded.  This isn't a security check, but it saves users
440          # a surprising hash mismatch in the future.
441          # file:/// URLs aren't pinnable, so don't complain about them
442          # not being pinned.
443          if req.original_link is None and not req.is_pinned:
444              raise HashUnpinned()
445  
446          # If known-good hashes are missing for this requirement,
447          # shim it with a facade object that will provoke hash
448          # computation and then raise a HashMissing exception
449          # showing the user what the hash should be.
450          return req.hashes(trust_internet=False) or MissingHashes()

452      def prepare_linked_requirement(self, req, parallel_builds=False):
453          # type: (InstallRequirement, bool) -> AbstractDistribution
454          """Prepare a requirement to be obtained from req.link."""
455          assert req.link
456          link = req.link
457          self._log_preparing_link(req)
458          if link.is_wheel and self.wheel_download_dir:
459              # Download wheels to a dedicated dir when doing `pip wheel`.
460              download_dir = self.wheel_download_dir
461          else:
462              download_dir = self.download_dir
463  
464          with indent_log():
465              self._ensure_link_req_src_dir(req, download_dir, parallel_builds)
466              try:
467                  local_file = unpack_url(
468                      link, req.source_dir, self.downloader, download_dir,
469                      hashes=self._get_linked_req_hashes(req)
470                  )
471              except NetworkConnectionError as exc:
472                  raise InstallationError(
473                      'Could not install requirement {} because of HTTP '
474                      'error {} for URL {}'.format(req, exc, link)
475                  )
476  
477              # For use in later processing, preserve the file path on the
478              # requirement.
479              if local_file:
480                  req.local_file_path = local_file.path
481  
482              abstract_dist = _get_prepared_distribution(
483                  req, self.req_tracker, self.finder, self.build_isolation,
484              )
485  
486              if download_dir:
487                  if link.is_existing_dir():
488                      logger.info('Link is a directory, ignoring download_dir')
489                  elif local_file:
490                      download_location = os.path.join(
491                          download_dir, link.filename
492                      )
493                      if not os.path.exists(download_location):
494                          shutil.copy(local_file.path, download_location)
495                          download_path = display_path(download_location)
496                          logger.info('Saved %s', download_path)
497  
498              if self._download_should_save:
499                  # Make a .zip of the source_dir we already created.
500                  if link.is_vcs:
501                      req.archive(self.download_dir)
502          return abstract_dist
503  
504      def prepare_editable_requirement(
505          self,
506          req,  # type: InstallRequirement
507      ):
508          # type: (...) -> AbstractDistribution
509          """Prepare an editable requirement 
510          """
511          assert req.editable, "cannot prepare a non-editable req as editable"
512  
513          logger.info('Obtaining %s', req)
514  
515          with indent_log():
516              if self.require_hashes:
517                  raise InstallationError(
518                      'The editable requirement {} cannot be installed when '
519                      'requiring hashes, because there is no single file to '
520                      'hash.'.format(req)
521                  )
522              req.ensure_has_source_dir(self.src_dir)
523              req.update_editable(not self._download_should_save)
524  
525              abstract_dist = _get_prepared_distribution(
526                  req, self.req_tracker, self.finder, self.build_isolation,
527              )
528  
529              if self._download_should_save:
530                  req.archive(self.download_dir)
531              req.check_if_exists(self.use_user_site)
532  
533          return abstract_dist
534  
535      def prepare_installed_requirement(
536          self,
537          req,  # type: InstallRequirement
538          skip_reason  # type: str
539      ):
540          # type: (...) -> AbstractDistribution
541          """Prepare an already-installed requirement 
542          """
543          assert req.satisfied_by, "req should have been satisfied but isn't"
544          assert skip_reason is not None, (
545              "did not get skip reason skipped but req.satisfied_by "
546              "is set to {}".format(req.satisfied_by)
547          )
548          logger.info(
549              'Requirement %s: %s (%s)',
550              skip_reason, req, req.satisfied_by.version
551          )
552          with indent_log():
553              if self.require_hashes:
554                  logger.debug(
555                      'Since it is already installed, we are trusting this '
556                      'package without checking its hash. To ensure a '
557                      'completely repeatable environment, install into an '
558                      'empty virtualenv.'
559                  )
560              abstract_dist = InstalledDistribution(req)
561  
562          return abstract_dist
