62       def is_immutable_rev_checkout(self, url, dest):
63           # type: (str, str) -> bool
64           _, rev_options = self.get_url_rev_options(hide_url(url))
65           if not rev_options.rev:
66               return False
67           if not self.is_commit_id_equal(dest, rev_options.rev):
68               # the current commit is different from rev,
69               # which means rev was something else than a commit hash
70               return False
71           # return False in the rare case rev is both a commit hash
72           # and a tag or a branch; we don't want to cache in that case
73           # because that branch/tag could point to something else in the future
74           is_tag_or_branch = bool(
75               self.get_revision_sha(dest, rev_options.rev)[0]
76           )
77           return not is_tag_or_branch

79       def get_git_version(self):
80           VERSION_PFX = 'git version '
81           version = self.run_command(['version'])
82           if version.startswith(VERSION_PFX):
83               version = version[len(VERSION_PFX):].split()[0]
84           else:
85               version = ''
86           # get first 3 positions of the git version because
87           # on windows it is x.y.z.windows.t, and this parses as
88           # LegacyVersion which always smaller than a Version.
89           version = '.'.join(version.split('.')[:3])
90           return parse_version(version)

92       @classmethod
93       def get_current_branch(cls, location):
94           """ 
95           Return the current branch, or None if HEAD isn't at a branch 
96           (e.g. detached HEAD). 
97           """
98           # git-symbolic-ref exits with empty stdout if "HEAD" is a detached
99           # HEAD rather than a symbolic ref.  In addition, the -q causes the
100          # command to exit with status code 1 instead of 128 in this case
101          # and to suppress the message to stderr.
102          args = ['symbolic-ref', '-q', 'HEAD']
103          output = cls.run_command(
104              args, extra_ok_returncodes=(1, ), cwd=location,
105          )
106          ref = output.strip()
107  
108          if ref.startswith('refs/heads/'):
109              return ref[len('refs/heads/'):]
110  
111          return None

113      def export(self, location, url):
114          # type: (str, HiddenText) -> None
115          """Export the Git repository at the url to the destination location"""
116          if not location.endswith('/'):
117              location = location + '/'
118  
119          with TempDirectory(kind="export") as temp_dir:
120              self.unpack(temp_dir.path, url=url)
121              self.run_command(
122                  ['checkout-index', '-a', '-f', '--prefix', location],
123                  cwd=temp_dir.path
124              )

126      @classmethod
127      def get_revision_sha(cls, dest, rev):
128          """ 
129          Return (sha_or_none, is_branch), where sha_or_none is a commit hash 
130          if the revision names a remote branch or tag, otherwise None. 
131   
132          Args: 
133            dest: the repository directory. 
134            rev: the revision name. 
135          """
136          # Pass rev to pre-filter the list.
137  
138          output = ''
139          try:
140              output = cls.run_command(['show-ref', rev], cwd=dest)
141          except SubProcessError:
142              pass
143  
144          refs = {}
145          for line in output.strip().splitlines():
146              try:
147                  sha, ref = line.split()
148              except ValueError:
149                  # Include the offending line to simplify troubleshooting if
150                  # this error ever occurs.
151                  raise ValueError('unexpected show-ref line: {!r}'.format(line))
152  
153              refs[ref] = sha
154  
155          branch_ref = 'refs/remotes/origin/{}'.format(rev)
156          tag_ref = 'refs/tags/{}'.format(rev)
157  
158          sha = refs.get(branch_ref)
159          if sha is not None:
160              return (sha, True)
161  
162          sha = refs.get(tag_ref)
163  
164          return (sha, False)

166      @classmethod
167      def resolve_revision(cls, dest, url, rev_options):
168          # type: (str, HiddenText, RevOptions) -> RevOptions
169          """ 
170          Resolve a revision to a new RevOptions object with the SHA1 of the 
171          branch, tag, or ref if found. 
172   
173          Args: 
174            rev_options: a RevOptions object. 
175          """
176          rev = rev_options.arg_rev
177          # The arg_rev property's implementation for Git ensures that the
178          # rev return value is always non-None.
179          assert rev is not None
180  
181          sha, is_branch = cls.get_revision_sha(dest, rev)
182  
183          if sha is not None:
184              rev_options = rev_options.make_new(sha)
185              rev_options.branch_name = rev if is_branch else None
186  
187              return rev_options
188  
189          # Do not show a warning for the common case of something that has
190          # the form of a Git commit hash.
191          if not looks_like_hash(rev):
192              logger.warning(
193                  "Did not find branch or tag '%s', assuming revision or ref.",
194                  rev,
195              )
196  
197          if not rev.startswith('refs/'):
198              return rev_options
199  
200          # If it looks like a ref, we have to fetch it explicitly.
201          cls.run_command(
202              make_command('fetch', '-q', url, rev_options.to_args()),
203              cwd=dest,
204          )
205          # Change the revision to the SHA of the ref we fetched
206          sha = cls.get_revision(dest, rev='FETCH_HEAD')
207          rev_options = rev_options.make_new(sha)
208  
209          return rev_options

211      @classmethod
212      def is_commit_id_equal(cls, dest, name):
213          """ 
214          Return whether the current commit hash equals the given name. 
215   
216          Args: 
217            dest: the repository directory. 
218            name: a string name. 
219          """
220          if not name:
221              # Then avoid an unnecessary subprocess call.
222              return False
223  
224          return cls.get_revision(dest) == name

226      def fetch_new(self, dest, url, rev_options):
227          # type: (str, HiddenText, RevOptions) -> None
228          rev_display = rev_options.to_display()
229          logger.info('Cloning %s%s to %s', url, rev_display, display_path(dest))
230          self.run_command(make_command('clone', '-q', url, dest))
231  
232          if rev_options.rev:
233              # Then a specific revision was requested.
234              rev_options = self.resolve_revision(dest, url, rev_options)
235              branch_name = getattr(rev_options, 'branch_name', None)
236              if branch_name is None:
237                  # Only do a checkout if the current commit id doesn't match
238                  # the requested revision.
239                  if not self.is_commit_id_equal(dest, rev_options.rev):
240                      cmd_args = make_command(
241                          'checkout', '-q', rev_options.to_args(),
242                      )
243                      self.run_command(cmd_args, cwd=dest)
244              elif self.get_current_branch(dest) != branch_name:
245                  # Then a specific branch was requested, and that branch
246                  # is not yet checked out.
247                  track_branch = 'origin/{}'.format(branch_name)
248                  cmd_args = [
249                      'checkout', '-b', branch_name, '--track', track_branch,
250                  ]
251                  self.run_command(cmd_args, cwd=dest)
252  
253          #: repo may contain submodules
254          self.update_submodules(dest)

256      def switch(self, dest, url, rev_options):
257          # type: (str, HiddenText, RevOptions) -> None
258          self.run_command(
259              make_command('config', 'remote.origin.url', url),
260              cwd=dest,
261          )
262          cmd_args = make_command('checkout', '-q', rev_options.to_args())
263          self.run_command(cmd_args, cwd=dest)
264  
265          self.update_submodules(dest)

267      def update(self, dest, url, rev_options):
268          # type: (str, HiddenText, RevOptions) -> None
269          # First fetch changes from the default remote
270          if self.get_git_version() >= parse_version('1.9.0'):
271              # fetch tags in addition to everything else
272              self.run_command(['fetch', '-q', '--tags'], cwd=dest)
273          else:
274              self.run_command(['fetch', '-q'], cwd=dest)
275          # Then reset to wanted revision (maybe even origin/master)
276          rev_options = self.resolve_revision(dest, url, rev_options)
277          cmd_args = make_command('reset', '--hard', '-q', rev_options.to_args())
278          self.run_command(cmd_args, cwd=dest)
279          #: update submodules
280          self.update_submodules(dest)

282      @classmethod
283      def get_remote_url(cls, location):
284          """ 
285          Return URL of the first remote encountered. 
286   
287          Raises RemoteNotFoundError if the repository does not have a remote 
288          url configured. 
289          """
290          # We need to pass 1 for extra_ok_returncodes since the command
291          # exits with return code 1 if there are no matching lines.
292          stdout = cls.run_command(
293              ['config', '--get-regexp', r'remote\..*\.url'],
294              extra_ok_returncodes=(1, ), cwd=location,
295          )
296          remotes = stdout.splitlines()
297          try:
298              found_remote = remotes[0]
299          except IndexError:
300              raise RemoteNotFoundError
301  
302          for remote in remotes:
303              if remote.startswith('remote.origin.url '):
304                  found_remote = remote
305                  break
306          url = found_remote.split(' ')[1]
307          return url.strip()

309      @classmethod
310      def get_revision(cls, location, rev=None):
311          if rev is None:
312              rev = 'HEAD'
313          current_rev = cls.run_command(
314              ['rev-parse', rev], cwd=location,
315          )
316          return current_rev.strip()

318      @classmethod
319      def get_subdirectory(cls, location):
320          """ 
321          Return the path to setup.py, relative to the repo root. 
322          Return None if setup.py is in the repo root. 
323          """
324          # find the repo root
325          git_dir = cls.run_command(
326              ['rev-parse', '--git-dir'],
327              cwd=location).strip()
328          if not os.path.isabs(git_dir):
329              git_dir = os.path.join(location, git_dir)
330          repo_root = os.path.abspath(os.path.join(git_dir, '..'))
331          return find_path_to_setup_from_repo_root(location, repo_root)

333      @classmethod
334      def get_url_rev_and_auth(cls, url):
335          # type: (str) -> Tuple[str, Optional[str], AuthInfo]
336          """ 
337          Prefixes stub URLs like 'user@hostname:user/repo.git' with 'ssh://'. 
338          That's required because although they use SSH they sometimes don't 
339          work with a ssh:// scheme (e.g. GitHub). But we need a scheme for 
340          parsing. Hence we remove it again afterwards and return it as a stub. 
341          """
342          # Works around an apparent Git bug
343          # (see https://article.gmane.org/gmane.comp.version-control.git/146500)
344          scheme, netloc, path, query, fragment = urlsplit(url)
345          if scheme.endswith('file'):
346              initial_slashes = path[:-len(path.lstrip('/'))]
347              newpath = (
348                  initial_slashes +
349                  urllib_request.url2pathname(path)
350                  .replace('\\', '/').lstrip('/')
351              )
352              url = urlunsplit((scheme, netloc, newpath, query, fragment))
353              after_plus = scheme.find('+') + 1
354              url = scheme[:after_plus] + urlunsplit(
355                  (scheme[after_plus:], netloc, newpath, query, fragment),
356              )
357  
358          if '://' not in url:
359              assert 'file:' not in url
360              url = url.replace('git+', 'git+ssh://')
361              url, rev, user_pass = super(Git, cls).get_url_rev_and_auth(url)
362              url = url.replace('ssh://', '')
363          else:
364              url, rev, user_pass = super(Git, cls).get_url_rev_and_auth(url)
365  
366          return url, rev, user_pass

368      @classmethod
369      def update_submodules(cls, location):
370          if not os.path.exists(os.path.join(location, '.gitmodules')):
371              return
372          cls.run_command(
373              ['submodule', 'update', '--init', '--recursive', '-q'],
374              cwd=location,
375          )

377      @classmethod
378      def get_repository_root(cls, location):
379          loc = super(Git, cls).get_repository_root(location)
380          if loc:
381              return loc
382          try:
383              r = cls.run_command(
384                  ['rev-parse', '--show-toplevel'],
385                  cwd=location,
386                  log_failed_cmd=False,
387              )
388          except BadCommand:
389              logger.debug("could not determine if %s is under git control "
390                           "because git is not available", location)
391              return None
392          except SubProcessError:
393              return None
394          return os.path.normpath(r.rstrip('\r\n'))
