16   class BlockContext:
17       def __init__(self):
18           # Dictionary of FIFO queues.
19           self.blocks = defaultdict(list)
20   
21       def add_blocks(self, blocks):
22           for name, block in blocks.items():
23               self.blocks[name].insert(0, block)
24   
25       def pop(self, name):
26           try:
27               return self.blocks[name].pop()
28           except IndexError:
29               return None
30   
31       def push(self, name, block):
32           self.blocks[name].append(block)
33   
34       def get_block(self, name):
35           try:
36               return self.blocks[name][-1]
37           except IndexError:
38               return None

41   class BlockNode(Node):
42       def __init__(self, name, nodelist, parent=None):
43           self.name, self.nodelist, self.parent = name, nodelist, parent
44   
45       def __repr__(self):
46           return "<Block Node: %s. Contents: %r>" % (self.name, self.nodelist)
47   
48       def render(self, context):
49           block_context = context.render_context.get(BLOCK_CONTEXT_KEY)
50           with context.push():
51               if block_context is None:
52                   context['block'] = self
53                   result = self.nodelist.render(context)
54               else:
55                   push = block = block_context.pop(self.name)
56                   if block is None:
57                       block = self
58                   # Create new block so we can store context without thread-safety issues.
59                   block = type(self)(block.name, block.nodelist)
60                   block.context = context
61                   context['block'] = block
62                   result = block.nodelist.render(context)
63                   if push is not None:
64                       block_context.push(self.name, push)
65           return result
66   
67       def super(self):
68           if not hasattr(self, 'context'):
69               raise TemplateSyntaxError(
70                   "'%s' object has no attribute 'context'. Did you use "
71                   "{{ block.super }} in a base template?" % self.__class__.__name__
72               )
73           render_context = self.context.render_context
74           if (BLOCK_CONTEXT_KEY in render_context and
75                   render_context[BLOCK_CONTEXT_KEY].get_block(self.name) is not None):
76               return mark_safe(self.render(self.context))
77           return ''

80   class ExtendsNode(Node):
81       must_be_first = True
82       context_key = 'extends_context'
83   
84       def __init__(self, nodelist, parent_name, template_dirs=None):
85           self.nodelist = nodelist
86           self.parent_name = parent_name
87           self.template_dirs = template_dirs
88           self.blocks = {n.name: n for n in nodelist.get_nodes_by_type(BlockNode)}
89   
90       def __repr__(self):
91           return '<%s: extends %s>' % (self.__class__.__name__, self.parent_name.token)
92   
93       def find_template(self, template_name, context):
94           """ 
95           This is a wrapper around engine.find_template(). A history is kept in 
96           the render_context attribute between successive extends calls and 
97           passed as the skip argument. This enables extends to work recursively 
98           without extending the same template twice. 
99           """
100          history = context.render_context.setdefault(
101              self.context_key, [self.origin],
102          )
103          template, origin = context.template.engine.find_template(
104              template_name, skip=history,
105          )
106          history.append(origin)
107          return template
108  
109      def get_parent(self, context):
110          parent = self.parent_name.resolve(context)
111          if not parent:
112              error_msg = "Invalid template name in 'extends' tag: %r." % parent
113              if self.parent_name.filters or\
114                      isinstance(self.parent_name.var, Variable):
115                  error_msg += " Got this from the '%s' variable." %\
116                      self.parent_name.token
117              raise TemplateSyntaxError(error_msg)
118          if isinstance(parent, Template):
119              # parent is a django.template.Template
120              return parent
121          if isinstance(getattr(parent, 'template', None), Template):
122              # parent is a django.template.backends.django.Template
123              return parent.template
124          return self.find_template(parent, context)
125  
126      def render(self, context):
127          compiled_parent = self.get_parent(context)
128  
129          if BLOCK_CONTEXT_KEY not in context.render_context:
130              context.render_context[BLOCK_CONTEXT_KEY] = BlockContext()
131          block_context = context.render_context[BLOCK_CONTEXT_KEY]
132  
133          # Add the block nodes from this node to the block context
134          block_context.add_blocks(self.blocks)
135  
136          # If this block's parent doesn't have an extends node it is the root,
137          # and its block nodes also need to be added to the block context.
138          for node in compiled_parent.nodelist:
139              # The ExtendsNode has to be the first non-text node.
140              if not isinstance(node, TextNode):
141                  if not isinstance(node, ExtendsNode):
142                      blocks = {n.name: n for n in
143                                compiled_parent.nodelist.get_nodes_by_type(BlockNode)}
144                      block_context.add_blocks(blocks)
145                  break
146  
147          # Call Template._render explicitly so the parser context stays
148          # the same.
149          with context.render_context.push_state(compiled_parent, isolated_context=False):
150              return compiled_parent._render(context)

153  class IncludeNode(Node):
154      context_key = '__include_context'
155  
156      def __init__(self, template, *args, extra_context=None, isolated_context=False, **kwargs):
157          self.template = template
158          self.extra_context = extra_context or {}
159          self.isolated_context = isolated_context
160          super().__init__(*args, **kwargs)
161  
162      def render(self, context):
163          """ 
164          Render the specified template and context. Cache the template object 
165          in render_context to avoid reparsing and loading when used in a for 
166          loop. 
167          """
168          template = self.template.resolve(context)
169          # Does this quack like a Template?
170          if not callable(getattr(template, 'render', None)):
171              # If not, try the cache and select_template().
172              template_name = template or ()
173              if isinstance(template_name, str):
174                  template_name = (template_name,)
175              else:
176                  template_name = tuple(template_name)
177              cache = context.render_context.dicts[0].setdefault(self, {})
178              template = cache.get(template_name)
179              if template is None:
180                  template = context.template.engine.select_template(template_name)
181                  cache[template_name] = template
182          # Use the base.Template of a backends.django.Template.
183          elif hasattr(template, 'template'):
184              template = template.template
185          values = {
186              name: var.resolve(context)
187              for name, var in self.extra_context.items()
188          }
189          if self.isolated_context:
190              return template.render(context.new(values))
191          with context.push(**values):
192              return template.render(context)

195  @register.tag('block')
196  def do_block(parser, token):
197      """ 
198      Define a block that can be overridden by child templates. 
199      """
200      # token.split_contents() isn't useful here because this tag doesn't accept variable as arguments
201      bits = token.contents.split()
202      if len(bits) != 2:
203          raise TemplateSyntaxError("'%s' tag takes only one argument" % bits[0])
204      block_name = bits[1]
205      # Keep track of the names of BlockNodes found in this template, so we can
206      # check for duplication.
207      try:
208          if block_name in parser.__loaded_blocks:
209              raise TemplateSyntaxError("'%s' tag with name '%s' appears more than once" % (bits[0], block_name))
210          parser.__loaded_blocks.append(block_name)
211      except AttributeError:  # parser.__loaded_blocks isn't a list yet
212          parser.__loaded_blocks = [block_name]
213      nodelist = parser.parse(('endblock',))
214  
215      # This check is kept for backwards-compatibility. See #3100.
216      endblock = parser.next_token()
217      acceptable_endblocks = ('endblock', 'endblock %s' % block_name)
218      if endblock.contents not in acceptable_endblocks:
219          parser.invalid_block_tag(endblock, 'endblock', acceptable_endblocks)
220  
221      return BlockNode(block_name, nodelist)

224  def construct_relative_path(current_template_name, relative_name):
225      """ 
226      Convert a relative path (starting with './' or '../') to the full template 
227      name based on the current_template_name. 
228      """
229      if not relative_name.startswith(("'./", "'../", '"./', '"../')):
230          # relative_name is a variable or a literal that doesn't contain a
231          # relative path.
232          return relative_name
233  
234      new_name = posixpath.normpath(
235          posixpath.join(
236              posixpath.dirname(current_template_name.lstrip('/')),
237              relative_name.strip('\'"')
238          )
239      )
240      if new_name.startswith('../'):
241          raise TemplateSyntaxError(
242              "The relative path '%s' points outside the file hierarchy that "
243              "template '%s' is in." % (relative_name, current_template_name)
244          )
245      if current_template_name.lstrip('/') == new_name:
246          raise TemplateSyntaxError(
247              "The relative path '%s' was translated to template name '%s', the "
248              "same template in which the tag appears."
249              % (relative_name, current_template_name)
250          )
251      return '"%s"' % new_name

254  @register.tag('extends')
255  def do_extends(parser, token):
256      """ 
257      Signal that this template extends a parent template. 
258   
259      This tag may be used in two ways: ``{% extends "base" %}`` (with quotes) 
260      uses the literal value "base" as the name of the parent template to extend, 
261      or ``{% extends variable %}`` uses the value of ``variable`` as either the 
262      name of the parent template to extend (if it evaluates to a string) or as 
263      the parent template itself (if it evaluates to a Template object). 
264      """
265      bits = token.split_contents()
266      if len(bits) != 2:
267          raise TemplateSyntaxError("'%s' takes one argument" % bits[0])
268      bits[1] = construct_relative_path(parser.origin.template_name, bits[1])
269      parent_name = parser.compile_filter(bits[1])
270      nodelist = parser.parse()
271      if nodelist.get_nodes_by_type(ExtendsNode):
272          raise TemplateSyntaxError("'%s' cannot appear more than once in the same template" % bits[0])
273      return ExtendsNode(nodelist, parent_name)

276  @register.tag('include')
277  def do_include(parser, token):
278      """ 
279      Load a template and render it with the current context. You can pass 
280      additional context using keyword arguments. 
281   
282      Example:: 
283   
284          {% include "foo/some_include" %} 
285          {% include "foo/some_include" with bar="BAZZ!" baz="BING!" %} 
286   
287      Use the ``only`` argument to exclude the current context when rendering 
288      the included template:: 
289   
290          {% include "foo/some_include" only %} 
291          {% include "foo/some_include" with bar="1" only %} 
292      """
293      bits = token.split_contents()
294      if len(bits) < 2:
295          raise TemplateSyntaxError(
296              "%r tag takes at least one argument: the name of the template to "
297              "be included." % bits[0]
298          )
299      options = {}
300      remaining_bits = bits[2:]
301      while remaining_bits:
302          option = remaining_bits.pop(0)
303          if option in options:
304              raise TemplateSyntaxError('The %r option was specified more '
305                                        'than once.' % option)
306          if option == 'with':
307              value = token_kwargs(remaining_bits, parser, support_legacy=False)
308              if not value:
309                  raise TemplateSyntaxError('"with" in %r tag needs at least '
310                                            'one keyword argument.' % bits[0])
311          elif option == 'only':
312              value = True
313          else:
314              raise TemplateSyntaxError('Unknown argument for %r tag: %r.' %
315                                        (bits[0], option))
316          options[option] = value
317      isolated_context = options.get('only', False)
318      namemap = options.get('with', {})
319      bits[1] = construct_relative_path(parser.origin.template_name, bits[1])
320      return IncludeNode(parser.compile_filter(bits[1]), extra_context=namemap,
321                         isolated_context=isolated_context)
