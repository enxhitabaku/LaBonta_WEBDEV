30   class DirectUrlValidationError(Exception):
31       pass

34   def _get(d, expected_type, key, default=None):
35       # type: (Dict[str, Any], Type[T], str, Optional[T]) -> Optional[T]
36       """Get value from dictionary and verify expected type."""
37       if key not in d:
38           return default
39       value = d[key]
40       if six.PY2 and expected_type is str:
41           expected_type = six.string_types  # type: ignore
42       if not isinstance(value, expected_type):
43           raise DirectUrlValidationError(
44               "{!r} has unexpected type for {} (expected {})".format(
45                   value, key, expected_type
46               )
47           )
48       return value
49   
50   
51   def _get_required(d, expected_type, key, default=None):
52       # type: (Dict[str, Any], Type[T], str, Optional[T]) -> T
53       value = _get(d, expected_type, key, default)
54       if value is None:
55           raise DirectUrlValidationError("{} must have a value".format(key))
56       return value

59   def _exactly_one_of(infos):
60       # type: (Iterable[Optional[InfoType]]) -> InfoType
61       infos = [info for info in infos if info is not None]
62       if not infos:
63           raise DirectUrlValidationError(
64               "missing one of archive_info, dir_info, vcs_info"
65           )
66       if len(infos) > 1:
67           raise DirectUrlValidationError(
68               "more than one of archive_info, dir_info, vcs_info"
69           )
70       assert infos[0] is not None
71       return infos[0]

74   def _filter_none(**kwargs):
75       # type: (Any) -> Dict[str, Any]
76       """Make dict excluding None values."""
77       return {k: v for k, v in kwargs.items() if v is not None}

80   class VcsInfo(object):
81       name = "vcs_info"
82   
83       def __init__(
84           self,
85           vcs,  # type: str
86           commit_id,  # type: str
87           requested_revision=None,  # type: Optional[str]
88           resolved_revision=None,  # type: Optional[str]
89           resolved_revision_type=None,  # type: Optional[str]
90       ):
91           self.vcs = vcs
92           self.requested_revision = requested_revision
93           self.commit_id = commit_id
94           self.resolved_revision = resolved_revision
95           self.resolved_revision_type = resolved_revision_type
96   
97       @classmethod
98       def _from_dict(cls, d):
99           # type: (Optional[Dict[str, Any]]) -> Optional[VcsInfo]
100          if d is None:
101              return None
102          return cls(
103              vcs=_get_required(d, str, "vcs"),
104              commit_id=_get_required(d, str, "commit_id"),
105              requested_revision=_get(d, str, "requested_revision"),
106              resolved_revision=_get(d, str, "resolved_revision"),
107              resolved_revision_type=_get(d, str, "resolved_revision_type"),
108          )
109  
110      def _to_dict(self):
111          # type: () -> Dict[str, Any]
112          return _filter_none(
113              vcs=self.vcs,
114              requested_revision=self.requested_revision,
115              commit_id=self.commit_id,
116              resolved_revision=self.resolved_revision,
117              resolved_revision_type=self.resolved_revision_type,
118          )

169  class DirectUrl(object):
170  
171      def __init__(
172          self,
173          url,  # type: str
174          info,  # type: InfoType
175          subdirectory=None,  # type: Optional[str]
176      ):
177          self.url = url
178          self.info = info
179          self.subdirectory = subdirectory
180  
181      def _remove_auth_from_netloc(self, netloc):
182          # type: (str) -> str
183          if "@" not in netloc:
184              return netloc
185          user_pass, netloc_no_user_pass = netloc.split("@", 1)
186          if (
187              isinstance(self.info, VcsInfo) and
188              self.info.vcs == "git" and
189              user_pass == "git"
190          ):
191              return netloc
192          if ENV_VAR_RE.match(user_pass):
193              return netloc
194          return netloc_no_user_pass
195  
196      @property
197      def redacted_url(self):
198          # type: () -> str
199          """url with user:password part removed unless it is formed with 
200          environment variables as specified in PEP 610, or it is ``git`` 
201          in the case of a git URL. 
202          """
203          purl = urllib_parse.urlsplit(self.url)
204          netloc = self._remove_auth_from_netloc(purl.netloc)
205          surl = urllib_parse.urlunsplit(
206              (purl.scheme, netloc, purl.path, purl.query, purl.fragment)
207          )
208          return surl
209  
210      def validate(self):
211          # type: () -> None
212          self.from_dict(self.to_dict())
213  
214      @classmethod
215      def from_dict(cls, d):
216          # type: (Dict[str, Any]) -> DirectUrl
217          return DirectUrl(
218              url=_get_required(d, str, "url"),
219              subdirectory=_get(d, str, "subdirectory"),
220              info=_exactly_one_of(
221                  [
222                      ArchiveInfo._from_dict(_get(d, dict, "archive_info")),
223                      DirInfo._from_dict(_get(d, dict, "dir_info")),
224                      VcsInfo._from_dict(_get(d, dict, "vcs_info")),
225                  ]
226              ),
227          )
228  
229      def to_dict(self):
230          # type: () -> Dict[str, Any]
231          res = _filter_none(
232              url=self.redacted_url,
233              subdirectory=self.subdirectory,
234          )
235          res[self.info.name] = self.info._to_dict()
236          return res
237  
238      @classmethod
239      def from_json(cls, s):
240          # type: (str) -> DirectUrl
241          return cls.from_dict(json.loads(s))
242  
243      def to_json(self):
244          # type: () -> str
245          return json.dumps(self.to_dict(), sort_keys=True)
