9    class GEOSContextHandle(GEOSBase):
10       """Represent a GEOS context handle."""
11       ptr_type = CONTEXT_PTR
12       destructor = lgeos.finishGEOS_r
13   
14       def __init__(self):
15           # Initializing the context handler for this thread with
16           # the notice and error handler.
17           self.ptr = lgeos.initGEOS_r(notice_h, error_h)
18   
19   
20   # Defining a thread-local object and creating an instance
21   # to hold a reference to GEOSContextHandle for this thread.
22   class GEOSContext(threading.local):
23       handle = None
24   
25   
26   thread_context = GEOSContext()

29   class GEOSFunc:
30       """ 
31       Serve as a wrapper for GEOS C Functions. Use thread-safe function 
32       variants when available. 
33       """
34       def __init__(self, func_name):
35           # GEOS thread-safe function signatures end with '_r' and take an
36           # additional context handle parameter.
37           self.cfunc = getattr(lgeos, func_name + '_r')
38           # Create a reference to thread_context so it's not garbage-collected
39           # before an attempt to call this object.
40           self.thread_context = thread_context
41   
42       def __call__(self, *args):
43           # Create a context handle if one doesn't exist for this thread.
44           self.thread_context.handle = self.thread_context.handle or GEOSContextHandle()
45           # Call the threaded GEOS routine with the pointer of the context handle
46           # as the first argument.
47           return self.cfunc(self.thread_context.handle.ptr, *args)
48   
49       def __str__(self):
50           return self.cfunc.__name__
51   
52       # argtypes property
53       def _get_argtypes(self):
54           return self.cfunc.argtypes
55   
56       def _set_argtypes(self, argtypes):
57           self.cfunc.argtypes = [CONTEXT_PTR, *argtypes]
58   
59       argtypes = property(_get_argtypes, _set_argtypes)
60   
61       # restype property
62       def _get_restype(self):
63           return self.cfunc.restype
64   
65       def _set_restype(self, restype):
66           self.cfunc.restype = restype
67   
68       restype = property(_get_restype, _set_restype)
