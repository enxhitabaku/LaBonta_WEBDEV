34       An AdminSite object encapsulates an instance of the Django admin application, ready 
35       to be hooked in to your URLconf. Models are registered with the AdminSite using the 
36       register() method, and the get_urls() method can then be used to access Django view 
37       functions that present a full admin interface for the collection of registered

65       def __init__(self, name='admin'):
66           self._registry = {}  # model_class class -> admin_class instance
67           self.name = name
68           self._actions = {'delete_selected': actions.delete_selected}
69           self._global_actions = self._actions.copy()
70           all_sites.add(self)

72       def check(self, app_configs):
73           """ 
74           Run the system checks on all ModelAdmins, except if they aren't 
75           customized at all. 
76           """
77           if app_configs is None:
78               app_configs = apps.get_app_configs()
79           app_configs = set(app_configs)  # Speed up lookups below
80   
81           errors = []
82           modeladmins = (o for o in self._registry.values() if o.__class__ is not ModelAdmin)
83           for modeladmin in modeladmins:
84               if modeladmin.model._meta.app_config in app_configs:
85                   errors.extend(modeladmin.check())
86           return errors

88       def register(self, model_or_iterable, admin_class=None, **options):
89           """ 
90           Register the given model(s) with the given admin class. 
91    
92           The model(s) should be Model classes, not instances. 
93    
94           If an admin class isn't given, use ModelAdmin (the default admin 
95           options). If keyword arguments are given -- e.g., list_display -- 
96           apply them as options to the admin class. 
97    
98           If a model is already registered, raise AlreadyRegistered. 
99    
100          If a model is abstract, raise ImproperlyConfigured. 
101          """
102          admin_class = admin_class or ModelAdmin
103          if isinstance(model_or_iterable, ModelBase):
104              model_or_iterable = [model_or_iterable]
105          for model in model_or_iterable:
106              if model._meta.abstract:
107                  raise ImproperlyConfigured(
108                      'The model %s is abstract, so it cannot be registered with admin.' % model.__name__
109                  )
110  
111              if model in self._registry:
112                  registered_admin = str(self._registry[model])
113                  msg = 'The model %s is already registered ' % model.__name__
114                  if registered_admin.endswith('.ModelAdmin'):
115                      # Most likely registered without a ModelAdmin subclass.
116                      msg += 'in app %r.' % re.sub(r'\.ModelAdmin$', '', registered_admin)
117                  else:
118                      msg += 'with %r.' % registered_admin
119                  raise AlreadyRegistered(msg)
120  
121              # Ignore the registration if the model has been
122              # swapped out.
123              if not model._meta.swapped:
124                  # If we got **options then dynamically construct a subclass of
125                  # admin_class with those **options.
126                  if options:
127                      # For reasons I don't quite understand, without a __module__
128                      # the created class appears to "live" in the wrong place,
129                      # which causes issues later on.
130                      options['__module__'] = __name__
131                      admin_class = type("%sAdmin" % model.__name__, (admin_class,), options)
132  
133                  # Instantiate the admin class to save in the registry
134                  self._registry[model] = admin_class(model, self)
135  
136      def unregister(self, model_or_iterable):
137          """ 
138          Unregister the given model(s). 
139   
140          If a model isn't already registered, raise NotRegistered. 
141          """
142          if isinstance(model_or_iterable, ModelBase):
143              model_or_iterable = [model_or_iterable]
144          for model in model_or_iterable:
145              if model not in self._registry:
146                  raise NotRegistered('The model %s is not registered' % model.__name__)
147              del self._registry[model]

149      def is_registered(self, model):
150          """ 
151          Check if a model class is registered with this `AdminSite`. 
152          """
153          return model in self._registry

155      def add_action(self, action, name=None):
156          """ 
157          Register an action to be available globally. 
158          """
159          name = name or action.__name__
160          self._actions[name] = action
161          self._global_actions[name] = action

163      def disable_action(self, name):
164          """ 
165          Disable a globally-registered action. Raise KeyError for invalid names. 
166          """
167          del self._actions[name]

170          """ 
171          Explicitly get a registered global action whether it's enabled or 
172          not. Raise KeyError for invalid names.

174          return self._global_actions[name]

191      def has_permission(self, request):
192          """ 
193          Return True if the given HttpRequest has permission to view 
194          *at least one* page in the admin site. 
195          """
196          return request.user.is_active and request.user.is_staff

198      def admin_view(self, view, cacheable=False):
199          """ 
200          Decorator to create an admin view attached to this ``AdminSite``. This 
201          wraps the view and provides permission checking by calling 
202          ``self.has_permission``. 
203   
204          You'll want to use this from within ``AdminSite.get_urls()``: 
205   
206              class MyAdminSite(AdminSite): 
207   
208                  def get_urls(self): 
209                      from django.urls import path 
210   
211                      urls = super().get_urls() 
212                      urls += [ 
213                          path('my_view/', self.admin_view(some_view)) 
214                      ] 
215                      return urls 
216   
217          By default, admin_views are marked non-cacheable using the 
218          ``never_cache`` decorator. If the view can be safely cached, set 
219          cacheable=True. 
220          """
221          def inner(request, *args, **kwargs):
222              if not self.has_permission(request):
223                  if request.path == reverse('admin:logout', current_app=self.name):
224                      index_path = reverse('admin:index', current_app=self.name)
225                      return HttpResponseRedirect(index_path)
226                  # Inner import to prevent django.contrib.admin (app) from
227                  # importing django.contrib.auth.models.User (unrelated model).
228                  from django.contrib.auth.views import redirect_to_login
229                  return redirect_to_login(
230                      request.get_full_path(),
231                      reverse('admin:login', current_app=self.name)
232                  )
233              return view(request, *args, **kwargs)
234          if not cacheable:
235              inner = never_cache(inner)
236          # We add csrf_protect here so this function can be used as a utility
237          # function for any view, without having to repeat 'csrf_protect'.
238          if not getattr(view, 'csrf_exempt', False):
239              inner = csrf_protect(inner)
240          return update_wrapper(inner, view)

242      def get_urls(self):
243          # Since this module gets imported in the application's root package,
244          # it cannot import models from other applications at the module level,
245          # and django.contrib.contenttypes.views imports ContentType.
246          from django.contrib.contenttypes import views as contenttype_views
247          from django.urls import include, path, re_path
248  
249          def wrap(view, cacheable=False):
250              def wrapper(*args, **kwargs):
251                  return self.admin_view(view, cacheable)(*args, **kwargs)
252              wrapper.admin_site = self
253              return update_wrapper(wrapper, view)
254  
255          # Admin-site-wide views.
256          urlpatterns = [
257              path('', wrap(self.index), name='index'),
258              path('login/', self.login, name='login'),
259              path('logout/', wrap(self.logout), name='logout'),
260              path('password_change/', wrap(self.password_change, cacheable=True), name='password_change'),
261              path(
262                  'password_change/done/',
263                  wrap(self.password_change_done, cacheable=True),
264                  name='password_change_done',
265              ),
266              path('jsi18n/', wrap(self.i18n_javascript, cacheable=True), name='jsi18n'),
267              path(
268                  'r/<int:content_type_id>/<path:object_id>/',
269                  wrap(contenttype_views.shortcut),
270                  name='view_on_site',
271              ),
272          ]
273  
274          # Add in each model's views, and create a list of valid URLS for the
275          # app_index
276          valid_app_labels = []
277          for model, model_admin in self._registry.items():
278              urlpatterns += [
279                  path('%s/%s/' % (model._meta.app_label, model._meta.model_name), include(model_admin.urls)),
280              ]
281              if model._meta.app_label not in valid_app_labels:
282                  valid_app_labels.append(model._meta.app_label)
283  
284          # If there were ModelAdmins registered, we should have a list of app
285          # labels for which we need to allow access to the app_index view,
286          if valid_app_labels:
287              regex = r'^(?P<app_label>' + '|'.join(valid_app_labels) + ')/$'
288              urlpatterns += [
289                  re_path(regex, wrap(self.app_index), name='app_list'),
290              ]
291          return urlpatterns

297      def each_context(self, request):
298          """ 
299          Return a dictionary of variables to put in the template context for 
300          *every* page in the admin site. 
301   
302          For sites running on a subpath, use the SCRIPT_NAME value if site_url 
303          hasn't been customized. 
304          """
305          script_name = request.META['SCRIPT_NAME']
306          site_url = script_name if self.site_url == '/' and script_name else self.site_url
307          return {
308              'site_title': self.site_title,
309              'site_header': self.site_header,
310              'site_url': site_url,
311              'has_permission': self.has_permission(request),
312              'available_apps': self.get_app_list(request),
313              'is_popup': False,
314              'is_nav_sidebar_enabled': self.enable_nav_sidebar,
315          }

317      def password_change(self, request, extra_context=None):
318          """ 
319          Handle the "change password" task -- both form display and validation. 
320          """
321          from django.contrib.admin.forms import AdminPasswordChangeForm
322          from django.contrib.auth.views import PasswordChangeView
323          url = reverse('admin:password_change_done', current_app=self.name)
324          defaults = {
325              'form_class': AdminPasswordChangeForm,
326              'success_url': url,
327              'extra_context': {**self.each_context(request), **(extra_context or {})},
328          }
329          if self.password_change_template is not None:
330              defaults['template_name'] = self.password_change_template
331          request.current_app = self.name
332          return PasswordChangeView.as_view(**defaults)(request)

356      @never_cache
357      def logout(self, request, extra_context=None):
358          """ 
359          Log out the user for the given HttpRequest. 
360   
361          This should *not* assume the user is already logged in. 
362          """
363          from django.contrib.auth.views import LogoutView
364          defaults = {
365              'extra_context': {
366                  **self.each_context(request),
367                  # Since the user isn't logged out at this point, the value of
368                  # has_permission must be overridden.
369                  'has_permission': False,
370                  **(extra_context or {})
371              },
372          }
373          if self.logout_template is not None:
374              defaults['template_name'] = self.logout_template
375          request.current_app = self.name
376          return LogoutView.as_view(**defaults)(request)

378      @never_cache
379      def login(self, request, extra_context=None):
380          """ 
381          Display the login form for the given HttpRequest. 
382          """
383          if request.method == 'GET' and self.has_permission(request):
384              # Already logged-in, redirect to admin index
385              index_path = reverse('admin:index', current_app=self.name)
386              return HttpResponseRedirect(index_path)
387  
388          # Since this module gets imported in the application's root package,
389          # it cannot import models from other applications at the module level,
390          # and django.contrib.admin.forms eventually imports User.
391          from django.contrib.admin.forms import AdminAuthenticationForm
392          from django.contrib.auth.views import LoginView
393          context = {
394              **self.each_context(request),
395              'title': _('Log in'),
396              'app_path': request.get_full_path(),
397              'username': request.user.get_username(),
398          }
399          if (REDIRECT_FIELD_NAME not in request.GET and
400                  REDIRECT_FIELD_NAME not in request.POST):
401              context[REDIRECT_FIELD_NAME] = reverse('admin:index', current_app=self.name)
402          context.update(extra_context or {})
403  
404          defaults = {
405              'extra_context': context,
406              'authentication_form': self.login_form or AdminAuthenticationForm,
407              'template_name': self.login_template or 'admin/login.html',
408          }
409          request.current_app = self.name
410          return LoginView.as_view(**defaults)(request)

412      def _build_app_dict(self, request, label=None):
413          """ 
414          Build the app dictionary. The optional `label` parameter filters models 
415          of a specific app. 
416          """
417          app_dict = {}
418  
419          if label:
420              models = {
421                  m: m_a for m, m_a in self._registry.items()
422                  if m._meta.app_label == label
423              }
424          else:
425              models = self._registry
426  
427          for model, model_admin in models.items():
428              app_label = model._meta.app_label
429  
430              has_module_perms = model_admin.has_module_permission(request)
431              if not has_module_perms:
432                  continue
433  
434              perms = model_admin.get_model_perms(request)
435  
436              # Check whether user has any perm for this module.
437              # If so, add the module to the model_list.
438              if True not in perms.values():
439                  continue
440  
441              info = (app_label, model._meta.model_name)
442              model_dict = {
443                  'name': capfirst(model._meta.verbose_name_plural),
444                  'object_name': model._meta.object_name,
445                  'perms': perms,
446                  'admin_url': None,
447                  'add_url': None,
448              }
449              if perms.get('change') or perms.get('view'):
450                  model_dict['view_only'] = not perms.get('change')
451                  try:
452                      model_dict['admin_url'] = reverse('admin:%s_%s_changelist' % info, current_app=self.name)
453                  except NoReverseMatch:
454                      pass
455              if perms.get('add'):
456                  try:
457                      model_dict['add_url'] = reverse('admin:%s_%s_add' % info, current_app=self.name)
458                  except NoReverseMatch:
459                      pass
460  
461              if app_label in app_dict:
462                  app_dict[app_label]['models'].append(model_dict)
463              else:
464                  app_dict[app_label] = {
465                      'name': apps.get_app_config(app_label).verbose_name,
466                      'app_label': app_label,
467                      'app_url': reverse(
468                          'admin:app_list',
469                          kwargs={'app_label': app_label},
470                          current_app=self.name,
471                      ),
472                      'has_module_perms': has_module_perms,
473                      'models': [model_dict],
474                  }
475  
476          if label:
477              return app_dict.get(label)
478          return app_dict

480      def get_app_list(self, request):
481          """ 
482          Return a sorted list of all the installed apps that have been 
483          registered in this site. 
484          """
485          app_dict = self._build_app_dict(request)
486  
487          # Sort the apps alphabetically.
488          app_list = sorted(app_dict.values(), key=lambda x: x['name'].lower())
489  
490          # Sort the models alphabetically within each app.
491          for app in app_list:
492              app['models'].sort(key=lambda x: x['name'])
493  
494          return app_list

496      @never_cache
497      def index(self, request, extra_context=None):
498          """ 
499          Display the main admin index page, which lists all of the installed 
500          apps that have been registered in this site. 
501          """
502          app_list = self.get_app_list(request)
503  
504          context = {
505              **self.each_context(request),
506              'title': self.index_title,
507              'app_list': app_list,
508              **(extra_context or {}),
509          }
510  
511          request.current_app = self.name
512  
513          return TemplateResponse(request, self.index_template or 'admin/index.html', context)
514  
515      def app_index(self, request, app_label, extra_context=None):
516          app_dict = self._build_app_dict(request, app_label)
517          if not app_dict:
518              raise Http404('The requested admin page does not exist.')
519          # Sort the models alphabetically within each app.
520          app_dict['models'].sort(key=lambda x: x['name'])
521          app_name = apps.get_app_config(app_label).verbose_name
522          context = {
523              **self.each_context(request),
524              'title': _('%(app)s administration') % {'app': app_name},
525              'app_list': [app_dict],
526              'app_label': app_label,
527              **(extra_context or {}),
528          }
529  
530          request.current_app = self.name
531  
532          return TemplateResponse(request, self.app_index_template or [
533              'admin/%s/app_index.html' % app_label,
534              'admin/app_index.html'
535          ], context)
