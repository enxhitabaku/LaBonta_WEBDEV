53       def render(self, context):
54           csrf_token = context.get('csrf_token')
55           if csrf_token:
56               if csrf_token == 'NOTPROVIDED':
57                   return format_html("")
58               else:
59                   return format_html('<input type="hidden" name="csrfmiddlewaretoken" value="{}">', csrf_token)

61               # It's very probable that the token is missing because of
62               # misconfiguration, so we raise a warning
63               if settings.DEBUG:
64                   warnings.warn(
65                       "A {% csrf_token %} was used in a template, but the context "
66                       "did not provide the value.  This is usually caused by not "
67                       "using RequestContext."
68                   )
69               return ''

72   class CycleNode(Node):
73       def __init__(self, cyclevars, variable_name=None, silent=False):
74           self.cyclevars = cyclevars
75           self.variable_name = variable_name
76           self.silent = silent
77   
78       def render(self, context):
79           if self not in context.render_context:
80               # First time the node is rendered in template
81               context.render_context[self] = itertools_cycle(self.cyclevars)
82           cycle_iter = context.render_context[self]
83           value = next(cycle_iter).resolve(context)
84           if self.variable_name:
85               context.set_upward(self.variable_name, value)
86           if self.silent:
87               return ''
88           return render_value_in_context(value, context)
89   
90       def reset(self, context):
91           """ 
92           Reset the cycle iteration back to the beginning. 
93           """
94           context.render_context[self] = itertools_cycle(self.cyclevars)

135  class ForNode(Node):
136      child_nodelists = ('nodelist_loop', 'nodelist_empty')
137  
138      def __init__(self, loopvars, sequence, is_reversed, nodelist_loop, nodelist_empty=None):
139          self.loopvars, self.sequence = loopvars, sequence
140          self.is_reversed = is_reversed
141          self.nodelist_loop = nodelist_loop
142          if nodelist_empty is None:
143              self.nodelist_empty = NodeList()
144          else:
145              self.nodelist_empty = nodelist_empty
146  
147      def __repr__(self):
148          reversed_text = ' reversed' if self.is_reversed else ''
149          return '<%s: for %s in %s, tail_len: %d%s>' % (
150              self.__class__.__name__,
151              ', '.join(self.loopvars),
152              self.sequence,
153              len(self.nodelist_loop),
154              reversed_text,
155          )
156  
157      def render(self, context):
158          if 'forloop' in context:
159              parentloop = context['forloop']
160          else:
161              parentloop = {}
162          with context.push():
163              values = self.sequence.resolve(context, ignore_failures=True)
164              if values is None:
165                  values = []
166              if not hasattr(values, '__len__'):
167                  values = list(values)
168              len_values = len(values)
169              if len_values < 1:
170                  return self.nodelist_empty.render(context)
171              nodelist = []
172              if self.is_reversed:
173                  values = reversed(values)
174              num_loopvars = len(self.loopvars)
175              unpack = num_loopvars > 1
176              # Create a forloop value in the context.  We'll update counters on each
177              # iteration just below.
178              loop_dict = context['forloop'] = {'parentloop': parentloop}
179              for i, item in enumerate(values):
180                  # Shortcuts for current loop iteration number.
181                  loop_dict['counter0'] = i
182                  loop_dict['counter'] = i + 1
183                  # Reverse counter iteration numbers.
184                  loop_dict['revcounter'] = len_values - i
185                  loop_dict['revcounter0'] = len_values - i - 1
186                  # Boolean values designating first and last times through loop.
187                  loop_dict['first'] = (i == 0)
188                  loop_dict['last'] = (i == len_values - 1)
189  
190                  pop_context = False
191                  if unpack:
192                      # If there are multiple loop variables, unpack the item into
193                      # them.
194                      try:
195                          len_item = len(item)
196                      except TypeError:  # not an iterable
197                          len_item = 1
198                      # Check loop variable count before unpacking
199                      if num_loopvars != len_item:
200                          raise ValueError(
201                              "Need {} values to unpack in for loop; got {}. "
202                              .format(num_loopvars, len_item),
203                          )
204                      unpacked_vars = dict(zip(self.loopvars, item))
205                      pop_context = True
206                      context.update(unpacked_vars)
207                  else:
208                      context[self.loopvars[0]] = item
209  
210                  for node in self.nodelist_loop:
211                      nodelist.append(node.render_annotated(context))
212  
213                  if pop_context:
214                      # Pop the loop variables pushed on to the context to avoid
215                      # the context ending up in an inconsistent state when other
216                      # tags (e.g., include and with) push data to context.
217                      context.pop()
218          return mark_safe(''.join(nodelist))

221  class IfChangedNode(Node):
222      child_nodelists = ('nodelist_true', 'nodelist_false')
223  
224      def __init__(self, nodelist_true, nodelist_false, *varlist):
225          self.nodelist_true, self.nodelist_false = nodelist_true, nodelist_false
226          self._varlist = varlist
227  
228      def render(self, context):
229          # Init state storage
230          state_frame = self._get_context_stack_frame(context)
231          state_frame.setdefault(self)
232  
233          nodelist_true_output = None
234          if self._varlist:
235              # Consider multiple parameters. This behaves like an OR evaluation
236              # of the multiple variables.
237              compare_to = [var.resolve(context, ignore_failures=True) for var in self._varlist]
238          else:
239              # The "{% ifchanged %}" syntax (without any variables) compares
240              # the rendered output.
241              compare_to = nodelist_true_output = self.nodelist_true.render(context)
242  
243          if compare_to != state_frame[self]:
244              state_frame[self] = compare_to
245              # render true block if not already rendered
246              return nodelist_true_output or self.nodelist_true.render(context)
247          elif self.nodelist_false:
248              return self.nodelist_false.render(context)
249          return ''
250  
251      def _get_context_stack_frame(self, context):
252          # The Context object behaves like a stack where each template tag can create a new scope.
253          # Find the place where to store the state to detect changes.
254          if 'forloop' in context:
255              # Ifchanged is bound to the local for loop.
256              # When there is a loop-in-loop, the state is bound to the inner loop,
257              # so it resets when the outer loop continues.
258              return context['forloop']
259          else:
260              # Using ifchanged outside loops. Effectively this is a no-op because the state is associated with 'self'.
261              return context.render_context

284  class IfNode(Node):
285  
286      def __init__(self, conditions_nodelists):
287          self.conditions_nodelists = conditions_nodelists
288  
289      def __repr__(self):
290          return '<%s>' % self.__class__.__name__
291  
292      def __iter__(self):
293          for _, nodelist in self.conditions_nodelists:
294              yield from nodelist
295  
296      @property
297      def nodelist(self):
298          return NodeList(self)
299  
300      def render(self, context):
301          for condition, nodelist in self.conditions_nodelists:
302  
303              if condition is not None:           # if / elif clause
304                  try:
305                      match = condition.eval(context)
306                  except VariableDoesNotExist:
307                      match = None
308              else:                               # else clause
309                  match = True
310  
311              if match:
312                  return nodelist.render(context)
313  
314          return ''

338  class RegroupNode(Node):
339      def __init__(self, target, expression, var_name):
340          self.target, self.expression = target, expression
341          self.var_name = var_name
342  
343      def resolve_expression(self, obj, context):
344          # This method is called for each object in self.target. See regroup()
345          # for the reason why we temporarily put the object in the context.
346          context[self.var_name] = obj
347          return self.expression.resolve(context, ignore_failures=True)
348  
349      def render(self, context):
350          obj_list = self.target.resolve(context, ignore_failures=True)
351          if obj_list is None:
352              # target variable wasn't found in context; fail silently.
353              context[self.var_name] = []
354              return ''
355          # List of dictionaries in the format:
356          # {'grouper': 'key', 'list': [list of contents]}.
357          context[self.var_name] = [
358              GroupedResult(grouper=key, list=list(val))
359              for key, val in
360              groupby(obj_list, lambda obj: self.resolve_expression(obj, context))
361          ]
362          return ''

423  class URLNode(Node):
424      def __init__(self, view_name, args, kwargs, asvar):
425          self.view_name = view_name
426          self.args = args
427          self.kwargs = kwargs
428          self.asvar = asvar
429  
430      def render(self, context):
431          from django.urls import NoReverseMatch, reverse
432          args = [arg.resolve(context) for arg in self.args]
433          kwargs = {k: v.resolve(context) for k, v in self.kwargs.items()}
434          view_name = self.view_name.resolve(context)
435          try:
436              current_app = context.request.current_app
437          except AttributeError:
438              try:
439                  current_app = context.request.resolver_match.namespace
440              except AttributeError:
441                  current_app = None
442          # Try to look up the URL. If it fails, raise NoReverseMatch unless the
443          # {% url ... as var %} construct is used, in which case return nothing.
444          url = ''
445          try:
446              url = reverse(view_name, args=args, kwargs=kwargs, current_app=current_app)
447          except NoReverseMatch:
448              if self.asvar is None:
449                  raise
450  
451          if self.asvar:
452              context[self.asvar] = url
453              return ''
454          else:
455              if context.autoescape:
456                  url = conditional_escape(url)
457              return url

468  class WidthRatioNode(Node):
469      def __init__(self, val_expr, max_expr, max_width, asvar=None):
470          self.val_expr = val_expr
471          self.max_expr = max_expr
472          self.max_width = max_width
473          self.asvar = asvar
474  
475      def render(self, context):
476          try:
477              value = self.val_expr.resolve(context)
478              max_value = self.max_expr.resolve(context)
479              max_width = int(self.max_width.resolve(context))
480          except VariableDoesNotExist:
481              return ''
482          except (ValueError, TypeError):
483              raise TemplateSyntaxError("widthratio final argument must be a number")
484          try:
485              value = float(value)
486              max_value = float(max_value)
487              ratio = (value / max_value) * max_width
488              result = str(round(ratio))
489          except ZeroDivisionError:
490              result = '0'
491          except (ValueError, TypeError, OverflowError):
492              result = ''
493  
494          if self.asvar:
495              context[self.asvar] = result
496              return ''
497          else:
498              return result

520  def autoescape(parser, token):
521      """ 
522      Force autoescape behavior for this block. 
523      """
524      # token.split_contents() isn't useful here because this tag doesn't accept variable as arguments
525      args = token.contents.split()
526      if len(args) != 2:
527          raise TemplateSyntaxError("'autoescape' tag requires exactly one argument.")
528      arg = args[1]
529      if arg not in ('on', 'off'):
530          raise TemplateSyntaxError("'autoescape' argument should be 'on' or 'off'")

532      parser.delete_first_token()
533      return AutoEscapeControlNode((arg == 'on'), nodelist)

545  @register.tag
546  def cycle(parser, token):
547      """ 
548      Cycle among the given strings each time this tag is encountered. 
549   
550      Within a loop, cycles among the given strings each time through 
551      the loop:: 
552   
553          {% for o in some_list %} 
554              <tr class="{% cycle 'row1' 'row2' %}"> 
555                  ... 
556              </tr> 
557          {% endfor %} 
558   
559      Outside of a loop, give the values a unique name the first time you call 
560      it, then use that name each successive time through:: 
561   
562              <tr class="{% cycle 'row1' 'row2' 'row3' as rowcolors %}">...</tr> 
563              <tr class="{% cycle rowcolors %}">...</tr> 
564              <tr class="{% cycle rowcolors %}">...</tr> 
565   
566      You can use any number of values, separated by spaces. Commas can also 
567      be used to separate values; if a comma is used, the cycle values are 
568      interpreted as literal strings. 
569   
570      The optional flag "silent" can be used to prevent the cycle declaration 
571      from returning any value:: 
572   
573          {% for o in some_list %} 
574              {% cycle 'row1' 'row2' as rowcolors silent %} 
575              <tr class="{{ rowcolors }}">{% include "subtemplate.html " %}</tr> 
576          {% endfor %} 
577      """
578      # Note: This returns the exact same node on each {% cycle name %} call;
579      # that is, the node object returned from {% cycle a b c as name %} and the
580      # one returned from {% cycle name %} are the exact same object. This
581      # shouldn't cause problems (heh), but if it does, now you know.
582      #
583      # Ugly hack warning: This stuffs the named template dict into parser so
584      # that names are only unique within each template (as opposed to using
585      # a global variable, which would make cycle names have to be unique across
586      # *all* templates.
587      #
588      # It keeps the last node in the parser to be able to reset it with
589      # {% resetcycle %}.
590  
591      args = token.split_contents()
592  
593      if len(args) < 2:
594          raise TemplateSyntaxError("'cycle' tag requires at least two arguments")
595  
596      if len(args) == 2:
597          # {% cycle foo %} case.
598          name = args[1]
599          if not hasattr(parser, '_named_cycle_nodes'):
600              raise TemplateSyntaxError("No named cycles in template. '%s' is not defined" % name)
601          if name not in parser._named_cycle_nodes:
602              raise TemplateSyntaxError("Named cycle '%s' does not exist" % name)
603          return parser._named_cycle_nodes[name]
604  
605      as_form = False
606  
607      if len(args) > 4:
608          # {% cycle ... as foo [silent] %} case.
609          if args[-3] == "as":
610              if args[-1] != "silent":
611                  raise TemplateSyntaxError("Only 'silent' flag is allowed after cycle's name, not '%s'." % args[-1])
612              as_form = True
613              silent = True
614              args = args[:-1]
615          elif args[-2] == "as":
616              as_form = True
617              silent = False
618  
619      if as_form:
620          name = args[-1]
621          values = [parser.compile_filter(arg) for arg in args[1:-2]]
622          node = CycleNode(values, name, silent=silent)
623          if not hasattr(parser, '_named_cycle_nodes'):
624              parser._named_cycle_nodes = {}
625          parser._named_cycle_nodes[name] = node
626      else:
627          values = [parser.compile_filter(arg) for arg in args[1:]]
628          node = CycleNode(values)
629      parser._last_cycle_node = node
630      return node

653  @register.tag('filter')
654  def do_filter(parser, token):
655      """ 
656      Filter the contents of the block through variable filters. 
657   
658      Filters can also be piped through each other, and they can have 
659      arguments -- just like in variable syntax. 
660   
661      Sample usage:: 
662   
663          {% filter force_escape|lower %} 
664              This text will be HTML-escaped, and will appear in lowercase. 
665          {% endfilter %} 
666   
667      Note that the ``escape`` and ``safe`` filters are not acceptable arguments. 
668      Instead, use the ``autoescape`` tag to manage autoescaping for blocks of 
669      template code. 
670      """
671      # token.split_contents() isn't useful here because this tag doesn't accept variable as arguments
672      _, rest = token.contents.split(None, 1)
673      filter_expr = parser.compile_filter("var|%s" % (rest))
674      for func, unused in filter_expr.filters:
675          filter_name = getattr(func, '_filter_name', None)
676          if filter_name in ('escape', 'safe'):
677              raise TemplateSyntaxError('"filter %s" is not permitted.  Use the "autoescape" tag instead.' % filter_name)
678      nodelist = parser.parse(('endfilter',))
679      parser.delete_first_token()
680      return FilterNode(filter_expr, nodelist)

683  @register.tag
684  def firstof(parser, token):
685      """ 
686      Output the first variable passed that is not False. 
687   
688      Output nothing if all the passed variables are False. 
689   
690      Sample usage:: 
691   
692          {% firstof var1 var2 var3 as myvar %} 
693   
694      This is equivalent to:: 
695   
696          {% if var1 %} 
697              {{ var1 }} 
698          {% elif var2 %} 
699              {{ var2 }} 
700          {% elif var3 %} 
701              {{ var3 }} 
702          {% endif %} 
703   
704      but much cleaner! 
705   
706      You can also use a literal string as a fallback value in case all 
707      passed variables are False:: 
708   
709          {% firstof var1 var2 var3 "fallback value" %} 
710   
711      If you want to disable auto-escaping of variables you can use:: 
712   
713          {% autoescape off %} 
714              {% firstof var1 var2 var3 "<strong>fallback value</strong>" %} 
715          {% autoescape %} 
716   
717      Or if only some variables should be escaped, you can use:: 
718   
719          {% firstof var1 var2|safe var3 "<strong>fallback value</strong>"|safe %} 
720      """
721      bits = token.split_contents()[1:]
722      asvar = None
723      if not bits:
724          raise TemplateSyntaxError("'firstof' statement requires at least one argument")
725  
726      if len(bits) >= 2 and bits[-2] == 'as':
727          asvar = bits[-1]
728          bits = bits[:-2]
729      return FirstOfNode([parser.compile_filter(bit) for bit in bits], asvar)

732  @register.tag('for')
733  def do_for(parser, token):
734      """ 
735      Loop over each item in an array. 
736   
737      For example, to display a list of athletes given ``athlete_list``:: 
738   
739          <ul> 
740          {% for athlete in athlete_list %} 
741              <li>{{ athlete.name }}</li> 
742          {% endfor %} 
743          </ul> 
744   
745      You can loop over a list in reverse by using 
746      ``{% for obj in list reversed %}``. 
747   
748      You can also unpack multiple values from a two-dimensional array:: 
749   
750          {% for key,value in dict.items %} 
751              {{ key }}: {{ value }} 
752          {% endfor %} 
753   
754      The ``for`` tag can take an optional ``{% empty %}`` clause that will 
755      be displayed if the given array is empty or could not be found:: 
756   
757          <ul> 
758            {% for athlete in athlete_list %} 
759              <li>{{ athlete.name }}</li> 
760            {% empty %} 
761              <li>Sorry, no athletes in this list.</li> 
762            {% endfor %} 
763          <ul> 
764   
765      The above is equivalent to -- but shorter, cleaner, and possibly faster 
766      than -- the following:: 
767   
768          <ul> 
769            {% if athlete_list %} 
770              {% for athlete in athlete_list %} 
771                <li>{{ athlete.name }}</li> 
772              {% endfor %} 
773            {% else %} 
774              <li>Sorry, no athletes in this list.</li> 
775            {% endif %} 
776          </ul> 
777   
778      The for loop sets a number of variables available within the loop: 
779   
780          ==========================  ================================================ 
781          Variable                    Description 
782          ==========================  ================================================ 
783          ``forloop.counter``         The current iteration of the loop (1-indexed) 
784          ``forloop.counter0``        The current iteration of the loop (0-indexed) 
785          ``forloop.revcounter``      The number of iterations from the end of the 
786                                      loop (1-indexed) 
787          ``forloop.revcounter0``     The number of iterations from the end of the 
788                                      loop (0-indexed) 
789          ``forloop.first``           True if this is the first time through the loop 
790          ``forloop.last``            True if this is the last time through the loop 
791          ``forloop.parentloop``      For nested loops, this is the loop "above" the 
792                                      current one 
793          ==========================  ================================================ 
794      """
795      bits = token.split_contents()
796      if len(bits) < 4:
797          raise TemplateSyntaxError("'for' statements should have at least four"
798                                    " words: %s" % token.contents)
799  
800      is_reversed = bits[-1] == 'reversed'
801      in_index = -3 if is_reversed else -2
802      if bits[in_index] != 'in':
803          raise TemplateSyntaxError("'for' statements should use the format"
804                                    " 'for x in y': %s" % token.contents)
805  
806      invalid_chars = frozenset((' ', '"', "'", FILTER_SEPARATOR))
807      loopvars = re.split(r' *, *', ' '.join(bits[1:in_index]))
808      for var in loopvars:
809          if not var or not invalid_chars.isdisjoint(var):
810              raise TemplateSyntaxError("'for' tag received an invalid argument:"
811                                        " %s" % token.contents)
812  
813      sequence = parser.compile_filter(bits[in_index + 1])
814      nodelist_loop = parser.parse(('empty', 'endfor',))
815      token = parser.next_token()
816      if token.contents == 'empty':
817          nodelist_empty = parser.parse(('endfor',))
818          parser.delete_first_token()
819      else:
820          nodelist_empty = None
821      return ForNode(loopvars, sequence, is_reversed, nodelist_loop, nodelist_empty)

903  @register.tag('if')
904  def do_if(parser, token):
905      """ 
906      Evaluate a variable, and if that variable is "true" (i.e., exists, is not 
907      empty, and is not a false boolean value), output the contents of the block: 
908   
909      :: 
910   
911          {% if athlete_list %} 
912              Number of athletes: {{ athlete_list|count }} 
913          {% elif athlete_in_locker_room_list %} 
914              Athletes should be out of the locker room soon! 
915          {% else %} 
916              No athletes. 
917          {% endif %} 
918   
919      In the above, if ``athlete_list`` is not empty, the number of athletes will 
920      be displayed by the ``{{ athlete_list|count }}`` variable. 
921   
922      The ``if`` tag may take one or several `` {% elif %}`` clauses, as well as 
923      an ``{% else %}`` clause that will be displayed if all previous conditions 
924      fail. These clauses are optional. 
925   
926      ``if`` tags may use ``or``, ``and`` or ``not`` to test a number of 
927      variables or to negate a given variable:: 
928   
929          {% if not athlete_list %} 
930              There are no athletes. 
931          {% endif %} 
932   
933          {% if athlete_list or coach_list %} 
934              There are some athletes or some coaches. 
935          {% endif %} 
936   
937          {% if athlete_list and coach_list %} 
938              Both athletes and coaches are available. 
939          {% endif %} 
940   
941          {% if not athlete_list or coach_list %} 
942              There are no athletes, or there are some coaches. 
943          {% endif %} 
944   
945          {% if athlete_list and not coach_list %} 
946              There are some athletes and absolutely no coaches. 
947          {% endif %} 
948   
949      Comparison operators are also available, and the use of filters is also 
950      allowed, for example:: 
951   
952          {% if articles|length >= 5 %}...{% endif %} 
953   
954      Arguments and operators _must_ have a space between them, so 
955      ``{% if 1>2 %}`` is not a valid if tag. 
956   
957      All supported operators are: ``or``, ``and``, ``in``, ``not in`` 
958      ``==``, ``!=``, ``>``, ``>=``, ``<`` and ``<=``. 
959   
960      Operator precedence follows Python. 
961      """
962      # {% if ... %}
963      bits = token.split_contents()[1:]
964      condition = TemplateIfParser(parser, bits).parse()
965      nodelist = parser.parse(('elif', 'else', 'endif'))
966      conditions_nodelists = [(condition, nodelist)]
967      token = parser.next_token()
968  
969      # {% elif ... %} (repeatable)
970      while token.contents.startswith('elif'):
971          bits = token.split_contents()[1:]
972          condition = TemplateIfParser(parser, bits).parse()
973          nodelist = parser.parse(('elif', 'else', 'endif'))
974          conditions_nodelists.append((condition, nodelist))
975          token = parser.next_token()
976  
977      # {% else %} (optional)
978      if token.contents == 'else':
979          nodelist = parser.parse(('endif',))
980          conditions_nodelists.append((None, nodelist))
981          token = parser.next_token()
982  
983      # {% endif %}
984      if token.contents != 'endif':
985          raise TemplateSyntaxError('Malformed template tag at line {}: "{}"'.format(token.lineno, token.contents))
986  
987      return IfNode(conditions_nodelists)

993      Check if a value has changed from the last iteration of a loop. 
994   
995      The ``{% ifchanged %}`` block tag is used within a loop. It has two 
996      possible uses. 
997   
998      1. Check its own rendered contents against its previous state and only

1043 def load_from_library(library, label, names):
1044     """

1060                 ),
1061             )
1062     return subset

1099     Create random Latin text useful for providing test data in templates.

1101     Usage format:: 
1102  
1103         {% lorem [count] [method] [random] %} 
1104  
1105     ``count`` is a number (or variable) containing the number of paragraphs or 
1106     words to generate (default is 1).

1136         count = '1'

1139         raise TemplateSyntaxError("Incorrect format for %r tag" % tagname)

1166 @register.tag
1167 def regroup(parser, token):
1168     """ 
1169     Regroup a list of alike objects by a common attribute. 
1170  
1171     This complex tag is best illustrated by use of an example: say that 
1172     ``musicians`` is a list of ``Musician`` objects that have ``name`` and 
1173     ``instrument`` attributes, and you'd like to display a list that 
1174     looks like: 
1175  
1176         * Guitar: 
1177             * Django Reinhardt 
1178             * Emily Remler 
1179         * Piano: 
1180             * Lovie Austin 
1181             * Bud Powell 
1182         * Trumpet: 
1183             * Duke Ellington 
1184  
1185     The following snippet of template code would accomplish this dubious task:: 
1186  
1187         {% regroup musicians by instrument as grouped %} 
1188         <ul> 
1189         {% for group in grouped %} 
1190             <li>{{ group.grouper }} 
1191             <ul> 
1192                 {% for musician in group.list %} 
1193                 <li>{{ musician.name }}</li> 
1194                 {% endfor %} 
1195             </ul> 
1196         {% endfor %} 
1197         </ul> 
1198  
1199     As you can see, ``{% regroup %}`` populates a variable with a list of 
1200     objects with ``grouper`` and ``list`` attributes. ``grouper`` contains the 
1201     item that was grouped by; ``list`` contains the list of objects that share 
1202     that ``grouper``. In this case, ``grouper`` would be ``Guitar``, ``Piano`` 
1203     and ``Trumpet``, and ``list`` is the list of musicians who play this 
1204     instrument. 
1205  
1206     Note that ``{% regroup %}`` does not work when the list to be grouped is not 
1207     sorted by the key you are grouping by! This means that if your list of 
1208     musicians was not sorted by instrument, you'd need to make sure it is sorted 
1209     before using it, i.e.:: 
1210  
1211         {% regroup musicians|dictsort:"instrument" by instrument as grouped %} 
1212     """
1213     bits = token.split_contents()
1214     if len(bits) != 6:
1215         raise TemplateSyntaxError("'regroup' tag takes five arguments")
1216     target = parser.compile_filter(bits[1])
1217     if bits[2] != 'by':
1218         raise TemplateSyntaxError("second argument to 'regroup' tag must be 'by'")
1219     if bits[4] != 'as':
1220         raise TemplateSyntaxError("next-to-last argument to 'regroup' tag must"
1221                                   " be 'as'")
1222     var_name = bits[5]
1223     # RegroupNode will take each item in 'target', put it in the context under
1224     # 'var_name', evaluate 'var_name'.'expression' in the current context, and
1225     # group by the resulting value. After all items are processed, it will
1226     # save the final result in the context under 'var_name', thus clearing the
1227     # temporary values. This hack is necessary because the template engine
1228     # doesn't provide a context-aware equivalent of Python's getattr.
1229     expression = parser.compile_filter(var_name +
1230                                        VARIABLE_ATTRIBUTE_SEPARATOR +
1231                                        bits[3])
1232     return RegroupNode(target, expression, var_name)

1327 @register.tag
1328 def url(parser, token):
1329     r""" 
1330     Return an absolute URL matching the given view with its parameters. 
1331  
1332     This is a way to define links that aren't tied to a particular URL 
1333     configuration:: 
1334  
1335         {% url "url_name" arg1 arg2 %} 
1336  
1337         or 
1338  
1339         {% url "url_name" name1=value1 name2=value2 %} 
1340  
1341     The first argument is a URL pattern name. Other arguments are 
1342     space-separated values that will be filled in place of positional and 
1343     keyword arguments in the URL. Don't mix positional and keyword arguments. 
1344     All arguments for the URL must be present. 
1345  
1346     For example, if you have a view ``app_name.views.client_details`` taking 
1347     the client's id and the corresponding line in a URLconf looks like this:: 
1348  
1349         path('client/<int:id>/', views.client_details, name='client-detail-view') 
1350  
1351     and this app's URLconf is included into the project's URLconf under some 
1352     path:: 
1353  
1354         path('clients/', include('app_name.urls')) 
1355  
1356     then in a template you can create a link for a certain client like this:: 
1357  
1358         {% url "client-detail-view" client.id %} 
1359  
1360     The URL will look like ``/clients/client/123/``. 
1361  
1362     The first argument may also be the name of a template variable that will be 
1363     evaluated to obtain the view name or the URL name, e.g.:: 
1364  
1365         {% with url_name="client-detail-view" %} 
1366         {% url url_name client.id %} 
1367         {% endwith %} 
1368     """
1369     bits = token.split_contents()
1370     if len(bits) < 2:
1371         raise TemplateSyntaxError("'%s' takes at least one argument, a URL pattern name." % bits[0])
1372     viewname = parser.compile_filter(bits[1])
1373     args = []
1374     kwargs = {}
1375     asvar = None
1376     bits = bits[2:]
1377     if len(bits) >= 2 and bits[-2] == 'as':
1378         asvar = bits[-1]
1379         bits = bits[:-2]

1417 @register.tag
1418 def widthratio(parser, token):
1419     """ 
1420     For creating bar charts and such. Calculate the ratio of a given value to a 
1421     maximum value, and then apply that ratio to a constant. 
1422  
1423     For example:: 
1424  
1425         <img src="bar.png" alt="Bar" 
1426              height="10" width="{% widthratio this_value max_value max_width %}"> 
1427  
1428     If ``this_value`` is 175, ``max_value`` is 200, and ``max_width`` is 100, 
1429     the image in the above example will be 88 pixels wide 
1430     (because 175/200 = .875; .875 * 100 = 87.5 which is rounded up to 88). 
1431  
1432     In some cases you might want to capture the result of widthratio in a 
1433     variable. It can be useful for instance in a blocktranslate like this:: 
1434  
1435         {% widthratio this_value max_value max_width as width %} 
1436         {% blocktranslate %}The width is: {{ width }}{% endblocktranslate %} 
1437     """
1438     bits = token.split_contents()
1439     if len(bits) == 4:
1440         tag, this_value_expr, max_value_expr, max_width = bits
1441         asvar = None
1442     elif len(bits) == 6:
1443         tag, this_value_expr, max_value_expr, max_width, as_, asvar = bits
1444         if as_ != 'as':
1445             raise TemplateSyntaxError("Invalid syntax in widthratio tag. Expecting 'as' keyword")
1446     else:
1447         raise TemplateSyntaxError("widthratio takes at least three arguments")
1448 
1449     return WidthRatioNode(parser.compile_filter(this_value_expr),
1450                           parser.compile_filter(max_value_expr),
1451                           parser.compile_filter(max_width),
1452                           asvar=asvar)

1469         {% with foo=1 bar=2 %}

1474     still accepted.

1484                                   (bits[0], remaining_bits[0]))
