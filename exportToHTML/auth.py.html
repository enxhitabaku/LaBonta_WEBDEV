44   def get_keyring_auth(url, username):
45       # type: (str, str) -> Optional[AuthInfo]
46       """Return the tuple auth for a given url from keyring."""
47       global keyring
48       if not url or not keyring:
49           return None
50   
51       try:
52           try:
53               get_credential = keyring.get_credential
54           except AttributeError:
55               pass
56           else:
57               logger.debug("Getting credentials from keyring for %s", url)
58               cred = get_credential(url, username)
59               if cred is not None:
60                   return cred.username, cred.password
61               return None
62   
63           if username:
64               logger.debug("Getting password from keyring for %s", url)
65               password = keyring.get_password(url, username)
66               if password:
67                   return username, password
68   
69       except Exception as exc:
70           logger.warning(
71               "Keyring is skipped due to an exception: %s", str(exc),
72           )
73           keyring = None
74       return None

77   class MultiDomainBasicAuth(AuthBase):
78   
79       def __init__(self, prompting=True, index_urls=None):
80           # type: (bool, Optional[List[str]]) -> None
81           self.prompting = prompting
82           self.index_urls = index_urls
83           self.passwords = {}  # type: Dict[str, AuthInfo]
84           # When the user is prompted to enter credentials and keyring is
85           # available, we will offer to save them. If the user accepts,
86           # this value is set to the credentials they entered. After the
87           # request authenticates, the caller should call
88           # ``save_credentials`` to save these.
89           self._credentials_to_save = None  # type: Optional[Credentials]
90   
91       def _get_index_url(self, url):
92           # type: (str) -> Optional[str]
93           """Return the original index URL matching the requested URL. 
94    
95           Cached or dynamically generated credentials may work against 
96           the original index URL rather than just the netloc. 
97    
98           The provided url should have had its username and password 
99           removed already. If the original index url had credentials then 
100          they will be included in the return value. 
101   
102          Returns None if no matching index was found, or if --no-index 
103          was specified by the user. 
104          """
105          if not url or not self.index_urls:
106              return None
107  
108          for u in self.index_urls:
109              prefix = remove_auth_from_url(u).rstrip("/") + "/"
110              if url.startswith(prefix):
111                  return u
112          return None

114      def _get_new_credentials(self, original_url, allow_netrc=True,
115                               allow_keyring=True):
116          # type: (str, bool, bool) -> AuthInfo
117          """Find and return credentials for the specified URL."""
118          # Split the credentials and netloc from the url.
119          url, netloc, url_user_password = split_auth_netloc_from_url(
120              original_url,
121          )
122  
123          # Start with the credentials embedded in the url
124          username, password = url_user_password
125          if username is not None and password is not None:
126              logger.debug("Found credentials in url for %s", netloc)
127              return url_user_password
128  
129          # Find a matching index url for this request
130          index_url = self._get_index_url(url)
131          if index_url:
132              # Split the credentials from the url.
133              index_info = split_auth_netloc_from_url(index_url)
134              if index_info:
135                  index_url, _, index_url_user_password = index_info
136                  logger.debug("Found index url %s", index_url)
137  
138          # If an index URL was found, try its embedded credentials
139          if index_url and index_url_user_password[0] is not None:
140              username, password = index_url_user_password
141              if username is not None and password is not None:
142                  logger.debug("Found credentials in index url for %s", netloc)
143                  return index_url_user_password
144  
145          # Get creds from netrc if we still don't have them
146          if allow_netrc:
147              netrc_auth = get_netrc_auth(original_url)
148              if netrc_auth:
149                  logger.debug("Found credentials in netrc for %s", netloc)
150                  return netrc_auth
151  
152          # If we don't have a password and keyring is available, use it.
153          if allow_keyring:
154              # The index url is more specific than the netloc, so try it first
155              kr_auth = (
156                  get_keyring_auth(index_url, username) or
157                  get_keyring_auth(netloc, username)
158              )
159              if kr_auth:
160                  logger.debug("Found credentials in keyring for %s", netloc)
161                  return kr_auth
162  
163          return username, password

165      def _get_url_and_credentials(self, original_url):
166          # type: (str) -> Tuple[str, Optional[str], Optional[str]]
167          """Return the credentials to use for the provided URL. 
168   
169          If allowed, netrc and keyring may be used to obtain the 
170          correct credentials. 
171   
172          Returns (url_without_credentials, username, password). Note 
173          that even if the original URL contains credentials, this 
174          function may return a different username and password. 
175          """
176          url, netloc, _ = split_auth_netloc_from_url(original_url)
177  
178          # Use any stored credentials that we have for this netloc
179          username, password = self.passwords.get(netloc, (None, None))
180  
181          if username is None and password is None:
182              # No stored credentials. Acquire new credentials without prompting
183              # the user. (e.g. from netrc, keyring, or the URL itself)
184              username, password = self._get_new_credentials(original_url)
185  
186          if username is not None or password is not None:
187              # Convert the username and password if they're None, so that
188              # this netloc will show up as "cached" in the conditional above.
189              # Further, HTTPBasicAuth doesn't accept None, so it makes sense to
190              # cache the value that is going to be used.
191              username = username or ""
192              password = password or ""
193  
194              # Store any acquired credentials.
195              self.passwords[netloc] = (username, password)
196  
197          assert (
198              # Credentials were found
199              (username is not None and password is not None) or
200              # Credentials were not found
201              (username is None and password is None)
202          ), "Could not load credentials from url: {}".format(original_url)
203  
204          return url, username, password
205  
206      def __call__(self, req):
207          # type: (Request) -> Request
208          # Get credentials for this request
209          url, username, password = self._get_url_and_credentials(req.url)
210  
211          # Set the url of the request to the url without any credentials
212          req.url = url
213  
214          if username is not None and password is not None:
215              # Send the basic auth with this request
216              req = HTTPBasicAuth(username, password)(req)
217  
218          # Attach a hook to handle 401 responses
219          req.register_hook("response", self.handle_401)
220  
221          return req

223      # Factored out to allow for easy patching in tests
224      def _prompt_for_password(self, netloc):
225          # type: (str) -> Tuple[Optional[str], Optional[str], bool]
226          username = ask_input("User for {}: ".format(netloc))
227          if not username:
228              return None, None, False
229          auth = get_keyring_auth(netloc, username)
230          if auth and auth[0] is not None and auth[1] is not None:
231              return auth[0], auth[1], False
232          password = ask_password("Password: ")
233          return username, password, True
234  
235      # Factored out to allow for easy patching in tests
236      def _should_save_password_to_keyring(self):
237          # type: () -> bool
238          if not keyring:
239              return False
240          return ask("Save credentials to keyring [y/N]: ", ["y", "n"]) == "y"
241  
242      def handle_401(self, resp, **kwargs):
243          # type: (Response, **Any) -> Response
244          # We only care about 401 responses, anything else we want to just
245          #   pass through the actual response
246          if resp.status_code != 401:
247              return resp
248  
249          # We are not able to prompt the user so simply return the response
250          if not self.prompting:
251              return resp
252  
253          parsed = urllib_parse.urlparse(resp.url)
254  
255          # Prompt the user for a new username and password
256          username, password, save = self._prompt_for_password(parsed.netloc)
257  
258          # Store the new username and password to use for future requests
259          self._credentials_to_save = None
260          if username is not None and password is not None:
261              self.passwords[parsed.netloc] = (username, password)
262  
263              # Prompt to save the password to keyring
264              if save and self._should_save_password_to_keyring():
265                  self._credentials_to_save = (parsed.netloc, username, password)
266  
267          # Consume content and release the original connection to allow our new
268          #   request to reuse the same one.
269          resp.content
270          resp.raw.release_conn()
271  
272          # Add our new username and password to the request
273          req = HTTPBasicAuth(username or "", password or "")(resp.request)
274          req.register_hook("response", self.warn_on_401)
275  
276          # On successful request, save the credentials that were used to
277          # keyring. (Note that if the user responded "no" above, this member
278          # is not set and nothing will be saved.)
279          if self._credentials_to_save:
280              req.register_hook("response", self.save_credentials)
281  
282          # Send our new request
283          new_resp = resp.connection.send(req, **kwargs)
284          new_resp.history.append(resp)
285  
286          return new_resp
287  
288      def warn_on_401(self, resp, **kwargs):
289          # type: (Response, **Any) -> None
290          """Response callback to warn about incorrect credentials."""
291          if resp.status_code == 401:
292              logger.warning(
293                  '401 Error, Credentials not correct for %s', resp.request.url,
294              )
295  
296      def save_credentials(self, resp, **kwargs):
297          # type: (Response, **Any) -> None
298          """Response callback to save credentials on success."""
299          assert keyring is not None, "should never reach here without keyring"
300          if not keyring:
301              return
302  
303          creds = self._credentials_to_save
304          self._credentials_to_save = None
305          if creds and resp.status_code < 400:
306              try:
307                  logger.info('Saving credentials to keyring')
308                  keyring.set_password(*creds)
309              except Exception:
310                  logger.exception('Failed to save credentials')
