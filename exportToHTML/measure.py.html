38   from decimal import Decimal
39   from functools import total_ordering

41   __all__ = ['A', 'Area', 'D', 'Distance']

43   NUMERIC_TYPES = (int, float, Decimal)

47   def pretty_name(obj):
48       return obj.__name__ if obj.__class__ == type else obj.__class__.__name__
49   
50   
51   @total_ordering
52   class MeasureBase:
53       STANDARD_UNIT = None
54       ALIAS = {}
55       UNITS = {}
56       LALIAS = {}
57   
58       def __init__(self, default_unit=None, **kwargs):
59           value, self._default_unit = self.default_units(kwargs)
60           setattr(self, self.STANDARD_UNIT, value)
61           if default_unit and isinstance(default_unit, str):
62               self._default_unit = default_unit
63   
64       def _get_standard(self):
65           return getattr(self, self.STANDARD_UNIT)
66   
67       def _set_standard(self, value):
68           setattr(self, self.STANDARD_UNIT, value)
69   
70       standard = property(_get_standard, _set_standard)
71   
72       def __getattr__(self, name):
73           if name in self.UNITS:
74               return self.standard / self.UNITS[name]
75           else:
76               raise AttributeError('Unknown unit type: %s' % name)
77   
78       def __repr__(self):
79           return '%s(%s=%s)' % (pretty_name(self), self._default_unit, getattr(self, self._default_unit))
80   
81       def __str__(self):
82           return '%s %s' % (getattr(self, self._default_unit), self._default_unit)
83   
84       # **** Comparison methods ****
85   
86       def __eq__(self, other):
87           if isinstance(other, self.__class__):
88               return self.standard == other.standard
89           else:
90               return NotImplemented
91   
92       def __lt__(self, other):
93           if isinstance(other, self.__class__):
94               return self.standard < other.standard
95           else:
96               return NotImplemented
97   
98       # **** Operators methods ****
99   
100      def __add__(self, other):
101          if isinstance(other, self.__class__):
102              return self.__class__(
103                  default_unit=self._default_unit,
104                  **{self.STANDARD_UNIT: (self.standard + other.standard)}
105              )
106          else:
107              raise TypeError('%(class)s must be added with %(class)s' % {"class": pretty_name(self)})
108  
109      def __iadd__(self, other):
110          if isinstance(other, self.__class__):
111              self.standard += other.standard
112              return self
113          else:
114              raise TypeError('%(class)s must be added with %(class)s' % {"class": pretty_name(self)})
115  
116      def __sub__(self, other):
117          if isinstance(other, self.__class__):
118              return self.__class__(
119                  default_unit=self._default_unit,
120                  **{self.STANDARD_UNIT: (self.standard - other.standard)}
121              )
122          else:
123              raise TypeError('%(class)s must be subtracted from %(class)s' % {"class": pretty_name(self)})
124  
125      def __isub__(self, other):
126          if isinstance(other, self.__class__):
127              self.standard -= other.standard
128              return self
129          else:
130              raise TypeError('%(class)s must be subtracted from %(class)s' % {"class": pretty_name(self)})
131  
132      def __mul__(self, other):
133          if isinstance(other, NUMERIC_TYPES):
134              return self.__class__(
135                  default_unit=self._default_unit,
136                  **{self.STANDARD_UNIT: (self.standard * other)}
137              )
138          else:
139              raise TypeError('%(class)s must be multiplied with number' % {"class": pretty_name(self)})
140  
141      def __imul__(self, other):
142          if isinstance(other, NUMERIC_TYPES):
143              self.standard *= float(other)
144              return self
145          else:
146              raise TypeError('%(class)s must be multiplied with number' % {"class": pretty_name(self)})
147  
148      def __rmul__(self, other):
149          return self * other
150  
151      def __truediv__(self, other):
152          if isinstance(other, self.__class__):
153              return self.standard / other.standard
154          if isinstance(other, NUMERIC_TYPES):
155              return self.__class__(
156                  default_unit=self._default_unit,
157                  **{self.STANDARD_UNIT: (self.standard / other)}
158              )
159          else:
160              raise TypeError('%(class)s must be divided with number or %(class)s' % {"class": pretty_name(self)})
161  
162      def __itruediv__(self, other):
163          if isinstance(other, NUMERIC_TYPES):
164              self.standard /= float(other)
165              return self
166          else:
167              raise TypeError('%(class)s must be divided with number' % {"class": pretty_name(self)})
168  
169      def __bool__(self):
170          return bool(self.standard)
171  
172      def default_units(self, kwargs):
173          """ 
174          Return the unit value and the default units specified 
175          from the given keyword arguments dictionary. 
176          """
177          val = 0.0
178          default_unit = self.STANDARD_UNIT
179          for unit, value in kwargs.items():
180              if not isinstance(value, float):
181                  value = float(value)
182              if unit in self.UNITS:
183                  val += self.UNITS[unit] * value
184                  default_unit = unit
185              elif unit in self.ALIAS:
186                  u = self.ALIAS[unit]
187                  val += self.UNITS[u] * value
188                  default_unit = u
189              else:
190                  lower = unit.lower()
191                  if lower in self.UNITS:
192                      val += self.UNITS[lower] * value
193                      default_unit = lower
194                  elif lower in self.LALIAS:
195                      u = self.LALIAS[lower]
196                      val += self.UNITS[u] * value
197                      default_unit = u
198                  else:
199                      raise AttributeError('Unknown unit type: %s' % unit)
200          return val, default_unit
201  
202      @classmethod
203      def unit_attname(cls, unit_str):
204          """ 
205          Retrieve the unit attribute name for the given unit string. 
206          For example, if the given unit string is 'metre', return 'm'. 
207          Raise an exception if an attribute cannot be found. 
208          """
209          lower = unit_str.lower()
210          if unit_str in cls.UNITS:
211              return unit_str
212          elif lower in cls.UNITS:
213              return lower
214          elif lower in cls.LALIAS:
215              return cls.LALIAS[lower]
216          else:
217              raise Exception('Could not find a unit keyword associated with "%s"' % unit_str)

299      def __mul__(self, other):
300          if isinstance(other, self.__class__):
301              return Area(
302                  default_unit=AREA_PREFIX + self._default_unit,
303                  **{AREA_PREFIX + self.STANDARD_UNIT: (self.standard * other.standard)}
304              )
305          elif isinstance(other, NUMERIC_TYPES):
306              return self.__class__(
307                  default_unit=self._default_unit,
308                  **{self.STANDARD_UNIT: (self.standard * other)}
309              )
310          else:
311              raise TypeError('%(distance)s must be multiplied with number or %(distance)s' % {
312                  "distance": pretty_name(self.__class__),
313              })

316  class Area(MeasureBase):
317      STANDARD_UNIT = AREA_PREFIX + Distance.STANDARD_UNIT
318      # Getting the square units values and the alias dictionary.
319      UNITS = {'%s%s' % (AREA_PREFIX, k): v ** 2 for k, v in Distance.UNITS.items()}
320      ALIAS = {k: '%s%s' % (AREA_PREFIX, v) for k, v in Distance.ALIAS.items()}
321      LALIAS = {k.lower(): v for k, v in ALIAS.items()}
322  
323      def __truediv__(self, other):
324          if isinstance(other, NUMERIC_TYPES):
325              return self.__class__(
326                  default_unit=self._default_unit,
327                  **{self.STANDARD_UNIT: (self.standard / other)}
328              )
329          else:
330              raise TypeError('%(class)s must be divided by a number' % {"class": pretty_name(self)})
