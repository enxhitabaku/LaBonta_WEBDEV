103          # Instantiating from the tuple state that was pickled.
104          wkb, srid = state
105          ptr = self._from_pickle_wkb(wkb)

116      @staticmethod
117      def from_ewkt(ewkt):
118          ewkt = force_bytes(ewkt)
119          srid = None
120          parts = ewkt.split(b';', 1)
121          if len(parts) == 2:
122              srid_part, wkt = parts
123              match = re.match(br'SRID=(?P<srid>\-?\d+)', srid_part)
124              if not match:
125                  raise ValueError('EWKT has invalid SRID part.')
126              srid = int(match['srid'])
127          else:
128              wkt = ewkt
129          if not wkt:
130              raise ValueError('Expected WKT but got an empty string.')
131          return GEOSGeometry(GEOSGeometry._from_wkt(wkt), srid=srid)

141      # Comparison operators
142      def __eq__(self, other):
143          """ 
144          Equivalence testing, a Geometry may be compared with another Geometry 
145          or an EWKT representation. 
146          """
147          if isinstance(other, str):
148              try:
149                  other = GEOSGeometry.from_ewkt(other)
150              except (ValueError, GEOSException):
151                  return False
152          return isinstance(other, GEOSGeometry) and self.srid == other.srid and self.equals_exact(other)

447      def transform(self, ct, clone=False):
448          """ 
449          Requires GDAL. Transform the geometry according to the given 
450          transformation object, which may be an integer SRID, and WKT or 
451          PROJ.4 string. By default, transform the geometry in-place and return 
452          nothing. However if the `clone` keyword is set, don't modify the 
453          geometry and return a transformed clone instead. 
454          """
455          srid = self.srid
456  
457          if ct == srid:
458              # short-circuit where source & dest SRIDs match
459              if clone:
460                  return self.clone()
461              else:
462                  return
463  
464          if isinstance(ct, gdal.CoordTransform):
465              # We don't care about SRID because CoordTransform presupposes
466              # source SRS.
467              srid = None
468          elif srid is None or srid < 0:
469              raise GEOSException("Calling transform() with no SRID set is not supported")
470  
471          # Creating an OGR Geometry, which is then transformed.
472          g = gdal.OGRGeometry(self._ogr_ptr(), srid)
473          g.transform(ct)
474          # Getting a new GEOS pointer
475          ptr = g._geos_ptr()
476          if clone:
477              # User wants a cloned transformed geometry returned.
478              return GEOSGeometry(ptr, srid=g.srid)
479          if ptr:
480              # Reassigning pointer, and performing post-initialization setup
481              # again due to the reassignment.
482              capi.destroy_geom(self.ptr)
483              self.ptr = ptr
484              self._post_init()
485              self.srid = g.srid
486          else:
487              raise GEOSException('Transformed WKB was invalid.')

563      def simplify(self, tolerance=0.0, preserve_topology=False):
564          """ 
565          Return the Geometry, simplified using the Douglas-Peucker algorithm 
566          to the specified tolerance (higher tolerance => less points).  If no 
567          tolerance provided, defaults to 0. 
568   
569          By default, don't preserve topology - e.g. polygons can be split, 
570          collapse to lines or disappear holes can be created or disappear, and 
571          lines can cross. By specifying preserve_topology=True, the result will 
572          have the same dimension and number of components as the input. This is 
573          significantly slower. 
574          """
575          if preserve_topology:
576              return self._topology(capi.geos_preservesimplify(self.ptr, tolerance))
577          else:
578              return self._topology(capi.geos_simplify(self.ptr, tolerance))

678  @deconstructible
679  class GEOSGeometry(GEOSGeometryBase, ListMixin):
680      "A class that, generally, encapsulates a GEOS geometry."
681  
682      def __init__(self, geo_input, srid=None):
683          """ 
684          The base constructor for GEOS geometry objects. It may take the 
685          following inputs: 
686   
687           * strings: 
688              - WKT 
689              - HEXEWKB (a PostGIS-specific canonical form) 
690              - GeoJSON (requires GDAL) 
691           * buffer: 
692              - WKB 
693   
694          The `srid` keyword specifies the Source Reference Identifier (SRID) 
695          number for this Geometry. If not provided, it defaults to None. 
696          """
697          input_srid = None
698          if isinstance(geo_input, bytes):
699              geo_input = force_str(geo_input)
700          if isinstance(geo_input, str):
701              wkt_m = wkt_regex.match(geo_input)
702              if wkt_m:
703                  # Handle WKT input.
704                  if wkt_m['srid']:
705                      input_srid = int(wkt_m['srid'])
706                  g = self._from_wkt(force_bytes(wkt_m['wkt']))
707              elif hex_regex.match(geo_input):
708                  # Handle HEXEWKB input.
709                  g = wkb_r().read(force_bytes(geo_input))
710              elif json_regex.match(geo_input):
711                  # Handle GeoJSON input.
712                  ogr = gdal.OGRGeometry.from_json(geo_input)
713                  g = ogr._geos_ptr()
714                  input_srid = ogr.srid
715              else:
716                  raise ValueError('String input unrecognized as WKT EWKT, and HEXEWKB.')
717          elif isinstance(geo_input, GEOM_PTR):
718              # When the input is a pointer to a geometry (GEOM_PTR).
719              g = geo_input
720          elif isinstance(geo_input, memoryview):
721              # When the input is a buffer (WKB).
722              g = wkb_r().read(geo_input)
723          elif isinstance(geo_input, GEOSGeometry):
724              g = capi.geom_clone(geo_input.ptr)
725          else:
726              raise TypeError('Improper geometry input type: %s' % type(geo_input))
727  
728          if not g:
729              raise GEOSException('Could not initialize GEOS Geometry with given input.')
730  
731          input_srid = input_srid or capi.geos_get_srid(g) or None
732          if input_srid and srid and input_srid != srid:
733              raise ValueError('Input geometry already has SRID: %d.' % input_srid)
734  
735          super().__init__(g, None)
736          # Set the SRID, if given.
737          srid = input_srid or srid
738          if srid and isinstance(srid, int):
739              self.srid = srid
