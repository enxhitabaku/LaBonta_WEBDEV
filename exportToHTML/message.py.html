55   def forbid_multi_line_headers(name, val, encoding):
56       """Forbid multi-line headers to prevent header injection."""
57       encoding = encoding or settings.DEFAULT_CHARSET
58       val = str(val)  # val may be lazy
59       if '\n' in val or '\r' in val:
60           raise BadHeaderError("Header values can't contain newlines (got %r for header %r)" % (val, name))
61       try:
62           val.encode('ascii')
63       except UnicodeEncodeError:
64           if name.lower() in ADDRESS_HEADERS:
65               val = ', '.join(sanitize_address(addr, encoding) for addr in getaddresses((val,)))
66           else:
67               val = Header(val, encoding).encode()
68       else:
69           if name.lower() == 'subject':
70               val = Header(val).encode()
71       return name, val

74   def sanitize_address(addr, encoding):
75       """ 
76       Format a pair of (name, address) or an email address string. 
77       """
78       address = None
79       if not isinstance(addr, tuple):
80           addr = force_str(addr)
81           try:
82               token, rest = parser.get_mailbox(addr)
83           except (HeaderParseError, ValueError, IndexError):
84               raise ValueError('Invalid address "%s"' % addr)
85           else:
86               if rest:
87                   # The entire email address must be parsed.
88                   raise ValueError(
89                       'Invalid address; only %s could be parsed from "%s"'
90                       % (token, addr)
91                   )
92               nm = token.display_name or ''
93               localpart = token.local_part
94               domain = token.domain or ''
95       else:
96           nm, address = addr
97           localpart, domain = address.rsplit('@', 1)
98   
99       address_parts = nm + localpart + domain
100      if '\n' in address_parts or '\r' in address_parts:
101          raise ValueError('Invalid address; address parts cannot contain newlines.')
102  
103      # Avoid UTF-8 encode, if it's possible.
104      try:
105          nm.encode('ascii')
106          nm = Header(nm).encode()
107      except UnicodeEncodeError:
108          nm = Header(nm, encoding).encode()
109      try:
110          localpart.encode('ascii')
111      except UnicodeEncodeError:
112          localpart = Header(localpart, encoding).encode()
113      domain = punycode(domain)
114  
115      parsed_address = Address(username=localpart, domain=domain)
116      return formataddr((nm, parsed_address.addr_spec))

119  class MIMEMixin:
120      def as_string(self, unixfrom=False, linesep='\n'):
121          """Return the entire formatted message as a string. 
122          Optional `unixfrom' when True, means include the Unix From_ envelope 
123          header. 
124   
125          This overrides the default as_string() implementation to not mangle 
126          lines that begin with 'From '. See bug #13433 for details. 
127          """
128          fp = StringIO()
129          g = generator.Generator(fp, mangle_from_=False)
130          g.flatten(self, unixfrom=unixfrom, linesep=linesep)
131          return fp.getvalue()
132  
133      def as_bytes(self, unixfrom=False, linesep='\n'):
134          """Return the entire formatted message as bytes. 
135          Optional `unixfrom' when True, means include the Unix From_ envelope 
136          header. 
137   
138          This overrides the default as_bytes() implementation to not mangle 
139          lines that begin with 'From '. See bug #13433 for details. 
140          """
141          fp = BytesIO()
142          g = generator.BytesGenerator(fp, mangle_from_=False)
143          g.flatten(self, unixfrom=unixfrom, linesep=linesep)
144          return fp.getvalue()

150          # message/rfc822 attachments must be ASCII
151          name, val = forbid_multi_line_headers(name, val, 'ascii')
152          MIMEMessage.__setitem__(self, name, val)

155  class SafeMIMEText(MIMEMixin, MIMEText):
156  
157      def __init__(self, _text, _subtype='plain', _charset=None):
158          self.encoding = _charset
159          MIMEText.__init__(self, _text, _subtype=_subtype, _charset=_charset)
160  
161      def __setitem__(self, name, val):
162          name, val = forbid_multi_line_headers(name, val, self.encoding)
163          MIMEText.__setitem__(self, name, val)
164  
165      def set_payload(self, payload, charset=None):
166          if charset == 'utf-8' and not isinstance(charset, Charset.Charset):
167              has_long_lines = any(
168                  len(line.encode()) > RFC5322_EMAIL_LINE_LENGTH_LIMIT
169                  for line in payload.splitlines()
170              )
171              # Quoted-Printable encoding has the side effect of shortening long
172              # lines, if any (#22561).
173              charset = utf8_charset_qp if has_long_lines else utf8_charset
174          MIMEText.set_payload(self, payload, charset=charset)

188  class EmailMessage:
189      """A container for email information."""
190      content_subtype = 'plain'
191      mixed_subtype = 'mixed'
192      encoding = None     # None => use settings default
193  
194      def __init__(self, subject='', body='', from_email=None, to=None, bcc=None,
195                   connection=None, attachments=None, headers=None, cc=None,
196                   reply_to=None):
197          """ 
198          Initialize a single email message (which can be sent to multiple 
199          recipients). 
200          """
201          if to:
202              if isinstance(to, str):
203                  raise TypeError('"to" argument must be a list or tuple')
204              self.to = list(to)
205          else:
206              self.to = []
207          if cc:
208              if isinstance(cc, str):
209                  raise TypeError('"cc" argument must be a list or tuple')
210              self.cc = list(cc)
211          else:
212              self.cc = []
213          if bcc:
214              if isinstance(bcc, str):
215                  raise TypeError('"bcc" argument must be a list or tuple')
216              self.bcc = list(bcc)
217          else:
218              self.bcc = []
219          if reply_to:
220              if isinstance(reply_to, str):
221                  raise TypeError('"reply_to" argument must be a list or tuple')
222              self.reply_to = list(reply_to)
223          else:
224              self.reply_to = []
225          self.from_email = from_email or settings.DEFAULT_FROM_EMAIL
226          self.subject = subject
227          self.body = body or ''
228          self.attachments = []
229          if attachments:
230              for attachment in attachments:
231                  if isinstance(attachment, MIMEBase):
232                      self.attach(attachment)
233                  else:
234                      self.attach(*attachment)
235          self.extra_headers = headers or {}
236          self.connection = connection
237  
238      def get_connection(self, fail_silently=False):
239          from django.core.mail import get_connection
240          if not self.connection:
241              self.connection = get_connection(fail_silently=fail_silently)
242          return self.connection
243  
244      def message(self):
245          encoding = self.encoding or settings.DEFAULT_CHARSET
246          msg = SafeMIMEText(self.body, self.content_subtype, encoding)
247          msg = self._create_message(msg)
248          msg['Subject'] = self.subject
249          msg['From'] = self.extra_headers.get('From', self.from_email)
250          self._set_list_header_if_not_empty(msg, 'To', self.to)
251          self._set_list_header_if_not_empty(msg, 'Cc', self.cc)
252          self._set_list_header_if_not_empty(msg, 'Reply-To', self.reply_to)
253  
254          # Email header names are case-insensitive (RFC 2045), so we have to
255          # accommodate that when doing comparisons.
256          header_names = [key.lower() for key in self.extra_headers]
257          if 'date' not in header_names:
258              # formatdate() uses stdlib methods to format the date, which use
259              # the stdlib/OS concept of a timezone, however, Django sets the
260              # TZ environment variable based on the TIME_ZONE setting which
261              # will get picked up by formatdate().
262              msg['Date'] = formatdate(localtime=settings.EMAIL_USE_LOCALTIME)
263          if 'message-id' not in header_names:
264              # Use cached DNS_NAME for performance
265              msg['Message-ID'] = make_msgid(domain=DNS_NAME)
266          for name, value in self.extra_headers.items():
267              if name.lower() != 'from':  # From is already handled
268                  msg[name] = value
269          return msg
270  
271      def recipients(self):
272          """ 
273          Return a list of all recipients of the email (includes direct 
274          addressees as well as Cc and Bcc entries). 
275          """
276          return [email for email in (self.to + self.cc + self.bcc) if email]
277  
278      def send(self, fail_silently=False):
279          """Send the email message."""
280          if not self.recipients():
281              # Don't bother creating the network connection if there's nobody to
282              # send to.
283              return 0
284          return self.get_connection(fail_silently).send_messages([self])

286      def attach(self, filename=None, content=None, mimetype=None):
287          """ 
288          Attach a file with the given filename and content. The filename can 
289          be omitted and the mimetype is guessed, if not provided. 
290   
291          If the first parameter is a MIMEBase subclass, insert it directly 
292          into the resulting message attachments. 
293   
294          For a text/* mimetype (guessed or specified), when a bytes object is 
295          specified as content, decode it as UTF-8. If that fails, set the 
296          mimetype to DEFAULT_ATTACHMENT_MIME_TYPE and don't decode the content. 
297          """
298          if isinstance(filename, MIMEBase):
299              assert content is None
300              assert mimetype is None
301              self.attachments.append(filename)
302          else:
303              assert content is not None
304              mimetype = mimetype or mimetypes.guess_type(filename)[0] or DEFAULT_ATTACHMENT_MIME_TYPE
305              basetype, subtype = mimetype.split('/', 1)
306  
307              if basetype == 'text':
308                  if isinstance(content, bytes):
309                      try:
310                          content = content.decode()
311                      except UnicodeDecodeError:
312                          # If mimetype suggests the file is text but it's
313                          # actually binary, read() raises a UnicodeDecodeError.
314                          mimetype = DEFAULT_ATTACHMENT_MIME_TYPE
315  
316              self.attachments.append((filename, content, mimetype))
317  
318      def attach_file(self, path, mimetype=None):
319          """ 
320          Attach a file from the filesystem. 
321   
322          Set the mimetype to DEFAULT_ATTACHMENT_MIME_TYPE if it isn't specified 
323          and cannot be guessed. 
324   
325          For a text/* mimetype (guessed or specified), decode the file's content 
326          as UTF-8. If that fails, set the mimetype to 
327          DEFAULT_ATTACHMENT_MIME_TYPE and don't decode the content. 
328          """
329          path = Path(path)
330          with path.open('rb') as file:
331              content = file.read()
332              self.attach(path.name, content, mimetype)
333  
334      def _create_message(self, msg):
335          return self._create_attachments(msg)
336  
337      def _create_attachments(self, msg):
338          if self.attachments:
339              encoding = self.encoding or settings.DEFAULT_CHARSET
340              body_msg = msg
341              msg = SafeMIMEMultipart(_subtype=self.mixed_subtype, encoding=encoding)
342              if self.body or body_msg.is_multipart():
343                  msg.attach(body_msg)
344              for attachment in self.attachments:
345                  if isinstance(attachment, MIMEBase):
346                      msg.attach(attachment)
347                  else:
348                      msg.attach(self._create_attachment(*attachment))
349          return msg
350  
351      def _create_mime_attachment(self, content, mimetype):
352          """ 
353          Convert the content, mimetype pair into a MIME attachment object. 
354   
355          If the mimetype is message/rfc822, content may be an 
356          email.Message or EmailMessage object, as well as a str. 
357          """
358          basetype, subtype = mimetype.split('/', 1)
359          if basetype == 'text':
360              encoding = self.encoding or settings.DEFAULT_CHARSET
361              attachment = SafeMIMEText(content, subtype, encoding)
362          elif basetype == 'message' and subtype == 'rfc822':
363              # Bug #18967: per RFC2046 s5.2.1, message/rfc822 attachments
364              # must not be base64 encoded.
365              if isinstance(content, EmailMessage):
366                  # convert content into an email.Message first
367                  content = content.message()
368              elif not isinstance(content, Message):
369                  # For compatibility with existing code, parse the message
370                  # into an email.Message object if it is not one already.
371                  content = message_from_string(force_str(content))
372  
373              attachment = SafeMIMEMessage(content, subtype)
374          else:
375              # Encode non-text attachments with base64.
376              attachment = MIMEBase(basetype, subtype)
377              attachment.set_payload(content)
378              Encoders.encode_base64(attachment)
379          return attachment
380  
381      def _create_attachment(self, filename, content, mimetype=None):
382          """ 
383          Convert the filename, content, mimetype triple into a MIME attachment 
384          object. 
385          """
386          attachment = self._create_mime_attachment(content, mimetype)
387          if filename:
388              try:
389                  filename.encode('ascii')
390              except UnicodeEncodeError:
391                  filename = ('utf-8', '', filename)
392              attachment.add_header('Content-Disposition', 'attachment', filename=filename)
393          return attachment
394  
395      def _set_list_header_if_not_empty(self, msg, header, values):
396          """ 
397          Set msg's header, either from self.extra_headers, if present, or from 
398          the values argument. 
399          """
400          if values:
401              try:
402                  value = self.extra_headers[header]
403              except KeyError:
404                  value = ', '.join(str(v) for v in values)
405              msg[header] = value

408  class EmailMultiAlternatives(EmailMessage):
409      """ 
410      A version of EmailMessage that makes it easy to send multipart/alternative 
411      messages. For example, including text and HTML versions of the text is 
412      made easier. 
413      """
414      alternative_subtype = 'alternative'
415  
416      def __init__(self, subject='', body='', from_email=None, to=None, bcc=None,
417                   connection=None, attachments=None, headers=None, alternatives=None,
418                   cc=None, reply_to=None):
419          """ 
420          Initialize a single email message (which can be sent to multiple 
421          recipients). 
422          """
423          super().__init__(
424              subject, body, from_email, to, bcc, connection, attachments,
425              headers, cc, reply_to,
426          )
427          self.alternatives = alternatives or []
428  
429      def attach_alternative(self, content, mimetype):
430          """Attach an alternative content representation."""
431          assert content is not None
432          assert mimetype is not None
433          self.alternatives.append((content, mimetype))
434  
435      def _create_message(self, msg):
436          return self._create_attachments(self._create_alternatives(msg))
437  
438      def _create_alternatives(self, msg):
439          encoding = self.encoding or settings.DEFAULT_CHARSET
440          if self.alternatives:
441              body_msg = msg
442              msg = SafeMIMEMultipart(_subtype=self.alternative_subtype, encoding=encoding)
443              if self.body:
444                  msg.attach(body_msg)
445              for alternative in self.alternatives:
446                  msg.attach(self._create_mime_attachment(*alternative))
447          return msg
