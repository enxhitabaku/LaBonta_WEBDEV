1    """ 
2     The OGRGeometry is a wrapper for using the OGR Geometry class 
3     (see https://www.gdal.org/classOGRGeometry.html).  OGRGeometry 
4     may be instantiated when reading geometries from OGR Data Sources 
5     (e.g. SHP files), or when given OGC WKT (a string). 
6     
7     While the 'full' API is not present yet, the API is "pythonic" unlike 
8     the traditional and "next-generation" OGR Python bindings.  One major 
9     advantage OGR Geometries have over their GEOS counterparts is support 
10    for spatial reference systems and their transformation. 
11    
12    Example: 
13     >>> from django.contrib.gis.gdal import OGRGeometry, OGRGeomType, SpatialReference 
14     >>> wkt1, wkt2 = 'POINT(-90 30)', 'POLYGON((0 0, 5 0, 5 5, 0 5)' 
15     >>> pnt = OGRGeometry(wkt1) 
16     >>> print(pnt) 
17     POINT (-90 30) 
18     >>> mpnt = OGRGeometry(OGRGeomType('MultiPoint'), SpatialReference('WGS84')) 
19     >>> mpnt.add(wkt1) 
20     >>> mpnt.add(wkt1) 
21     >>> print(mpnt) 
22     MULTIPOINT (-90 30,-90 30) 
23     >>> print(mpnt.srs.name) 
24     WGS 84 
25     >>> print(mpnt.srs.proj) 
26     +proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs 
27     >>> mpnt.transform(SpatialReference('NAD27')) 
28     >>> print(mpnt.proj) 
29     +proj=longlat +ellps=clrk66 +datum=NAD27 +no_defs 
30     >>> print(mpnt) 
31     MULTIPOINT (-89.999930378602485 29.999797886557641,-89.999930378602485 29.999797886557641) 
32    
33     The OGRGeomType class is to make it easy to specify an OGR geometry type: 
34     >>> from django.contrib.gis.gdal import OGRGeomType 
35     >>> gt1 = OGRGeomType(3) # Using an integer for the type 
36     >>> gt2 = OGRGeomType('Polygon') # Using a string 
37     >>> gt3 = OGRGeomType('POLYGON') # It's case-insensitive 
38     >>> print(gt1 == 3, gt1 == 'Polygon') # Equivalence works w/non-OGRGeomType objects 
39     True True 
40   """

55   # For more information, see the OGR C API source code:
56   #  https://www.gdal.org/ogr__api_8h.html

58   # The OGR_G_* routines are relevant here.
59   class OGRGeometry(GDALBase):
60       """Encapsulate an OGR geometry."""
61       destructor = capi.destroy_geom
62   
63       def __init__(self, geom_input, srs=None):
64           """Initialize Geometry on either WKT or an OGR pointer as input."""
65           str_instance = isinstance(geom_input, str)
66   
67           # If HEX, unpack input to a binary buffer.
68           if str_instance and hex_regex.match(geom_input):
69               geom_input = memoryview(bytes.fromhex(geom_input))
70               str_instance = False
71   
72           # Constructing the geometry,
73           if str_instance:
74               wkt_m = wkt_regex.match(geom_input)
75               json_m = json_regex.match(geom_input)
76               if wkt_m:
77                   if wkt_m['srid']:
78                       # If there's EWKT, set the SRS w/value of the SRID.
79                       srs = int(wkt_m['srid'])
80                   if wkt_m['type'].upper() == 'LINEARRING':
81                       # OGR_G_CreateFromWkt doesn't work with LINEARRING WKT.
82                       #  See https://trac.osgeo.org/gdal/ticket/1992.
83                       g = capi.create_geom(OGRGeomType(wkt_m['type']).num)
84                       capi.import_wkt(g, byref(c_char_p(wkt_m['wkt'].encode())))
85                   else:
86                       g = capi.from_wkt(byref(c_char_p(wkt_m['wkt'].encode())), None, byref(c_void_p()))
87               elif json_m:
88                   g = self._from_json(geom_input.encode())
89               else:
90                   # Seeing if the input is a valid short-hand string
91                   # (e.g., 'Point', 'POLYGON').
92                   OGRGeomType(geom_input)
93                   g = capi.create_geom(OGRGeomType(geom_input).num)
94           elif isinstance(geom_input, memoryview):
95               # WKB was passed in
96               g = self._from_wkb(geom_input)
97           elif isinstance(geom_input, OGRGeomType):
98               # OGRGeomType was passed in, an empty geometry will be created.
99               g = capi.create_geom(geom_input.num)
100          elif isinstance(geom_input, self.ptr_type):
101              # OGR pointer (c_void_p) was the input.
102              g = geom_input
103          else:
104              raise GDALException('Invalid input type for OGR Geometry construction: %s' % type(geom_input))
105  
106          # Now checking the Geometry pointer before finishing initialization
107          # by setting the pointer for the object.
108          if not g:
109              raise GDALException('Cannot create OGR Geometry from input: %s' % geom_input)
110          self.ptr = g
111  
112          # Assigning the SpatialReference object to the geometry, if valid.
113          if srs:
114              self.srs = srs
115  
116          # Setting the class depending upon the OGR Geometry Type
117          self.__class__ = GEO_CLASSES[self.geom_type.num]
118  
119      # Pickle routines
120      def __getstate__(self):
121          srs = self.srs
122          if srs:
123              srs = srs.wkt
124          else:
125              srs = None
126          return bytes(self.wkb), srs
127  
128      def __setstate__(self, state):
129          wkb, srs = state
130          ptr = capi.from_wkb(wkb, None, byref(c_void_p()), len(wkb))
131          if not ptr:
132              raise GDALException('Invalid OGRGeometry loaded from pickled state.')
133          self.ptr = ptr
134          self.srs = srs

258      # The SRS property
259      def _get_srs(self):
260          "Return the Spatial Reference for this Geometry."
261          try:
262              srs_ptr = capi.get_geom_srs(self.ptr)
263              return SpatialReference(srs_api.clone_srs(srs_ptr))
264          except SRSException:
265              return None
266  
267      def _set_srs(self, srs):
268          "Set the SpatialReference for this geometry."
269          # Do not have to clone the `SpatialReference` object pointer because
270          # when it is assigned to this `OGRGeometry` it's internal OGR
271          # reference count is incremented, and will likewise be released
272          # (decremented) when this geometry's destructor is called.
273          if isinstance(srs, SpatialReference):
274              srs_ptr = srs.ptr
275          elif isinstance(srs, (int, str)):
276              sr = SpatialReference(srs)
277              srs_ptr = sr.ptr
278          elif srs is None:
279              srs_ptr = None
280          else:
281              raise TypeError('Cannot assign spatial reference with object of type: %s' % type(srs))
282          capi.assign_srs(self.ptr, srs_ptr)
283

340      @property
341      def wkb(self):
342          "Return the WKB representation of the Geometry."
343          if sys.byteorder == 'little':
344              byteorder = 1  # wkbNDR (from ogr_core.h)
345          else:
346              byteorder = 0  # wkbXDR
347          sz = self.wkb_size
348          # Creating the unsigned character buffer, and passing it in by reference.
349          buf = (c_ubyte * sz)()
350          capi.to_wkb(self.ptr, byteorder, byref(buf))
351          # Returning a buffer of the string at the pointer.
352          return memoryview(string_at(buf, sz))

382      def transform(self, coord_trans, clone=False):
383          """ 
384          Transform this geometry to a different spatial reference system. 
385          May take a CoordTransform object, a SpatialReference object, string 
386          WKT or PROJ.4, and/or an integer SRID.  By default, return nothing 
387          and transform the geometry in-place. However, if the `clone` keyword is 
388          set, return a transformed clone of this geometry. 
389          """
390          if clone:
391              klone = self.clone()
392              klone.transform(coord_trans)
393              return klone
394  
395          # Depending on the input type, use the appropriate OGR routine
396          # to perform the transformation.
397          if isinstance(coord_trans, CoordTransform):
398              capi.geom_transform(self.ptr, coord_trans.ptr)
399          elif isinstance(coord_trans, SpatialReference):
400              capi.geom_transform_to(self.ptr, coord_trans.ptr)
401          elif isinstance(coord_trans, (int, str)):
402              sr = SpatialReference(coord_trans)
403              capi.geom_transform_to(self.ptr, sr.ptr)
404          else:
405              raise TypeError('Transform only accepts CoordTransform, '
406                              'SpatialReference, string, and integer objects.')

408      # #### Topology Methods ####
409      def _topology(self, func, other):
410          """A generalized function for topology operations, takes a GDAL function and 
411          the other geometry to perform the operation on."""
412          if not isinstance(other, OGRGeometry):
413              raise TypeError('Must use another OGRGeometry object for topology operations!')
414  
415          # Returning the output of the given function with the other geometry's
416          # pointer.
417          return func(self.ptr, other.ptr)

538  class LineString(OGRGeometry):
539  
540      def __getitem__(self, index):
541          "Return the Point at the given index."
542          if 0 <= index < self.point_count:
543              x, y, z = c_double(), c_double(), c_double()
544              capi.get_point(self.ptr, index, byref(x), byref(y), byref(z))
545              dim = self.coord_dim
546              if dim == 1:
547                  return (x.value,)
548              elif dim == 2:
549                  return (x.value, y.value)
550              elif dim == 3:
551                  return (x.value, y.value, z.value)
552          else:
553              raise IndexError('Index out of range when accessing points of a line string: %s.' % index)
554  
555      def __len__(self):
556          "Return the number of points in the LineString."
557          return self.point_count
558  
559      @property
560      def tuple(self):
561          "Return the tuple representation of this LineString."
562          return tuple(self[i] for i in range(len(self)))
563      coords = tuple
564  
565      def _listarr(self, func):
566          """ 
567          Internal routine that returns a sequence (list) corresponding with 
568          the given function. 
569          """
570          return [func(self.ptr, i) for i in range(len(self))]

647          "Return the number of geometries in this Geometry Collection."
648          return self.geom_count

650      def add(self, geom):
651          "Add the geometry to this Geometry Collection."
652          if isinstance(geom, OGRGeometry):
653              if isinstance(geom, self.__class__):
654                  for g in geom:
655                      capi.add_geom(self.ptr, g.ptr)
656              else:
657                  capi.add_geom(self.ptr, geom.ptr)
658          elif isinstance(geom, str):
659              tmp = OGRGeometry(geom)
660              capi.add_geom(self.ptr, tmp.ptr)
661          else:
662              raise GDALException('Must add an OGRGeometry.')

666          "Return the number of Points in this Geometry Collection."
667          # Summing up the number of points in each geometry in this collection
668          return sum(self[i].point_count for i in range(self.geom_count))
