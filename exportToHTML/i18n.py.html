37           # ``language`` is either a language code string or a sequence

68   class TranslateNode(Node):
69       def __init__(self, filter_expression, noop, asvar=None,
70                    message_context=None):
71           self.noop = noop
72           self.asvar = asvar
73           self.message_context = message_context
74           self.filter_expression = filter_expression
75           if isinstance(self.filter_expression.var, str):
76               self.filter_expression.var = Variable("'%s'" %
77                                                     self.filter_expression.var)
78   
79       def render(self, context):
80           self.filter_expression.var.translate = not self.noop
81           if self.message_context:
82               self.filter_expression.var.message_context = (
83                   self.message_context.resolve(context))
84           output = self.filter_expression.resolve(context)
85           value = render_value_in_context(output, context)
86           # Restore percent signs. Percent signs in template text are doubled
87           # so they are not interpreted as string format flags.
88           is_safe = isinstance(value, SafeData)
89           value = value.replace('%%', '%')
90           value = mark_safe(value) if is_safe else value
91           if self.asvar:
92               context[self.asvar] = value
93               return ''
94           else:
95               return value

98   class BlockTranslateNode(Node):
99   
100      def __init__(self, extra_context, singular, plural=None, countervar=None,
101                   counter=None, message_context=None, trimmed=False, asvar=None,
102                   tag_name='blocktranslate'):
103          self.extra_context = extra_context
104          self.singular = singular
105          self.plural = plural
106          self.countervar = countervar
107          self.counter = counter
108          self.message_context = message_context
109          self.trimmed = trimmed
110          self.asvar = asvar
111          self.tag_name = tag_name
112  
113      def render_token_list(self, tokens):
114          result = []
115          vars = []
116          for token in tokens:
117              if token.token_type == TokenType.TEXT:
118                  result.append(token.contents.replace('%', '%%'))
119              elif token.token_type == TokenType.VAR:
120                  result.append('%%(%s)s' % token.contents)
121                  vars.append(token.contents)
122          msg = ''.join(result)
123          if self.trimmed:
124              msg = translation.trim_whitespace(msg)
125          return msg, vars
126  
127      def render(self, context, nested=False):
128          if self.message_context:
129              message_context = self.message_context.resolve(context)
130          else:
131              message_context = None
132          # Update() works like a push(), so corresponding context.pop() is at
133          # the end of function
134          context.update({var: val.resolve(context) for var, val in self.extra_context.items()})
135          singular, vars = self.render_token_list(self.singular)
136          if self.plural and self.countervar and self.counter:
137              count = self.counter.resolve(context)
138              context[self.countervar] = count
139              plural, plural_vars = self.render_token_list(self.plural)
140              if message_context:
141                  result = translation.npgettext(message_context, singular,
142                                                 plural, count)
143              else:
144                  result = translation.ngettext(singular, plural, count)
145              vars.extend(plural_vars)
146          else:
147              if message_context:
148                  result = translation.pgettext(message_context, singular)
149              else:
150                  result = translation.gettext(singular)
151          default_value = context.template.engine.string_if_invalid
152  
153          def render_value(key):
154              if key in context:
155                  val = context[key]
156              else:
157                  val = default_value % key if '%s' in default_value else default_value
158              return render_value_in_context(val, context)
159  
160          data = {v: render_value(v) for v in vars}
161          context.pop()
162          try:
163              result = result % data
164          except (KeyError, ValueError):
165              if nested:
166                  # Either string is malformed, or it's a bug
167                  raise TemplateSyntaxError(
168                      '%r is unable to format string returned by gettext: %r '
169                      'using %r' % (self.tag_name, result, data)
170                  )
171              with translation.override(None):
172                  result = self.render(context, nested=True)
173          if self.asvar:
174              context[self.asvar] = result
175              return ''
176          else:
177              return result

180  class LanguageNode(Node):
181      def __init__(self, nodelist, language):
182          self.nodelist = nodelist
183          self.language = language
184  
185      def render(self, context):
186          with translation.override(self.language.resolve(context)):
187              output = self.nodelist.render(context)
188          return output

191  @register.tag("get_available_languages")
192  def do_get_available_languages(parser, token):
193      """ 
194      Store a list of available languages in the context. 
195   
196      Usage:: 
197   
198          {% get_available_languages as languages %} 
199          {% for language in languages %} 
200          ... 
201          {% endfor %} 
202   
203      This puts settings.LANGUAGES into the named variable. 
204      """
205      # token.split_contents() isn't useful here because this tag doesn't accept variable as arguments
206      args = token.contents.split()
207      if len(args) != 3 or args[1] != 'as':
208          raise TemplateSyntaxError("'get_available_languages' requires 'as variable' (got %r)" % args)
209      return GetAvailableLanguagesNode(args[2])

318  @register.tag("translate")
319  @register.tag("trans")
320  def do_translate(parser, token):
321      """ 
322      Mark a string for translation and translate the string for the current 
323      language. 
324   
325      Usage:: 
326   
327          {% translate "this is a test" %} 
328   
329      This marks the string for translation so it will be pulled out by 
330      makemessages into the .po files and runs the string through the translation 
331      engine. 
332   
333      There is a second form:: 
334   
335          {% translate "this is a test" noop %} 
336   
337      This marks the string for translation, but returns the string unchanged. 
338      Use it when you need to store values into forms that should be translated 
339      later on. 
340   
341      You can use variables instead of constant strings 
342      to translate stuff you marked somewhere else:: 
343   
344          {% translate variable %} 
345   
346      This tries to translate the contents of the variable ``variable``. Make 
347      sure that the string in there is something that is in the .po file. 
348   
349      It is possible to store the translated string into a variable:: 
350   
351          {% translate "this is a test" as var %} 
352          {{ var }} 
353   
354      Contextual translations are also supported:: 
355   
356          {% translate "this is a test" context "greeting" %} 
357   
358      This is equivalent to calling pgettext instead of (u)gettext. 
359      """
360      bits = token.split_contents()
361      if len(bits) < 2:
362          raise TemplateSyntaxError("'%s' takes at least one argument" % bits[0])
363      message_string = parser.compile_filter(bits[1])
364      remaining = bits[2:]
365  
366      noop = False
367      asvar = None
368      message_context = None
369      seen = set()
370      invalid_context = {'as', 'noop'}
371  
372      while remaining:
373          option = remaining.pop(0)
374          if option in seen:
375              raise TemplateSyntaxError(
376                  "The '%s' option was specified more than once." % option,
377              )
378          elif option == 'noop':
379              noop = True
380          elif option == 'context':
381              try:
382                  value = remaining.pop(0)
383              except IndexError:
384                  raise TemplateSyntaxError(
385                      "No argument provided to the '%s' tag for the context option." % bits[0]
386                  )
387              if value in invalid_context:
388                  raise TemplateSyntaxError(
389                      "Invalid argument '%s' provided to the '%s' tag for the context option" % (value, bits[0]),
390                  )
391              message_context = parser.compile_filter(value)
392          elif option == 'as':
393              try:
394                  value = remaining.pop(0)
395              except IndexError:
396                  raise TemplateSyntaxError(
397                      "No argument provided to the '%s' tag for the as option." % bits[0]
398                  )
399              asvar = value
400          else:
401              raise TemplateSyntaxError(
402                  "Unknown argument for '%s' tag: '%s'. The only options "
403                  "available are 'noop', 'context' \"xxx\", and 'as VAR'." % (
404                      bits[0], option,
405                  )
406              )
407          seen.add(option)
408  
409      return TranslateNode(message_string, noop, asvar, message_context)

412  @register.tag("blocktranslate")
413  @register.tag("blocktrans")
414  def do_block_translate(parser, token):
415      """ 
416      Translate a block of text with parameters. 
417   
418      Usage:: 
419   
420          {% blocktranslate with bar=foo|filter boo=baz|filter %} 
421          This is {{ bar }} and {{ boo }}. 
422          {% endblocktranslate %} 
423   
424      Additionally, this supports pluralization:: 
425   
426          {% blocktranslate count count=var|length %} 
427          There is {{ count }} object. 
428          {% plural %} 
429          There are {{ count }} objects. 
430          {% endblocktranslate %} 
431   
432      This is much like ngettext, only in template syntax. 
433   
434      The "var as value" legacy format is still supported:: 
435   
436          {% blocktranslate with foo|filter as bar and baz|filter as boo %} 
437          {% blocktranslate count var|length as count %} 
438   
439      The translated string can be stored in a variable using `asvar`:: 
440   
441          {% blocktranslate with bar=foo|filter boo=baz|filter asvar var %} 
442          This is {{ bar }} and {{ boo }}. 
443          {% endblocktranslate %} 
444          {{ var }} 
445   
446      Contextual translations are supported:: 
447   
448          {% blocktranslate with bar=foo|filter context "greeting" %} 
449              This is {{ bar }}. 
450          {% endblocktranslate %} 
451   
452      This is equivalent to calling pgettext/npgettext instead of 
453      (u)gettext/(u)ngettext. 
454      """
455      bits = token.split_contents()
456  
457      options = {}
458      remaining_bits = bits[1:]
459      asvar = None
460      while remaining_bits:
461          option = remaining_bits.pop(0)
462          if option in options:
463              raise TemplateSyntaxError('The %r option was specified more '
464                                        'than once.' % option)
465          if option == 'with':
466              value = token_kwargs(remaining_bits, parser, support_legacy=True)
467              if not value:
468                  raise TemplateSyntaxError('"with" in %r tag needs at least '
469                                            'one keyword argument.' % bits[0])
470          elif option == 'count':
471              value = token_kwargs(remaining_bits, parser, support_legacy=True)
472              if len(value) != 1:
473                  raise TemplateSyntaxError('"count" in %r tag expected exactly '
474                                            'one keyword argument.' % bits[0])
475          elif option == "context":
476              try:
477                  value = remaining_bits.pop(0)
478                  value = parser.compile_filter(value)
479              except Exception:
480                  raise TemplateSyntaxError(
481                      '"context" in %r tag expected exactly one argument.' % bits[0]
482                  )
483          elif option == "trimmed":
484              value = True
485          elif option == "asvar":
486              try:
487                  value = remaining_bits.pop(0)
488              except IndexError:
489                  raise TemplateSyntaxError(
490                      "No argument provided to the '%s' tag for the asvar option." % bits[0]
491                  )
492              asvar = value
493          else:
494              raise TemplateSyntaxError('Unknown argument for %r tag: %r.' %
495                                        (bits[0], option))
496          options[option] = value
497  
498      if 'count' in options:
499          countervar, counter = next(iter(options['count'].items()))
500      else:
501          countervar, counter = None, None
502      if 'context' in options:
503          message_context = options['context']
504      else:
505          message_context = None
506      extra_context = options.get('with', {})
507  
508      trimmed = options.get("trimmed", False)
509  
510      singular = []
511      plural = []
512      while parser.tokens:
513          token = parser.next_token()
514          if token.token_type in (TokenType.VAR, TokenType.TEXT):
515              singular.append(token)
516          else:
517              break
518      if countervar and counter:
519          if token.contents.strip() != 'plural':
520              raise TemplateSyntaxError("%r doesn't allow other block tags inside it" % bits[0])
521          while parser.tokens:
522              token = parser.next_token()
523              if token.token_type in (TokenType.VAR, TokenType.TEXT):
524                  plural.append(token)
525              else:
526                  break
527      end_tag_name = 'end%s' % bits[0]
528      if token.contents.strip() != end_tag_name:
529          raise TemplateSyntaxError("%r doesn't allow other block tags (seen %r) inside it" % (bits[0], token.contents))
530  
531      return BlockTranslateNode(extra_context, singular, plural, countervar,
532                                counter, message_context, trimmed=trimmed,
533                                asvar=asvar, tag_name=bits[0])

536  @register.tag
537  def language(parser, token):
538      """ 
539      Enable the given language just for this block. 
540   
541      Usage:: 
542   
543          {% language "de" %} 
544              This is {{ bar }} and {{ boo }}. 
545          {% endlanguage %} 
546      """
547      bits = token.split_contents()
548      if len(bits) != 2:
549          raise TemplateSyntaxError("'%s' takes one argument (language)" % bits[0])
550      language = parser.compile_filter(bits[1])
551      nodelist = parser.parse(('endlanguage',))
552      parser.delete_first_token()
553      return LanguageNode(nodelist, language)
